26 // SparseColumn class. In addition to index(), it also provides row() for better
27 // readability on the client side.

60 // RandomAccessSparseColumn
61 // --------------------------------------------------------
62 // A RandomAccessSparseColumn is a mix between a DenseColumn and a SparseColumn.
63 // It makes it possible to populate a dense column from a sparse column in
64 // O(num_entries) instead of O(num_rows), and to access an entry in O(1).
65 // As the constructor runs in O(num_rows), a RandomAccessSparseColumn should be
66 // used several times to amortize the creation cost.
67 class RandomAccessSparseColumn {
68  public:
69  // Creates a RandomAccessSparseColumn.
70  // Runs in O(num_rows).
71  explicit RandomAccessSparseColumn(RowIndex num_rows);
72  virtual ~RandomAccessSparseColumn();
73 
74  // Clears the column.
75  // Runs in O(num_entries).
76  void Clear();
77 
78  void Resize(RowIndex num_rows);
79 
80  // Sets value at row.
81  // Runs in O(1).
82  void SetCoefficient(RowIndex row, Fractional value) {
83  column_[row] = value;
84  MarkRowAsChanged(row);
85  }
86 
87  // Adds value to the current value at row.
88  // Runs in O(1).
89  void AddToCoefficient(RowIndex row, Fractional value) {
90  column_[row] += value;
91  MarkRowAsChanged(row);
92  }
93 
94  // Populates from a sparse column.
95  // Runs in O(num_entries).
96  void PopulateFromSparseColumn(const SparseColumn& sparse_column);
97 
98  // Populates a sparse column from the lazy dense column.
99  // Runs in O(num_entries).
100  void PopulateSparseColumn(SparseColumn* sparse_column) const;
101 
102  // Returns the number of rows.
103  // Runs in O(1).
104  RowIndex GetNumberOfRows() const { return RowIndex(column_.size()); }
105 
106  // Returns the value in position row.
107  // Runs in O(1).
108  Fractional GetCoefficient(RowIndex row) const { return column_[row]; }
109 
110  private:
111  // Keeps a trace of which rows have been changed.
112  void MarkRowAsChanged(RowIndex row) {
113  if (!changed_[row]) {
114  changed_[row] = true;
115  row_change_.push_back(row);
116  }
117  }
118 
119  // The dense version of the column.
120  DenseColumn column_;
121 
122  // Dense Boolean vector used to mark changes.
123  DenseBooleanColumn changed_;
124 
125  // Stack to store changes.
126  std::vector<RowIndex> row_change_;
127 
128  DISALLOW_COPY_AND_ASSIGN(RandomAccessSparseColumn);
129 };
