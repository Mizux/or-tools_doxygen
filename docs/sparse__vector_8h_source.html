54 // --------------------------------------------------------
55 // SparseVector
56 // --------------------------------------------------------
57 // This class allows to store a vector taking advantage of its sparsity.
58 // Space complexity is in O(num_entries).
59 // In the current implementation, entries are stored in a first-in order (order
60 // of SetCoefficient() calls) when they are added; then the "cleaning" process
61 // sorts them by index (and duplicates are removed: the last entry takes
62 // precedence).
63 // Many methods assume that the entries are sorted by index and without
64 // duplicates, and DCHECK() that.
65 //
66 // Default copy construction is fully supported.
67 //
68 // This class uses strong integer types (i.e. no implicit cast to/from other
69 // integer types) for both:
70 // - the index of entries (eg. SparseVector<RowIndex> is a SparseColumn,
71 // see ./sparse_column.h).
72 // - the *internal* indices of entries in the internal storage, which is an
73 // entirely different type: EntryType.
74 // This class can be extended with a custom iterator/entry type for the
75 // iterator-based API. This can be used to extend the interface with additional
76 // methods for the entries returned by the iterators; for an example of such
77 // extension, see SparseColumnEntry in sparse_column.h. The custom entries and
78 // iterators should be derived from SparseVectorEntry and SparseVectorIterator,
79 // or at least provide the same public and protected interface.
80 //
81 // TODO(user): un-expose this type to client; by getting rid of the
82 // index-based APIs and leveraging iterator-based APIs; if possible.
83 template <typename IndexType,
84  typename IteratorType =
85  SparseVectorIterator<SparseVectorEntry<IndexType>>>

317  protected:
318  // Adds a new entry to the sparse vector, growing the internal buffer if
319  // needed. It does not set may_contain_duplicates_ to true.
320  void AddEntry(Index index, Fractional value) {
321  DCHECK_GE(index, 0);
322  // Grow the internal storage if there is no space left for the new entry. We
323  // increase the size to max(4, 1.5*current capacity).
324  if (num_entries_ == capacity_) {
325  // Reserve(capacity_ == 0 ? EntryIndex(4)
326  // : EntryIndex(2 * capacity_.value()));
327  Reserve(capacity_ == 0 ? EntryIndex(4)
328  : EntryIndex(2 * capacity_.value()));
329  DCHECK_LT(num_entries_, capacity_);
330  }
331  const EntryIndex new_entry_index = num_entries_;
332  ++num_entries_;
333  MutableIndex(new_entry_index) = index;
334  MutableCoefficient(new_entry_index) = value;
335  }
336 
337  // Resizes the sparse vector to a smaller size, without re-allocating the
338  // internal storage.
339  void ResizeDown(EntryIndex new_size) {
340  DCHECK_GE(new_size, 0);
341  DCHECK_LE(new_size, num_entries_);
342  num_entries_ = new_size;
343  }
344 
345  // Read-only access to the indices and coefficients of the entries of the
346  // sparse vector.
347  Index GetIndex(EntryIndex i) const {
348  DCHECK_GE(i, 0);
349  DCHECK_LT(i, num_entries_);
350  return index_[i.value()];
351  }
352  Fractional GetCoefficient(EntryIndex i) const {
353  DCHECK_GE(i, 0);
354  DCHECK_LT(i, num_entries_);
355  return coefficient_[i.value()];
356  }
357 
358  // Mutable access to the indices and coefficients of the entries of the sparse
359  // vector.
360  Index& MutableIndex(EntryIndex i) {
361  DCHECK_GE(i, 0);
362  DCHECK_LT(i, num_entries_);
363  return index_[i.value()];
364  }
365  Fractional& MutableCoefficient(EntryIndex i) {
366  DCHECK_GE(i, 0);
367  DCHECK_LT(i, num_entries_);
368  return coefficient_[i.value()];
369  }
370 
371  // The internal storage of the sparse vector. Both the indices and the
372  // coefficients are stored in the same buffer; the first
373  // sizeof(Index)*capacity_ bytes are used for storing the indices, the
374  // following sizeof(Fractional)*capacity_ bytes contain the values. This
375  // representation ensures that for small vectors, both the indices and the
376  // coefficients are in the same page/cache line.
377  // We use a single buffer for both arrays. The amount of data copied during
378  // relocations is the same in both cases, and it is much smaller than the cost
379  // of an additional allocation - especially when the vectors are small.
380  // Moreover, using two separate vectors/buffers would mean that even small
381  // vectors would be spread across at least two different cache lines.
382  std::unique_ptr<char[]> buffer_;
383  EntryIndex num_entries_;
384  EntryIndex capacity_;
385 
386  // Pointers to the first elements of the index and coefficient arrays.
387  Index* index_;
388  Fractional* coefficient_;
389 
390  // This is here to speed up the CheckNoDuplicates() methods and is mutable
391  // so we can perform checks on const argument.
392  mutable bool may_contain_duplicates_;
393 
394  private:
395  // Actual implementation of AddMultipleToSparseVectorAndDeleteCommonIndex()
396  // and AddMultipleToSparseVectorAndIgnoreCommonIndex() which is shared.
397  void AddMultipleToSparseVectorInternal(
398  bool delete_common_index, Fractional multiplier, Index common_index,
399  Fractional drop_tolerance, SparseVector* accumulator_vector) const;
400 };

402 // --------------------------------------------------------
403 // SparseVectorEntry
404 // --------------------------------------------------------
405 
406 // A reference-like class that points to a certain element of a sparse data
407 // structure that stores its elements in two parallel arrays. The main purpose
408 // of the entry class is to support implementation of iterator objects over the
409 // sparse data structure.
410 // Note that the entry object does not own the data, and it is valid only as
411 // long as the underlying sparse data structure; it may also be invalidated if
412 // the underlying sparse data structure is modified.
413 template <typename IndexType>
414 class SparseVectorEntry {
415  public:
416  using Index = IndexType;
417 
418  Index index() const { return index_[i_.value()]; }
419  Fractional coefficient() const { return coefficient_[i_.value()]; }
420 
421  protected:
422  // Creates the sparse vector entry from the given base pointers and the index.
423  // We accept the low-level data structures rather than a SparseVector
424  // reference to make it possible to use the SparseVectorEntry and
425  // SparseVectorIterator classes also for other data structures using the same
426  // internal data representation.
427  // Note that the constructor is intentionally made protected, so that the
428  // entry can be created only as a part of the construction of an iterator over
429  // a sparse data structure.
430  SparseVectorEntry(const Index* indices, const Fractional* coefficients,
431  EntryIndex i)
432  : i_(i), index_(indices), coefficient_(coefficients) {}
433 
434  // The index of the sparse vector entry represented by this object.
435  EntryIndex i_;
436  // The index and coefficient arrays of the sparse vector.
437  // NOTE(user): Keeping directly the index and the base pointers gives the
438  // best performance with a tiny margin of the options:
439  // 1. keep the base pointers and an index of the current entry,
440  // 2. keep pointers to the current index and the current coefficient and
441  // increment both when moving the iterator.
442  // 3. keep a pointer to the sparse vector object and the index of the current
443  // entry.
444  const Index* index_;
445  const Fractional* coefficient_;
446 };

448 // --------------------------------------------------------
449 // SparseVectorIterator
450 // --------------------------------------------------------
451 
452 // An iterator over the elements of a sparse data structure that stores the
453 // elements in parallel arrays for indices and coefficients. The iterator is
454 // built as a wrapper over a sparse vector entry class; the concrete entry class
455 // is provided through the template argument EntryType and it must either be
456 // derived from SparseVectorEntry or it must provide the same public and
457 // protected interface.
458 template <typename EntryType>
459 class SparseVectorIterator : EntryType {
460  public:
461  using Index = typename EntryType::Index;
462  using Entry = EntryType;
463 
464  SparseVectorIterator(const Index* indices, const Fractional* coefficients,
465  EntryIndex i)
466  : EntryType(indices, coefficients, i) {}
467 
468  void operator++() { ++this->i_; }
469  bool operator!=(const SparseVectorIterator& other) const {
470  // This operator is intended for use in natural range iteration ONLY.
471  // Therefore, we prefer to use '<' so that a buggy range iteration which
472  // start point is *after* its end point stops immediately, instead of
473  // iterating 2^(number of bits of EntryIndex) times.
474  return this->i_ < other.i_;
475  }
476  const Entry& operator*() const { return *this; }
477 };

528 template <typename IndexType, typename IteratorType>
529 void SparseVector<IndexType, IteratorType>::Reserve(EntryIndex new_capacity) {
530  if (new_capacity <= capacity_) return;
531  // Round up the capacity to a multiple of four. This way, the start of the
532  // coefficient array will be aligned to 16-bytes, provided that the buffer
533  // used for storing the data is aligned in that way.
534  if (new_capacity.value() & 3) {
535  new_capacity += EntryIndex(4 - (new_capacity.value() & 3));
536  }
537 
538  const size_t index_buffer_size = new_capacity.value() * sizeof(Index);
539  const size_t value_buffer_size = new_capacity.value() * sizeof(Fractional);
540  const size_t new_buffer_size = index_buffer_size + value_buffer_size;
541  std::unique_ptr<char[]> new_buffer(new char[new_buffer_size]);
542  IndexType* const new_index = reinterpret_cast<Index*>(new_buffer.get());
543  Fractional* const new_coefficient =
544  reinterpret_cast<Fractional*>(new_index + new_capacity.value());
545 
546  // Avoid copying the data if the vector is empty.
547  if (num_entries_ > 0) {
548  // NOTE(user): We use memmove instead of std::copy, because the latter
549  // leads to naive copying code when used with strong ints (a loop that
550  // copies a single 32-bit value in each iteration), and as of 06/2016,
551  // memmove is 3-4x faster on Haswell.
552  std::memmove(new_index, index_, sizeof(IndexType) * num_entries_.value());
553  std::memmove(new_coefficient, coefficient_,
554  sizeof(Fractional) * num_entries_.value());
555  }
556  std::swap(buffer_, new_buffer);
557  index_ = new_index;
558  coefficient_ = new_coefficient;
559  capacity_ = new_capacity;
560 }

577 template <typename IndexType, typename IteratorType>
578 void SparseVector<IndexType, IteratorType>::CleanUp() {
579  // TODO(user): Implement in-place sorting of the entries and cleanup. The
580  // current version converts the data to an array-of-pairs representation that
581  // can be sorted easily with std::stable_sort, and the converts the sorted
582  // data back to the struct-of-arrays implementation.
583  // The current version is ~20% slower than the in-place sort on the
584  // array-of-struct representation. It is not visible on GLOP benchmarks, but
585  // it increases peak memory usage by ~8%.
586  // Implementing in-place search will require either implementing a custom
587  // sorting code, or custom iterators that abstract away the internal
588  // representation.
589  std::vector<std::pair<Index, Fractional>> entries;
590  entries.reserve(num_entries_.value());
591  for (EntryIndex i(0); i < num_entries_; ++i) {
592  entries.emplace_back(GetIndex(i), GetCoefficient(i));
593  }
594  std::stable_sort(
595  entries.begin(), entries.end(),

603  if (i + 1 == num_entries_ || entry.first != entries[i + 1].first) {
604  MutableIndex(new_size) = entry.first;
605  MutableCoefficient(new_size) = entry.second;

625 template <typename IndexType, typename IteratorType>
626 void SparseVector<IndexType, IteratorType>::PopulateFromSparseVector(
627  const SparseVector& sparse_vector) {
628  // Clear the sparse vector before reserving the new capacity. If we didn't do
629  // this, Reserve would have to copy the current contents of the vector if it
630  // allocated a new buffer. This would be wasteful, since we overwrite it in
631  // the next step anyway.
632  Clear();
633  Reserve(sparse_vector.capacity_);
634  // NOTE(user): Using a single memmove would be slightly faster, but it
635  // would not work correctly if this already had a greater capacity than
636  // sparse_vector, because the coefficient_ pointer would be positioned
637  // incorrectly.
638  std::memmove(index_, sparse_vector.index_,
639  sizeof(Index) * sparse_vector.num_entries_.value());
640  std::memmove(coefficient_, sparse_vector.coefficient_,
641  sizeof(Fractional) * sparse_vector.num_entries_.value());
642  num_entries_ = sparse_vector.num_entries_;
643  may_contain_duplicates_ = sparse_vector.may_contain_duplicates_;
644 }

670 template <typename IndexType, typename IteratorType>
671 bool SparseVector<IndexType, IteratorType>::CheckNoDuplicates(
672  StrictITIVector<IndexType, bool>* boolean_vector) const {
673  RETURN_VALUE_IF_NULL(boolean_vector, false);
674  // Note(user): Using num_entries() or any function that call
675  // CheckNoDuplicates() again will cause an infinite loop!
676  if (!may_contain_duplicates_ || num_entries_ <= 1) return true;
677 
678  // Update size if needed.
679  const Index max_index =
680  *std::max_element(index_, index_ + num_entries_.value());
681  if (boolean_vector->size() <= max_index) {
682  boolean_vector->resize(max_index + 1, false);
683  }
684 
685  may_contain_duplicates_ = false;
686  for (const EntryIndex i : AllEntryIndices()) {
687  const Index index = GetIndex(i);
688  if ((*boolean_vector)[index]) {
689  may_contain_duplicates_ = true;
690  break;
691  }
692  (*boolean_vector)[index] = true;
693  }
694 
695  // Reset boolean_vector to false.
696  for (const EntryIndex i : AllEntryIndices()) {
697  (*boolean_vector)[GetIndex(i)] = false;
698  }
699  return !may_contain_duplicates_;
700 }

876 template <typename IndexType, typename IteratorType>
877 void SparseVector<IndexType, IteratorType>::AddMultipleToSparseVectorInternal(
878  bool delete_common_index, Fractional multiplier, Index common_index,
879  Fractional drop_tolerance, SparseVector* accumulator_vector) const {
880  // DCHECK that the input is correct.
881  DCHECK(IsCleanedUp());
882  DCHECK(accumulator_vector->IsCleanedUp());
883  DCHECK(CheckNoDuplicates());
884  DCHECK(accumulator_vector->CheckNoDuplicates());
885  DCHECK_NE(0.0, LookUpCoefficient(common_index));
886  DCHECK_NE(0.0, accumulator_vector->LookUpCoefficient(common_index));
887 
888  // Implementation notes: we create a temporary SparseVector "c" to hold the
889  // result. We call "a" the first vector (i.e. the current object, which will
890  // be multiplied by "multiplier"), and "b" the second vector (which will be
891  // swapped with "c" at the end to hold the result).
892  // We incrementally build c as: a * multiplier + b.
893  const SparseVector& a = *this;
894  const SparseVector& b = *accumulator_vector;
895  SparseVector c;
896  EntryIndex ia(0); // Index in the vector "a"
897  EntryIndex ib(0); // ... and "b"
898  EntryIndex ic(0); // ... and "c"
899  const EntryIndex size_a = a.num_entries();
900  const EntryIndex size_b = b.num_entries();
901  const int size_adjustment = delete_common_index ? -2 : 0;
902  const EntryIndex new_size_upper_bound = size_a + size_b + size_adjustment;
903  c.Reserve(new_size_upper_bound);
904  c.num_entries_ = new_size_upper_bound;
905  while ((ia < size_a) && (ib < size_b)) {
906  const Index index_a = a.GetIndex(ia);
907  const Index index_b = b.GetIndex(ib);
908  // Benchmarks done by fdid@ in 2012 showed that it was faster to put the
909  // "if" clauses in that specific order.
910  if (index_a == index_b) {
911  if (index_a != common_index) {
912  const Fractional a_coeff_mul = multiplier * a.GetCoefficient(ia);
913  const Fractional b_coeff = b.GetCoefficient(ib);
914  const Fractional sum = a_coeff_mul + b_coeff;
915 
916  // We do not want to leave near-zero entries.
917  // TODO(user): expose the tolerance used here.
918  if (std::abs(sum) > drop_tolerance) {
919  c.MutableIndex(ic) = index_a;
920  c.MutableCoefficient(ic) = sum;
921  ++ic;
922  }
923  } else if (!delete_common_index) {
924  c.MutableIndex(ic) = b.GetIndex(ib);
925  c.MutableCoefficient(ic) = b.GetCoefficient(ib);
926  ++ic;
927  }
928  ++ia;
929  ++ib;
930  } else if (index_a < index_b) {
931  c.MutableIndex(ic) = index_a;
932  c.MutableCoefficient(ic) = multiplier * a.GetCoefficient(ia);
933  ++ia;
934  ++ic;
935  } else { // index_b < index_a
936  c.MutableIndex(ic) = b.GetIndex(ib);
937  c.MutableCoefficient(ic) = b.GetCoefficient(ib);
938  ++ib;
939  ++ic;
940  }
941  }
942  while (ia < size_a) {
943  c.MutableIndex(ic) = a.GetIndex(ia);
944  c.MutableCoefficient(ic) = multiplier * a.GetCoefficient(ia);
945  ++ia;
946  ++ic;
947  }
948  while (ib < size_b) {
949  c.MutableIndex(ic) = b.GetIndex(ib);
950  c.MutableCoefficient(ic) = b.GetCoefficient(ib);
951  ++ib;
952  ++ic;
953  }
954  c.ResizeDown(ic);
955  c.may_contain_duplicates_ = false;
956  c.Swap(accumulator_vector);
957 }

983 void SparseVector<IndexType, IteratorType>::MoveTaggedEntriesTo(
984  const IndexPermutation& index_perm, SparseVector* output) {

986  // and it is why we optimized the "nothing to do" case.
987  const EntryIndex end(num_entries_);
