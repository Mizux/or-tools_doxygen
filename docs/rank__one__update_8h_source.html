25 // This class holds a matrix of the form T = I + u.Tr(v) where I is the
26 // identity matrix and u and v are two column vectors of the same size as I. It
27 // allows for efficient left and right solves with T. When T is non-singular,
28 // it is easy to show that T^{-1} = I - 1 / mu * u.Tr(v) where
29 // mu = 1.0 + Tr(v).u
30 //
31 // Note that when v is a unit vector, T is a regular Eta matrix and when u
32 // is a unit vector, T is a row-wise Eta matrix.
33 //
34 // This is based on section 3.1 of:
35 // Qi Huangfu, J. A. Julian Hall, "Novel update techniques for the revised
36 // simplex method", 28 january 2013, Technical Report ERGO-13-0001
37 class RankOneUpdateElementaryMatrix {
38  public:
39  // Rather than copying the vectors u and v, RankOneUpdateElementaryMatrix
40  // takes two columns of a provided CompactSparseMatrix which is used for
41  // storage. This has a couple of advantages, especially in the context of the
42  // RankOneUpdateFactorization below:
43  // - It uses less overall memory (and avoid allocation overhead).
44  // - It has a better cache behavior for the RankOneUpdateFactorization solves.
45  RankOneUpdateElementaryMatrix(const CompactSparseMatrix* storage,
46  ColIndex u_index, ColIndex v_index,
47  Fractional u_dot_v)
48  : storage_(storage),
49  u_index_(u_index),
50  v_index_(v_index),
51  mu_(1.0 + u_dot_v) {}
52 
53  // Returns whether or not this matrix is singular.
54  // Note that the RightSolve() and LeftSolve() function will fail if this is
55  // the case.
56  bool IsSingular() const { return mu_ == 0.0; }
57 
58  // Solves T.x = rhs with rhs initialy in x (a column vector).
59  // The non-zeros version keeps track of the new non-zeros.
60  void RightSolve(DenseColumn* x) const {
61  DCHECK(!IsSingular());
62  const Fractional multiplier =
63  -storage_->ColumnScalarProduct(v_index_, Transpose(*x)) / mu_;
64  storage_->ColumnAddMultipleToDenseColumn(u_index_, multiplier, x);
65  }
66  void RightSolveWithNonZeros(ScatteredColumn* x) const {
67  DCHECK(!IsSingular());
68  const Fractional multiplier =
69  -storage_->ColumnScalarProduct(v_index_, Transpose(x->values)) / mu_;
70  if (multiplier != 0.0) {
71  storage_->ColumnAddMultipleToSparseScatteredColumn(u_index_, multiplier,
72  x);
73  }
74  }
75 
76  // Solves y.T = rhs with rhs initialy in y (a row vector).
77  // The non-zeros version keeps track of the new non-zeros.
78  void LeftSolve(DenseRow* y) const {
79  DCHECK(!IsSingular());
80  const Fractional multiplier =
81  -storage_->ColumnScalarProduct(u_index_, *y) / mu_;
82  storage_->ColumnAddMultipleToDenseColumn(v_index_, multiplier,
83  reinterpret_cast<DenseColumn*>(y));
84  }
85  void LeftSolveWithNonZeros(ScatteredRow* y) const {
86  DCHECK(!IsSingular());
87  const Fractional multiplier =
88  -storage_->ColumnScalarProduct(u_index_, y->values) / mu_;
89  if (multiplier != 0.0) {
90  storage_->ColumnAddMultipleToSparseScatteredColumn(
91  v_index_, multiplier, reinterpret_cast<ScatteredColumn*>(y));
92  }
93  }
94 
95  // Computes T.x for a given column vector.
96  void RightMultiply(DenseColumn* x) const {
97  const Fractional multiplier =
98  storage_->ColumnScalarProduct(v_index_, Transpose(*x));
99  storage_->ColumnAddMultipleToDenseColumn(u_index_, multiplier, x);
100  }
101 
102  // Computes y.T for a given row vector.
103  void LeftMultiply(DenseRow* y) const {
104  const Fractional multiplier = storage_->ColumnScalarProduct(u_index_, *y);
105  storage_->ColumnAddMultipleToDenseColumn(v_index_, multiplier,
106  reinterpret_cast<DenseColumn*>(y));
107  }
108 
109  EntryIndex num_entries() const {
110  return storage_->column(u_index_).num_entries() +
111  storage_->column(v_index_).num_entries();
112  }
113 
114  private:
115  // This is only used in debug mode.
116  Fractional ComputeUScalarV() const {
117  DenseColumn dense_u;
118  storage_->ColumnCopyToDenseColumn(u_index_, &dense_u);
119  return storage_->ColumnScalarProduct(v_index_, Transpose(dense_u));
120  }
121 
122  // Note that we allow copy and assignment so we can store a
123  // RankOneUpdateElementaryMatrix in an STL container.
124  const CompactSparseMatrix* storage_;
125  ColIndex u_index_;
126  ColIndex v_index_;
127  Fractional mu_;
128 };

130 // A rank one update factorization corresponds to the product of k rank one
131 // update elementary matrices, i.e. T = T_0.T_1. ... .T_{k-1}
132 class RankOneUpdateFactorization {
133  public:
134  // TODO(user): make the 5% a parameter and share it between all the places
135  // that switch between a sparse/dense version.
136  RankOneUpdateFactorization() : hypersparse_ratio_(0.05) {}
137 
138  // This is currently only visible for testing.
139  void set_hypersparse_ratio(double value) { hypersparse_ratio_ = value; }
140 
141  // Deletes all elementary matrices of this factorization.
142  void Clear() {
143  elementary_matrices_.clear();
144  num_entries_ = 0;
145  }
146 
147  // Updates the factorization.
148  void Update(const RankOneUpdateElementaryMatrix& update_matrix) {
149  elementary_matrices_.push_back(update_matrix);
150  num_entries_ += update_matrix.num_entries();
151  }
152 
153  // Left-solves all systems from right to left, i.e. y_i = y_{i+1}.(T_i)^{-1}
154  void LeftSolve(DenseRow* y) const {
155  RETURN_IF_NULL(y);
156  for (int i = elementary_matrices_.size() - 1; i >= 0; --i) {
157  elementary_matrices_[i].LeftSolve(y);
158  }
159  }
160 
161  // Same as LeftSolve(), but if the given non_zeros are not empty, then all
162  // the new non-zeros in the result are happended to it.
163  void LeftSolveWithNonZeros(ScatteredRow* y) const {
164  RETURN_IF_NULL(y);
165  if (y->non_zeros.empty()) {
166  LeftSolve(&y->values);
167  return;
168  }
169 
170  // tmp_row_is_non_zero_ is always all false before and after this code.
171  y->is_non_zero.resize(y->values.size(), false);
172  DCHECK(IsAllFalse(y->is_non_zero));
173  for (const ColIndex col : y->non_zeros) y->is_non_zero[col] = true;
174  const int hypersparse_threshold = static_cast<int>(
175  hypersparse_ratio_ * static_cast<double>(y->values.size().value()));
176  for (int i = elementary_matrices_.size() - 1; i >= 0; --i) {
177  if (y->non_zeros.size() < hypersparse_threshold) {
178  elementary_matrices_[i].LeftSolveWithNonZeros(y);
179  } else {
180  elementary_matrices_[i].LeftSolve(&y->values);
181  }
182  }
183  for (const ColIndex col : y->non_zeros) y->is_non_zero[col] = false;
184  if (y->non_zeros.size() >= hypersparse_threshold) y->non_zeros.clear();
185  }
186 
187  // Right-solves all systems from left to right, i.e. T_i.d_{i+1} = d_i
188  void RightSolve(DenseColumn* d) const {
189  RETURN_IF_NULL(d);
190  const size_t end = elementary_matrices_.size();
191  for (int i = 0; i < end; ++i) {
192  elementary_matrices_[i].RightSolve(d);
193  }
194  }
195 
196  // Same as RightSolve(), but if the given non_zeros are not empty, then all
197  // the new non-zeros in the result are happended to it.
198  void RightSolveWithNonZeros(ScatteredColumn* d) const {
199  RETURN_IF_NULL(d);
200  if (d->non_zeros.empty()) {
201  RightSolve(&d->values);
202  return;
203  }
204 
205  // d->is_non_zero is always all false before and after this code.
206  d->is_non_zero.resize(d->values.size(), false);
207  DCHECK(IsAllFalse(d->is_non_zero));
208  for (const RowIndex row : d->non_zeros) d->is_non_zero[row] = true;
209  const size_t end = elementary_matrices_.size();
210  const int hypersparse_threshold = static_cast<int>(
211  hypersparse_ratio_ * static_cast<double>(d->values.size().value()));
212  for (int i = 0; i < end; ++i) {
213  if (d->non_zeros.size() < hypersparse_threshold) {
214  elementary_matrices_[i].RightSolveWithNonZeros(d);
215  } else {
216  elementary_matrices_[i].RightSolve(&d->values);
217  }
218  }
219  for (const RowIndex row : d->non_zeros) d->is_non_zero[row] = false;
220  if (d->non_zeros.size() >= hypersparse_threshold) d->non_zeros.clear();
221  }
222 
223  EntryIndex num_entries() const { return num_entries_; }
224 
225  private:
226  double hypersparse_ratio_;
227  EntryIndex num_entries_;
228  std::vector<RankOneUpdateElementaryMatrix> elementary_matrices_;
229  DISALLOW_COPY_AND_ASSIGN(RankOneUpdateFactorization);
230 };
