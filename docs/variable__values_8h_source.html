25 // Class holding all the variable values and responsible for updating them. The
26 // variable values 'x' are such that 'A.x = 0' where A is the linear program
27 // matrix. This is because slack variables with bounds corresponding to the
28 // constraints bounds were added to the linear program matrix A.
29 //
30 // Some remarks:
31 // - For convenience, the variable values are stored in a DenseRow and indexed
32 // by ColIndex, like the variables and the columns of A.
33 // - During the dual-simplex, all non-basic variable values are at their exact
34 // bounds or exactly at 0.0 for a free variable.
35 // - During the primal-simplex, the non-basic variable values may not be exactly
36 // at their bounds because of bound-shifting during degenerate simplex
37 // pivoting which is implemented by not setting the variable values exactly at
38 // their bounds to have a lower primal residual error.
39 class VariableValues {
40  public:
41  VariableValues(const GlopParameters& parameters,
42  const CompactSparseMatrix& matrix,
43  const RowToColMapping& basis,
44  const VariablesInfo& variables_info,
45  const BasisFactorization& basis_factorization);
46 
47  // Getters for the variable values.
48  const Fractional Get(ColIndex col) const { return variable_values_[col]; }
49  const DenseRow& GetDenseRow() const { return variable_values_; }
50 
51  // Sets the value of a non-basic variable to the exact value implied by its
52  // current status. Note that the basic variable values are NOT updated by this
53  // function and it is up to the client to call RecomputeBasicVariableValues().
54  void SetNonBasicVariableValueFromStatus(ColIndex col);
55 
56  // Calls SetNonBasicVariableValueFromStatus() on all non-basic variables.
57  void ResetAllNonBasicVariableValues();
58 
59  // Recomputes the value of the basic variables from the non-basic ones knowing
60  // that the linear program matrix A times the variable values vector must be
61  // zero. It is better to call this when the basis is refactorized. This
62  // is checked in debug mode.
63  void RecomputeBasicVariableValues();
64 
65  // Computes the infinity norm of A.x where A is the linear_program matrix and
66  // x is the variable values column.
67  Fractional ComputeMaximumPrimalResidual() const;
68 
69  // Computes the maximum bound error for all the variables, defined as the
70  // distance of the current value of the variable to its interval
71  // [lower bound, upper bound]. The infeasibility is thus equal to 0.0 if the
72  // current value falls within the bounds, to the distance to lower_bound
73  // (resp. upper_bound), if the current value is below (resp. above)
74  // lower_bound (resp. upper_bound).
75  Fractional ComputeMaximumPrimalInfeasibility() const;
76  Fractional ComputeSumOfPrimalInfeasibilities() const;
77 
78  // Updates the variable during a simplex pivot:
79  // - step * direction is substracted from the basic variables value.
80  // - step is added to the entering column value.
81  void UpdateOnPivoting(const ScatteredColumn& direction, ColIndex entering_col,
82  Fractional step);
83 
84  // Batch version of SetNonBasicVariableValueFromStatus(). This function also
85  // updates the basic variable values and infeasibility statuses if
86  // update_basic_variables is true. The update is done in an incremental way
87  // and is thus more efficient than calling afterwards
88  // RecomputeBasicVariableValues() and ResetPrimalInfeasibilityInformation().
89  void UpdateGivenNonBasicVariables(const std::vector<ColIndex>& cols_to_update,
90  bool update_basic_variables);
91 
92  // Functions dealing with the primal-infeasible basic variables. A basic
93  // variable is primal-infeasible if its infeasibility is stricly greater than
94  // the primal feasibility tolerance.
95  //
96  // This information is only available after a call to
97  // ResetPrimalInfeasibilityInformation() and has to be kept in sync by calling
98  // UpdatePrimalInfeasibilityInformation() for the rows that changed values.
99  const DenseBitColumn& GetPrimalInfeasiblePositions() const;
100  const DenseColumn& GetPrimalSquaredInfeasibilities() const;
101  void ResetPrimalInfeasibilityInformation();
102  void UpdatePrimalInfeasibilityInformation(const std::vector<RowIndex>& rows);
103 
104  // The primal phase I objective is related to the primal infeasible
105  // information above. The cost of a basic column will be 1 if the variable is
106  // above its upper bound by strictly more than the primal tolerance, and -1 if
107  // it is lower than its lower bound by strictly less than the same tolerance.
108  //
109  // Returns true iff some cost changed.
110  template <typename Rows>
111  bool UpdatePrimalPhaseICosts(const Rows& rows, DenseRow* objective);
112 
113  // Sets the variable value of a given column.
114  void Set(ColIndex col, Fractional value) { variable_values_[col] = value; }
115 
116  // Parameters and stats functions.
117  std::string StatString() const { return stats_.StatString(); }
118 
119  private:
120  // It is important that the infeasibility is always computed in the same
121  // way. So the code should always use these functions that returns a positive
122  // value when the variable is out of bounds.
123  Fractional GetUpperBoundInfeasibility(ColIndex col) const {
124  return variable_values_[col] -
125  variables_info_.GetVariableUpperBounds()[col];
126  }
127  Fractional GetLowerBoundInfeasibility(ColIndex col) const {
128  return variables_info_.GetVariableLowerBounds()[col] -
129  variable_values_[col];
130  }
131 
132  // Input problem data.
133  const GlopParameters& parameters_;
134  const CompactSparseMatrix& matrix_;
135  const RowToColMapping& basis_;
136  const VariablesInfo& variables_info_;
137  const BasisFactorization& basis_factorization_;
138 
139  // Values of the variables.
140  DenseRow variable_values_;
141 
142  // Members used for the basic primal-infeasible variables.
143  DenseColumn primal_squared_infeasibilities_;
144  DenseBitColumn primal_infeasible_positions_;
145 
146  mutable StatsGroup stats_;
147  mutable ScatteredColumn scratchpad_;
148 
149  // A temporary scattered column that is always reset to all zero after use.
150  ScatteredColumn initially_all_zero_scratchpad_;
151 
152  DISALLOW_COPY_AND_ASSIGN(VariableValues);
153 };

155 template <typename Rows>
156 bool VariableValues::UpdatePrimalPhaseICosts(const Rows& rows,
157  DenseRow* objective) {
158  SCOPED_TIME_STAT(&stats_);
159  bool changed = false;
160  const Fractional tolerance = parameters_.primal_feasibility_tolerance();
161  for (const RowIndex row : rows) {
162  const ColIndex col = basis_[row];
163  Fractional new_cost = 0.0;
164  if (GetUpperBoundInfeasibility(col) > tolerance) {
165  new_cost = 1.0;
166  } else if (GetLowerBoundInfeasibility(col) > tolerance) {
167  new_cost = -1.0;
168  }
169  if (new_cost != (*objective)[col]) {
170  changed = true;
171  (*objective)[col] = new_cost;
172  }
173  }
174  return changed;
175 }
