27 // A strongly quadratic version of Time Tabling filtering. This propagator
28 // is similar to the CumulativeTimeTable propagator of the constraint solver.
29 class TimeTablingPerTask : public PropagatorInterface {
30  public:
31  TimeTablingPerTask(const std::vector<IntegerVariable>& demand_vars,
32  IntegerVariable capacity, IntegerTrail* integer_trail,
33  SchedulingConstraintHelper* helper);
34 
35  bool Propagate() final;
36 
37  void RegisterWith(GenericLiteralWatcher* watcher);
38 
39  private:
40  // The rectangle will be ordered by start, and the end of each rectangle
41  // will be equal to the start of the next one.
42  struct ProfileRectangle {
43  /* const */ IntegerValue start;
44  /* const */ IntegerValue height;
45 
46  ProfileRectangle(IntegerValue start, IntegerValue height)
47  : start(start), height(height) {}
48 
49  bool operator<(const ProfileRectangle& other) const {
50  return start < other.start;
51  }
52  };
53 
54  // Builds the profile and increases the lower bound of the capacity
55  // variable accordingly.
56  bool BuildProfile();
57 
58  // Reduces the set of tasks to be considered by BuildProfile(). This is done
59  // by not considering parts of mandatory part that are overlaped by no task.
60  // The set of tasks to consider is restored to its previous state each time a
61  // backtrack occurs. This function must be called only if the by_start_max_
62  // and by_end_min_ vectors are up to date and if the profile contains no
63  // profile rectangle that can exceed the capacity of the resource.
64  void ReduceProfile();
65 
66  // Reverses the profile. This is needed to reuse a given profile to update
67  // both the start and end times.
68  void ReverseProfile();
69 
70  // Tries to increase the minimum start time of each task according to the
71  // current profile. This function can be called after ReverseProfile() and
72  // ReverseVariables to update the maximum end time of each task.
73  bool SweepAllTasks(bool is_forward);
74 
75  // Tries to increase the minimum start time of task_id.
76  bool SweepTask(int task_id);
77 
78  // Updates the starting time of task_id to right and explain it. The reason is
79  // all the mandatory parts contained in [left, right).
80  bool UpdateStartingTime(int task_id, IntegerValue left, IntegerValue right);
81 
82  // Increases the minimum capacity to new_min and explain it. The reason is all
83  // the mandatory parts that overlap time.
84  bool IncreaseCapacity(IntegerValue time, IntegerValue new_min);
85 
86  // Explains the state of the profile in the time interval [left, right). The
87  // reason is all the mandatory parts that overlap the interval. The current
88  // reason is not cleared when this method is called.
89  void AddProfileReason(IntegerValue left, IntegerValue right);
90 
91  IntegerValue CapacityMin() const {
92  return integer_trail_->LowerBound(capacity_var_);
93  }
94 
95  IntegerValue CapacityMax() const {
96  return integer_trail_->UpperBound(capacity_var_);
97  }
98 
99  IntegerValue DemandMin(int task_id) const {
100  return integer_trail_->LowerBound(demand_vars_[task_id]);
101  }
102 
103  IntegerValue DemandMax(int task_id) const {
104  return integer_trail_->UpperBound(demand_vars_[task_id]);
105  }
106 
107  // Returns true if the tasks is present and has a mantatory part.
108  bool IsInProfile(int t) const {
109  return positions_in_profile_tasks_[t] < num_profile_tasks_;
110  }
111 
112  // Number of tasks.
113  const int num_tasks_;
114 
115  // The demand variables of the tasks.
116  std::vector<IntegerVariable> demand_vars_;
117 
118  // Capacity of the resource.
119  const IntegerVariable capacity_var_;
120 
121  IntegerTrail* integer_trail_;
122  SchedulingConstraintHelper* helper_;
123 
124  RevRepository<int> rev_repository_int_;
125  RevRepository<IntegerValue> rev_repository_integer_value_;
126 
127  // Vector of tasks sorted by maximum starting (resp. minimum ending) time.
128  std::vector<TaskTime> by_start_max_;
129  std::vector<TaskTime> by_end_min_;
130 
131  // Tasks contained in the range [left_start_, right_start_) of by_start_max_
132  // must be sorted and considered when building the profile. The state of these
133  // bounds is restored when a backtrack occurs.
134  int left_start_;
135  int right_start_;
136 
137  // Tasks contained in the range [left_end_, right_end_) of by_end_min_ must be
138  // sorted and considered when building the profile. The state of these bounds
139  // is restored when a backtrack occurs.
140  int left_end_;
141  int right_end_;
142 
143  // Optimistic profile of the resource consumption over time.
144  std::vector<ProfileRectangle> profile_;
145  IntegerValue profile_max_height_;
146 
147  // Reversible starting height of the reduced profile. This corresponds to the
148  // height of the leftmost profile rectangle that can be used for propagation.
149  IntegerValue starting_profile_height_;
150 
151  // True if the last call of the propagator has filtered the domain of a task
152  // and changed the shape of the profile.
153  bool profile_changed_;
154 
155  // Reversible sets of tasks to consider for the forward (resp. backward)
156  // propagation. A task with a fixed start do not need to be considered for the
157  // forward pass, same for task with fixed end for the backward pass. It is why
158  // we use two sets.
159  std::vector<int> forward_tasks_to_sweep_;
160  std::vector<int> backward_tasks_to_sweep_;
161  int forward_num_tasks_to_sweep_;
162  int backward_num_tasks_to_sweep_;
163 
164  // Reversible set of tasks to consider for reducing the profile. The set
165  // contains the [0, num_active_tasks_) prefix of active_tasks_.
166  std::vector<int> active_tasks_;
167  int num_active_tasks_;
168 
169  // Reversible set (with random access) of tasks to consider for building the
170  // profile. The set contains the tasks in the [0, num_profile_tasks_) prefix
171  // of profile_tasks_. The positions of a task in profile_tasks_ is contained
172  // in positions_in_profile_tasks_.
173  std::vector<int> profile_tasks_;
174  std::vector<int> positions_in_profile_tasks_;
175  int num_profile_tasks_;
176 
177  DISALLOW_COPY_AND_ASSIGN(TimeTablingPerTask);
178 };
