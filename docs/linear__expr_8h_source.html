17 // This file allows you to write natural code (like a mathematical equation) to
18 // model optimization problems with MPSolver. It is syntatic sugar on top of
19 // the MPSolver API, it provides no additional functionality. Use of these APIs
20 // makes it much easier to write code that is both simple and big-O optimal for
21 // creating your model, at the cost of some additional constant factor
22 // overhead. If model creation is a bottleneck in your problem, consider using
23 // the MPSolver API directly instead.
24 //
25 // This file contains two classes:
26 // 1. LinearExpr: models offset + sum_{i in S} a_i*x_i for decision var x_i,
27 // 2. LinearRange: models lb <= sum_{i in S} a_i*x_i <= ub,
28 // and it provides various operator overloads to build up "LinearExpr"s and
29 // then convert them to "LinearRange"s.
30 //
31 // Recommended use (avoids dangerous code):
32 //
33 // MPSolver solver = ...;
34 // const LinearExpr x = solver.MakeVar(...); // Note: implicit conversion
35 // const LinearExpr y = solver.MakeVar(...);
36 // const LinearExpr z = solver.MakeVar(...);
37 // const LinearExpr e1 = x + y;
38 // const LinearExpr e2 = (e1 + 7.0 + z)/3.0;
39 // const LinearRange r = e1 <= e2;
40 // solver.MakeRowConstraint(r);
41 //
42 // WARNING, AVOID THIS TRAP:
43 //
44 // MPSolver solver = ...;
45 // MPVariable* x = solver.MakeVar(...);
46 // LinearExpr y = x + 5;
47 //
48 // In evaluating "x+5" above, x is NOT converted to a LinearExpr before the
49 // addition, but rather is treated as a pointer, so x+5 gives a new pointer to
50 // garbage memory.
51 //
52 // For this reason, when using LinearExpr, it is best practice to:
53 // 1. use double literals instead of ints (e.g. "x + 5.0", not "x + 5"),
54 // 2. Immediately convert all MPVariable* to LinearExpr on creation, and only
55 // hold references to the "LinearExpr"s.
56 //
57 // Likewise, the following code is NOT recommended:
58 // MPSolver solver = ...;
59 // MPVariable* x = solver.MakeVar(...);
60 // MPVariable* y = solver.MakeVar(...);
61 // LinearExpr e1 = LinearExpr(x) + y + 5;
62 //
63 // While it is correct, it violates the natural assumption that the + operator
64 // is associative. Thus you are setting a trap for future modifications of the
65 // code, as any of the following changes would lead to the above failure mode:
66 // * LinearExpr e1 = LinearExpr(x) + (y + 5);
67 // * LinearExpr e1 = y + 5 + LinearExpr(x);

77 // LinearExpr models a quantity that is linear in the decision variables
78 // (MPVariable) of an optimization problem, i.e.
79 //
80 // offset + sum_{i in S} a_i*x_i,
81 //
82 // where the a_i and offset are constants and the x_i are MPVariables. You can
83 // use a LinearExpr "linear_expr" with an MPSolver "solver" to:
84 // * Set as the objective of your optimization problem, e.g.
85 //
86 // solver.MutableObjective()->MaximizeLinearExpr(linear_expr);
87 //
88 // * Create a constraint in your optimization, e.g.
89 //
90 // solver.MakeRowConstraint(linear_expr1 <= linear_expr2);
91 //
92 // * Get the value of the quantity after solving, e.g.
93 //
94 // solver.Solve();
95 // linear_expr.SolutionValue();
96 //
97 // LinearExpr is allowed to delete variables with coefficient zero from the map,
98 // but is not obligated to do so.
99 class LinearExpr {
100  public:
101  LinearExpr();
102  // Possible implicit conversions are intentional.
103  LinearExpr(double constant); // NOLINT
104  // Possible implicit conversions are intentional.
105  // Warning: var is not owned.
106  LinearExpr(const MPVariable* var); // NOLINT
107 
108  // Returns 1-var.
109  // NOTE(user): if var is binary variable, this corresponds to the logical
110  // negation of var.
111  // Passing by value is intentional, see the discussion on binary ops.
112  static LinearExpr NotVar(LinearExpr var);

134 // NOTE(user): in the ops below, the non-"const LinearExpr&" are intentional.
135 // We need to create a new LinearExpr for the result, so we lose nothing by

147 // An expression of the form:
148 // lower_bound <= sum_{i in S} a_i*x_i <= upper_bound.
149 // The sum is represented as a LinearExpr with offset 0.
150 //
151 // Must be added to model with
152 // MPSolver::AddRowConstraint(const LinearRange& range[, const std::string&
153 // name]);
154 class LinearRange {
155  public:
156  LinearRange() : lower_bound_(0), upper_bound_(0) {}
157  // The bounds of the linear range are updated so that they include the offset
158  // from "linear_expr", i.e., we form the range:
159  // lower_bound - offset <= linear_expr - offset <= upper_bound - offset.
160  LinearRange(double lower_bound, const LinearExpr& linear_expr,
161  double upper_bound);
162 
163  double lower_bound() const { return lower_bound_; }
164  const LinearExpr& linear_expr() const { return linear_expr_; }
165  double upper_bound() const { return upper_bound_; }
166 
167  private:
168  double lower_bound_;
169  // invariant: linear_expr_.offset() == 0.
170  LinearExpr linear_expr_;
171  double upper_bound_;
172 };
