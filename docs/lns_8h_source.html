29 // A simple deterministic and multithreaded LNS design.
30 //
31 // While !stop_function(), we call a batch of num_threads
32 // generate_and_solve_function() in parallel. Each such functions should return
33 // an update_global_solution() function. These update functions will only be
34 // called sequentially (and in seed order) once the batch is done.
35 //
36 // The seed starts at zero and will be increased one by one. Each
37 // generate_and_solve_function() will get a different seed.
38 //
39 // Only the generate_and_solve_function(int seed) need to be thread-safe.
40 //
41 // The two templated types should behave like:
42 // - StopFunction: std::function<bool()>
43 // - SolveNeighborhoodFunction: std::function<std::function<void()>(int seed)>
44 //
45 // TODO(user): As the improving solution become more difficult to find, we
46 // should trigger a lot more parallel LNS than the number of threads, so that we
47 // have a better utilization. For this, add a simple mecanism so that we always
48 // have num_threads working LNS, and do the barrier sync at the end when all the
49 // LNS task have been executed.
50 template <class StopFunction, class SolveNeighborhoodFunction>
51 void OptimizeWithLNS(int num_threads, StopFunction stop_function,
52  SolveNeighborhoodFunction generate_and_solve_function);

54 // Basic adaptive [0.0, 1.0] parameter that can be increased or decreased with a
55 // step that get smaller and smaller with the number of updates.
56 //
57 // Note(user): The current logic work well in practice, but has no theoretical
58 // foundation. So it might be possible to use better formulas depending on the
59 // situation.
60 //
61 // TODO(user): In multithread, we get Increase()/Decrease() signal from
62 // different thread potentially working on different difficulty. The class need
63 // to be updated to properly handle this case. Increase()/Decrease() should take
64 // in the difficulty at which the signal was computed, and the update formula
65 // should be changed accordingly.
66 class AdaptiveParameterValue {
67  public:
68  // Initial value is in [0.0, 1.0], both 0.0 and 1.0 are valid.
69  explicit AdaptiveParameterValue(double initial_value)
70  : value_(initial_value) {}
71 
72  void Reset() { num_changes_ = 0; }
73 
74  void Increase() {
75  const double factor = IncreaseNumChangesAndGetFactor();
76  value_ = std::min(1.0 - (1.0 - value_) / factor, value_ * factor);
77  }
78 
79  void Decrease() {
80  const double factor = IncreaseNumChangesAndGetFactor();
81  value_ = std::max(value_ / factor, 1.0 - (1.0 - value_) * factor);
82  }
83 
84  double value() const { return value_; }
85 
86  private:
87  // We want to change the parameters more and more slowly.
88  double IncreaseNumChangesAndGetFactor() {
89  ++num_changes_;
90  return 1.0 + 1.0 / std::sqrt(num_changes_ + 1);
91  }
92 
93  double value_;
94  int64_t num_changes_ = 0;
95 };

101 template <class StopFunction, class SolveNeighborhoodFunction>
102 inline void OptimizeWithLNS(
103  int num_threads, StopFunction stop_function,
104  SolveNeighborhoodFunction generate_and_solve_function) {
105  int64_t seed = 0;
106 #if !defined(__PORTABLE_PLATFORM__)
107  while (!stop_function()) {
108  if (num_threads > 1) {
109  std::vector<std::function<void()>> update_functions(num_threads);
110  {
111  ThreadPool pool("Parallel_LNS", num_threads);
112  pool.StartWorkers();
113  for (int i = 0; i < num_threads; ++i) {
114  pool.Schedule(
115  [&update_functions, &generate_and_solve_function, i, seed]() {
116  update_functions[i] = generate_and_solve_function(seed + i);
117  });
118  }
119  }
120 
121  seed += num_threads;
122  for (int i = 0; i < num_threads; ++i) {
123  update_functions[i]();
124  }
125  } else {
126  std::function<void()> update_function =
127  generate_and_solve_function(seed++);
128  update_function();
129  }
130  }
131 #else // __PORTABLE_PLATFORM__
132  while (!stop_function()) {
133  std::function<void()> update_function = generate_and_solve_function(seed++);
134  update_function();
135  }
136 #endif // __PORTABLE_PLATFORM__
137 }
