30 // The model "singleton" random engine used in the solver.
31 struct ModelRandomGenerator : public random_engine_t {

45 // Context: this function is not really generic, but required to be unit-tested.
46 // It is used in a clause minimization algorithm when we try to detect if any of
47 // the clause literals can be propagated by a subset of the other literal being
48 // false. For that, we want to enqueue in the solver all the subset of size n-1.
49 //
50 // This moves one of the unprocessed literal from literals to the last position.
51 // The function tries to do that while preserving the longest possible prefix of
52 // literals "amortized" through the calls assuming that we want to move each
53 // literal to the last position once.
54 //
55 // For a vector of size n, if we want to call this n times so that each literal
56 // is last at least once, the sum of the size of the changed suffixes will be
57 // O(n log n). If we were to use a simpler algorithm (like moving the last
58 // unprocessed literal to the last position), this sum would be O(n^2).
59 //
60 // Returns the size of the common prefix of literals before and after the move,
61 // or -1 if all the literals are already processed. The argument
62 // relevant_prefix_size is used as a hint when keeping more that this prefix
63 // size do not matter. The returned value will always be lower or equal to
64 // relevant_prefix_size.
65 int MoveOneUnprocessedLiteralLast(const std::set<LiteralIndex>& processed,
66  int relevant_prefix_size,
67  std::vector<Literal>* literals);

73 template <typename URBG>
74 inline void RandomizeDecisionHeuristic(URBG* random,
75  SatParameters* parameters) {
76 #if !defined(__PORTABLE_PLATFORM__)
77  // Random preferred variable order.
78  const google::protobuf::EnumDescriptor* order_d =
79  SatParameters::VariableOrder_descriptor();
80  parameters->set_preferred_variable_order(
81  static_cast<SatParameters::VariableOrder>(
82  order_d->value(absl::Uniform(*random, 0, order_d->value_count()))
83  ->number()));
84 
85  // Random polarity initial value.
86  const google::protobuf::EnumDescriptor* polarity_d =
87  SatParameters::Polarity_descriptor();
88  parameters->set_initial_polarity(static_cast<SatParameters::Polarity>(
89  polarity_d->value(absl::Uniform(*random, 0, polarity_d->value_count()))
90  ->number()));
91 #endif // __PORTABLE_PLATFORM__
92  // Other random parameters.
93  parameters->set_use_phase_saving(absl::Bernoulli(*random, 0.5));
94  parameters->set_random_polarity_ratio(absl::Bernoulli(*random, 0.5) ? 0.01
95  : 0.0);
96  parameters->set_random_branches_ratio(absl::Bernoulli(*random, 0.5) ? 0.01
97  : 0.0);
98 }
