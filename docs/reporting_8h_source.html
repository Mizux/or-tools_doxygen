25 // This class is used to abstract the interface to parallelism from
26 // the search code. It offers two sets of API:
27 // - Create specific search objects (Objective(), Limit()).
28 // - Report solution (SatSolution(), OptimizeSolution(), FinalOutput(),
29 // EndSearch(), BestSolution(), Interrupted(),
30 // Log(), Print()).
31 //
32 // There will be only one SearchReporting class shared among all the solver
33 // threads.
34 class SearchReportingInterface {
35  public:
36  enum Type {
37  UNDEF,
38  SATISFY,
39  MINIMIZE,
40  MAXIMIZE,
41  };
42 
43  SearchReportingInterface(bool print_all, int max_num_solutions)
44  : print_all_solutions_(print_all),
45  num_solutions_(0),
46  max_num_solutions_(max_num_solutions) {}
47 
48  virtual ~SearchReportingInterface() {}
49 
50  // ----- Events on the search -----
51 
52  // Initialize the interface for a given thread id.
53  // In sequential mode, the thread id is always -1.
54  // In parallel mode, it ranges from 0 to num_threads - 1.
55  virtual void Init(int thread_id, const std::string& init_string) = 0;
56 
57  // Callback on the start search event.
58  virtual void OnSearchStart(int thread_id, Type type) = 0;
59 
60  // Worker 'thread_id' notifies a new solution in a satisfaction
61  // problem. 'solution_string' is the solution to display if needed.
62  virtual void OnSatSolution(int thread_id,
63  const std::string& solution_string) = 0;
64 
65  // Worker 'thread_id' notifies a new solution in an optimization
66  // problem. 'solution_string' is the solution to display if needed.
67  virtual void OnOptimizeSolution(int thread_id, int64 value,
68  const std::string& solution_string) = 0;
69 
70  // Callback on the end search event.
71  virtual void OnSearchEnd(int thread_id, bool interrupted) = 0;
72 
73  // ----- Log methods ------
74 
75  // Logs the message from the given thread.
76  virtual void Log(int thread_id, const std::string& message) const = 0;
77  // Prints message to std::cout and adds a std::endl at the end.
78  // The minizinc specifications indicates that solutions and search status
79  // must be printed to std::cout.
80  virtual void Print(int thread_id, const std::string& output) const = 0;
81 
82  // ----- Getters -----
83 
84  // Checks if we should finish the search right away, for instance,
85  // in a satisfaction problem if a solution has already be found.
86  virtual bool ShouldFinish() const = 0;
87 
88  // Returns the value of the best solution found during search.
89  virtual int64 BestSolution() const = 0;
90 
91  // Returns true if the search was interrupted, usually by a time or
92  // solution limit.
93  virtual bool Interrupted() const = 0;
94 
95  // Returns the number of solutions found.
96  int NumSolutions() const { return num_solutions_; }
97 
98  // Returns the limit on the number of solutions to find.
99  int MaxNumSolutions() const { return max_num_solutions_; }
100 
101  // Indicates if we should print all solutions.
102  bool ShouldPrintAllSolutions() const { return print_all_solutions_; }
103 
104  // ----- Dedicated methods to create MT/Sequential specific objects -----
105 
106 #if !defined(SWIG)
107  // Creates the objective used by the search.
108  // Each solver thread will get a different one.
109  virtual OptimizeVar* CreateObjective(Solver* s, bool maximize, IntVar* var,
110  int64 step, int thread_id) const = 0;
111  // Creates a dedicated search limit.
112  // Each solver thread will get a different one.
113  virtual SearchLimit* CreateLimit(Solver* s, int thread_id) const = 0;
114 #endif // SWIG
115 
116  protected:
117  // Increments the number of solutions found.
118  void IncrementSolutions() { num_solutions_++; }
119 
120  private:
121  const bool print_all_solutions_;
122  int num_solutions_;
123  const int max_num_solutions_;
124 };
