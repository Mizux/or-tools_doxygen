75 // Returns true if a proto.domain() contain the given value.
76 // The domain is expected to be encoded as a sorted disjoint interval list.
77 template <typename ProtoWithDomain>
78 bool DomainInProtoContains(const ProtoWithDomain& proto, int64 value) {
79  for (int i = 0; i < proto.domain_size(); i += 2) {
80  if (value >= proto.domain(i) && value <= proto.domain(i + 1)) return true;
81  }
82  return false;
83 }

85 // Serializes a Domain into the domain field of a proto.
86 template <typename ProtoWithDomain>
87 void FillDomainInProto(const Domain& domain, ProtoWithDomain* proto) {
88  proto->clear_domain();
89  for (const ClosedInterval& interval : domain) {
90  proto->add_domain(interval.start);

95 // Reads a Domain from the domain field of a proto.
96 template <typename ProtoWithDomain>
97 Domain ReadDomainFromProto(const ProtoWithDomain& proto) {
98  std::vector<ClosedInterval> intervals;
99  for (int i = 0; i < proto.domain_size(); i += 2) {
100  intervals.push_back({proto.domain(i), proto.domain(i + 1)});
101  }
102  return Domain::FromIntervals(intervals);
103 }

105 // Returns the list of values in a given domain.
106 // This will fail if the domain contains more than one millions values.
107 //
108 // TODO(user): work directly on the Domain class instead.
109 template <typename ProtoWithDomain>
110 std::vector<int64> AllValuesInDomain(const ProtoWithDomain& proto) {
111  std::vector<int64> result;
112  for (int i = 0; i < proto.domain_size(); i += 2) {
113  for (int64 v = proto.domain(i); v <= proto.domain(i + 1); ++v) {
114  CHECK_LE(result.size(), 1e6);
115  result.push_back(v);
116  }
117  }
118  return result;
119 }

121 // Scale back a objective value to a double value from the original model.
122 inline double ScaleObjectiveValue(const CpObjectiveProto& proto, int64 value) {
123  double result = static_cast<double>(value);
124  if (value == kint64min) result = -std::numeric_limits<double>::infinity();
125  if (value == kint64max) result = std::numeric_limits<double>::infinity();
126  result += proto.offset();
127  if (proto.scaling_factor() == 0) return result;
128  return proto.scaling_factor() * result;
129 }
