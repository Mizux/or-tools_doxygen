29 // An eta matrix E corresponds to the identity matrix except for one column e of
30 // index j. In particular, B.E is the matrix of the new basis obtained from B by
31 // replacing the j-th vector of B by B.e, note that this is exactly what happens
32 // during a "pivot" of the current basis in the simplex algorithm.
33 //
34 // E = [ 1 ... 0 e_0 0 ... 0
35 // ... ... ... ... ... ... ...
36 // 0 ... 1 e_{j-1} 0 ... 0
37 // 0 ... 0 e_j 0 ... 0
38 // 0 ... 0 e_{j+1} 1 ... 0
39 // ... ... ... ... ... ... ...
40 // 0 ... 0 e_{n-1} 0 ... 1 ]
41 //
42 // The inverse of the eta matrix is:
43 // E^{-1} = [ 1 ... 0 -e_0/e_j 0 ... 0
44 // ... ... ... ... ... ... ...
45 // 0 ... 1 -e_{j-1}/e_j 0 ... 0
46 // 0 ... 0 1/e_j 0 ... 0
47 // 0 ... 0 -e_{j+1}/e_j 1 ... 0
48 // ... ... ... ... ... ... ...
49 // 0 ... 0 -e_{n-1}/e_j 0 ... 1 ]
50 class EtaMatrix {
51  public:
52  EtaMatrix(ColIndex eta_col, const ScatteredColumn& direction);
53  virtual ~EtaMatrix();
54 
55  // Solves the system y.E = c, 'c' beeing the initial value of 'y'.
56  // Then y = c.E^{-1}, so y is equal to c except for
57  // y_j = (c_j - \sum_{i != j}{c_i * e_i}) / e_j.
58  void LeftSolve(DenseRow* y) const;
59 
60  // Same as LeftSolve(), but 'pos' contains the non-zero positions of c. The
61  // order of the positions is not important, but there must be no duplicates.
62  // The values not in 'pos' are not used. If eta_col_ was not already in 'pos',
63  // it is added.
64  void SparseLeftSolve(DenseRow* y, ColIndexVector* pos) const;
65 
66  // Solves the system E.d = a, 'a' beeing the initial value of 'd'.
67  // Then d = E^{-1}.a = [ a_0 - e_0 * a_j / e_j
68  // ...
69  // a_{j-1} - e_{j-1} * a_j / e_j
70  // a_j / e_j
71  // a_{j+1} - e_{j+1} * a_j / e_j
72  // ...
73  // a_{n-1} - e_{n-1} * a_j / e_j ]
74  void RightSolve(DenseColumn* d) const;
75 
76  private:
77  // Internal RightSolve() and LeftSolve() implementations using either the
78  // dense or the sparse representation of the eta vector.
79  void LeftSolveWithDenseEta(DenseRow* y) const;
80  void LeftSolveWithSparseEta(DenseRow* y) const;
81  void RightSolveWithDenseEta(DenseColumn* d) const;
82  void RightSolveWithSparseEta(DenseColumn* d) const;
83 
84  // If an eta vector density is smaller than this threshold, we use the
85  // sparse version of the Solve() functions rather than the dense version.
86  // TODO(user): Detect automatically a good parameter? 0.5 is a good value on
87  // the Netlib (I only did a few experiments though). Note that in the future
88  // we may not even keep the dense representation at all.
89  static const Fractional kSparseThreshold;
90 
91  const ColIndex eta_col_;
92  const Fractional eta_col_coefficient_;
93 
94  // Note that to optimize solves, the position eta_col_ is set to 0.0 and
95  // stored in eta_col_coefficient_ instead.
96  DenseColumn eta_coeff_;
97  SparseColumn sparse_eta_coeff_;
98 
99  DISALLOW_COPY_AND_ASSIGN(EtaMatrix);
100 };

102 // An eta factorization corresponds to the product of k eta matrices,
103 // i.e. E = E_0.E_1. ... .E_{k-1}
104 // It is used to solve two systems:
105 // - E.d = a (where a is usually the entering column).
106 // - y.E = c (where c is usually the objective row).
107 class EtaFactorization {
108  public:
109  EtaFactorization();
110  virtual ~EtaFactorization();
111 
112  // Deletes all eta matrices.
113  void Clear();
114 
115  // Updates the eta factorization, i.e. adds the new eta matrix defined by
116  // the leaving variable and the corresponding eta column.
117  void Update(ColIndex entering_col, RowIndex leaving_variable_row,
118  const ScatteredColumn& direction);
119 
120  // Left solves all systems from right to left, i.e. y_i = y_{i+1}.(E_i)^{-1}
121  void LeftSolve(DenseRow* y) const;
122 
123  // Same as LeftSolve(), but 'pos' contains the non-zero positions of c. The
124  // order of the positions is not important, but there must be no duplicates.
125  // The values not in 'pos' are not used. If eta_col_ was not already in 'pos',
126  // it is added.
127  void SparseLeftSolve(DenseRow* y, ColIndexVector* pos) const;
128 
129  // Right solves all systems from left to right, i.e. E_i.d_{i+1} = d_i
130  void RightSolve(DenseColumn* d) const;
131 
132  private:
133  std::vector<EtaMatrix*> eta_matrix_;
134 
135  DISALLOW_COPY_AND_ASSIGN(EtaFactorization);
136 };

138 // A basis factorization is the product of an eta factorization and
139 // a L.U decomposition, i.e. B = L.U.E_0.E_1. ... .E_{k-1}
140 // It is used to solve two systems:
141 // - B.d = a where a is the entering column.
142 // - y.B = c where c is the objective row.
143 //
144 // To speed-up and improve stability the factorization is refactorized at least
145 // every 'refactorization_period' updates.
146 class BasisFactorization {
147  public:
148  BasisFactorization(const MatrixView& matrix,
149  const CompactSparseMatrix& compact_matrix,
150  const RowToColMapping& basis);

157  parameters.use_middle_product_form_update();
158  parameters_ = parameters;
159  lu_factorization_.SetParameters(parameters);
160  }

202  // Updates the factorization. The 'eta' column will be modified with a swap to

205  ABSL_MUST_USE_RESULT Status Update(ColIndex entering_col,
206  RowIndex leaving_variable_row,
207  const ScatteredColumn& direction);

209  // Left solves the system y.B = rhs, where y initialy contains rhs.
210  void LeftSolve(ScatteredRow* y) const;

212  // Left solves the system y.B = e_j, where e_j has only 1 non-zero
213  // coefficient of value 1.0 at position 'j'.
214  void LeftSolveForUnitRow(ColIndex j, ScatteredRow* y) const;
215 
216  // Same as LeftSolveForUnitRow() but does not update any internal data.
217  void TemporaryLeftSolveForUnitRow(ColIndex j, ScatteredRow* y) const;
218 
219  // Right solves the system B.d = a where the input is the initial value of d.
220  void RightSolve(ScatteredColumn* d) const;
221 
222  // Same as RightSolve() for matrix.column(col). This also exploits its
223  // sparsity.
224  void RightSolveForProblemColumn(ColIndex col, ScatteredColumn* d) const;
225 
226  // Specialized version for ComputeTau() in DualEdgeNorms. This reuses an
227  // intermediate result of the last LeftSolveForUnitRow() in order to save a
228  // permutation if it is available. Note that the input 'a' should always be
229  // equal to the last result of LeftSolveForUnitRow() and will be used for a
230  // DCHECK() or if the intermediate result wasn't kept.
231  const DenseColumn& RightSolveForTau(const ScatteredColumn& a) const;
232 
233  // Returns the norm of B^{-1}.a, this is a specific function because
234  // it is a bit faster and it avoids polluting the stats of RightSolve().
235  // It can be called only when IsRefactorized() is true.
236  Fractional RightSolveSquaredNorm(const SparseColumn& a) const;
237 
238  // Returns the norm of (B^T)^{-1}.e_row where e is an unit vector.
239  // This is a bit faster and avoids polluting the stats of LeftSolve().
240  // It can be called only when IsRefactorized() is true.
241  Fractional DualEdgeSquaredNorm(RowIndex row) const;
242 
243  // Computes the condition number of B.
244  // For a given norm, this is the matrix norm times the norm of its inverse.
245  // A condition number greater than 1E7 will lead to precision problems.
246  Fractional ComputeOneNormConditionNumber() const;
247  Fractional ComputeInfinityNormConditionNumber() const;
248  Fractional ComputeInfinityNormConditionNumberUpperBound() const;
249 
250  // Computes the 1-norm of B.
251  // The 1-norm |A| is defined as max_j sum_i |a_ij|
252  // http://en.wikipedia.org/wiki/Matrix_norm
253  Fractional ComputeOneNorm() const;
254 
255  // Computes the infinity-norm of B.
256  // The infinity-norm |A| is defined as max_i sum_j |a_ij|
257  // http://en.wikipedia.org/wiki/Matrix_norm
258  Fractional ComputeInfinityNorm() const;
259 
260  // Computes the 1-norm of the inverse of B.
261  // For this we iteratively solve B.x = e_j, where e_j is the jth unit vector.
262  // The result of this computation is the jth column of B^-1.
263  Fractional ComputeInverseOneNorm() const;
264 
265  // Computes the infinity-norm of the inverse of B.
266  Fractional ComputeInverseInfinityNorm() const;

336  // LeftSolveForUnitRow(), this way the optimization will automatically switch
337  // itself on when switching from the primal simplex (where RightSolveForTau()
338  // is never called) to the dual where it is called after each
339  // LeftSolveForUnitRow(), and back off again in the other direction.
340  mutable bool tau_computation_can_be_optimized_;
341  mutable bool tau_is_computed_;

350  mutable ColMapping right_pool_mapping_;

354  int num_updates_;

356  LuFactorization lu_factorization_;
357 
358  // mutable because the Solve() functions are const but need to update this.
359  mutable double deterministic_time_;
