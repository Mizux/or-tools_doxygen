27 // Neighborhood returned by Neighborhood generators.
28 struct Neighborhood {

30  // This is not expected to happen often but allows to handle this case
31  // properly.

35  // feasible solution to the base model too.

39 // Contains pre-computed information about a given CpModelProto that is meant
40 // to be used to generate LNS neighborhood. This class can be shared between
41 // more than one generator in order to reduce memory usage.
42 //

44 class NeighborhoodGeneratorHelper {

46  NeighborhoodGeneratorHelper(CpModelProto const* model_proto,
47  bool focus_on_decision_variables);
48 
49  // Updates private variables using the current 'model_proto_'.
50  void UpdateHelperData(bool focus_on_decision_variables);
51 
52  // Returns the LNS fragment where the given variables are fixed to the value
53  // they take in the given solution.
54  Neighborhood FixGivenVariables(
55  const CpSolverResponse& initial_solution,
56  const std::vector<int>& variables_to_fix) const;
57 
58  // Returns the LNS fragment which will relax all inactive variables and all
59  // variables in relaxed_variables.
60  Neighborhood RelaxGivenVariables(

63 
64  // Returns a trivial model by fixing all active variables to the initial
65  // solution values.
66  Neighborhood FixAllVariables(const CpSolverResponse& initial_solution) const;
67 
68  // Indicates if the variable can be frozen. It happens if the variable is non
69  // constant, and if it is a decision variable, or if
70  // focus_on_decision_variables is false.
71  bool IsActive(int var) const;
72 
73  // Returns the list of "active" variables.
74  const std::vector<int>& ActiveVariables() const { return active_variables_; }
75 
76  // Constraints <-> Variables graph.
77  const std::vector<std::vector<int>>& ConstraintToVar() const {
78  return constraint_to_var_;
79  }
80  const std::vector<std::vector<int>>& VarToConstraint() const {
81  return var_to_constraint_;

83 
84  // Returns all the constraints indices of a given type.
85  const absl::Span<const int> TypeToConstraints(
86  ConstraintProto::ConstraintCase type) const {
87  if (type >= type_to_constraints_.size()) return {};
88  return absl::MakeSpan(type_to_constraints_[type]);

90 
91  // The initial problem.
92  const CpModelProto& ModelProto() const { return model_proto_; }

94  private:
95  // Indicates if a variable is fixed in the model.
96  bool IsConstant(int var) const;
97 
98  const CpModelProto& model_proto_;
99 
100  // Constraints by types.
101  std::vector<std::vector<int>> type_to_constraints_;
102 
103  // Variable-Constraint graph.
104  std::vector<std::vector<int>> constraint_to_var_;
105  std::vector<std::vector<int>> var_to_constraint_;
106 
107  // The set of active variables, that is the list of non constant variables

110  // It is stored both as a list and as a set (using a Boolean vector).
111  std::vector<bool> active_variables_set_;
112  std::vector<int> active_variables_;

115 // Base class for a CpModelProto neighborhood generator.
116 class NeighborhoodGenerator {
117  public:
118  NeighborhoodGenerator(const std::string& name,
119  NeighborhoodGeneratorHelper const* helper)
120  : helper_(*helper), name_(name) {}
121  virtual ~NeighborhoodGenerator() {}
122 
123  // Generates a "local" subproblem for the given seed.
124  //
125  // The difficulty will be in [0, 1] and is related to the asked neighborhood
126  // size (and thus local problem difficulty). A difficulty of 0.0 means empty
127  // neighborhood and a difficulty of 1.0 means the full problem. The algorithm
128  // should try to generate a neighborhood according to this difficulty which
129  // will be dynamically adjusted depending on whether or not we can solve the
130  // subproblem in a given time limit.
131  //
132  // The given initial_solution should contains a feasible solution to the
133  // initial CpModelProto given to this class. Any solution to the returned
134  // CPModelProto should also be valid solution to the same initial model.
135  //
136  // This function should be thread-safe.
137  virtual Neighborhood Generate(const CpSolverResponse& initial_solution,
138  int64 seed, double difficulty) const = 0;
139 
140  // Returns a short description of the generator.
141  std::string name() const { return name_; }
142 
143  // Uses UCB1 algorithm to compute the score (Multi armed bandit problem).
144  // Details are at
145  // https://lilianweng.github.io/lil-log/2018/01/23/the-multi-armed-bandit-problem-and-its-solutions.html.
146  // 'total_num_calls' should be the sum of calls across all generators part of
147  // the multi armed bandit problem.
148  // If the generator is called less than 10 times then the method returns
149  // inifinity as score in order to get more data about the generator
150  // performance.
151  double GetUCBScore(int64 total_num_calls) const;
152 
153  // Updates the records using the current improvement in objective for the
154  // generator.
155  void AddSolveData(double objective_diff, double deterministic_time);
156 
157  // Number of times this generator is called.
158  int64 num_calls() const { return num_calls_; }
159 
160  protected:
161  const NeighborhoodGeneratorHelper& helper_;
162  const std::string name_;
163  int64 num_calls_ = 0;
164  double current_average_ = 0.0;
165 };

177 // Pick a random subset of variables that are constructed by a BFS in the
178 // variable <-> constraint graph. That is, pick a random variable, then all the
179 // variable connected by some constraint to the first one, and so on. The
180 // variable of the last "level" are selected randomly.
181 class VariableGraphNeighborhoodGenerator : public NeighborhoodGenerator {
182  public:
183  explicit VariableGraphNeighborhoodGenerator(
184  NeighborhoodGeneratorHelper const* helper, const std::string& name)
185  : NeighborhoodGenerator(name, helper) {}
186  Neighborhood Generate(const CpSolverResponse& initial_solution, int64 seed,
187  double difficulty) const final;
188 };

190 // Pick a random subset of constraint and relax all of their variables. We are a
191 // bit smarter than this because after the first contraint is selected, we only
192 // select constraints that share at least one variable with the already selected
193 // constraints. The variable from the "last" constraint are selected randomly.
194 class ConstraintGraphNeighborhoodGenerator : public NeighborhoodGenerator {
195  public:
196  explicit ConstraintGraphNeighborhoodGenerator(
197  NeighborhoodGeneratorHelper const* helper, const std::string& name)
198  : NeighborhoodGenerator(name, helper) {}
199  Neighborhood Generate(const CpSolverResponse& initial_solution, int64 seed,
200  double difficulty) const final;
201 };
