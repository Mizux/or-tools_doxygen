14 // This file defines the needed classes to efficiently perform Local Search in
15 // Bop.
16 // Local Search is a technique used to locally improve an existing solution by
17 // flipping a limited number of variables. To be successful the produced
18 // solution has to satisfy all constraints of the problem and improve the
19 // objective cost.
20 //
21 // The class BopLocalSearchOptimizer is the only public interface for Local
22 // Search in Bop. For unit-testing purposes this file also contains the four
23 // internal classes AssignmentAndConstraintFeasibilityMaintainer,
24 // OneFlipConstraintRepairer, SatWrapper and LocalSearchAssignmentIterator.
25 // They are implementation details and should not be used outside of bop_ls.

103 // This class defines a Local Search optimizer. The goal is to find a new
104 // solution with a better cost than the given solution by iterating on all
105 // assignments that can be reached in max_num_decisions decisions or less.
106 // The bop parameter max_number_of_explored_assignments_per_try_in_ls can be
107 // used to specify the number of new assignments to iterate on each time the
108 // method Optimize() is called. Limiting that parameter allows to reduce the
109 // time spent in the Optimize() method at once, and still explore all the
110 // reachable assignments (if Optimize() is called enough times).
111 // Note that due to propagation, the number of variables with a different value
112 // in the new solution can be greater than max_num_decisions.
113 class LocalSearchOptimizer : public BopOptimizerBase {
114  public:
115  LocalSearchOptimizer(const std::string& name, int max_num_decisions,
116  sat::SatSolver* sat_propagator);
117  ~LocalSearchOptimizer() override;
118 
119  private:
120  bool ShouldBeRun(const ProblemState& problem_state) const override;
121  Status Optimize(const BopParameters& parameters,
122  const ProblemState& problem_state, LearnedInfo* learned_info,
123  TimeLimit* time_limit) override;
124 
125  int64 state_update_stamp_;
126 
127  // Maximum number of decisions the Local Search can take.
128  // Note that there is no limit on the number of changed variables due to
129  // propagation.
130  const int max_num_decisions_;
131 
132  // A wrapper around the given sat_propagator.
133  SatWrapper sat_wrapper_;
134 
135  // Iterator on all reachable assignments.
136  // Note that this iterator is only reset when Synchronize() is called, i.e.
137  // the iterator continues its iteration of the next assignments each time
138  // Optimize() is called until everything is explored or a solution is found.
139  std::unique_ptr<LocalSearchAssignmentIterator> assignment_iterator_;
140 };

147 // Maintains some information on a sparse set of integers in [0, n). More
148 // specifically this class:
149 // - Allows to dynamically add/remove element from the set.
150 // - Has a backtracking support.
151 // - Maintains the number of elements in the set.
152 // - Maintains a superset of the elements of the set that contains all the
153 // modified elements.
154 template <typename IntType>
155 class BacktrackableIntegerSet {
156  public:
157  BacktrackableIntegerSet() {}
158 
159  // Prepares the class for integers in [0, n) and initializes the set to the
160  // empty one. Note that this run in O(n). Once resized, it is better to call
161  // BacktrackAll() instead of this to clear the set.
162  void ClearAndResize(IntType n);
163 
164  // Changes the state of the given integer i to be either inside or outside the
165  // set. Important: this should only be called with the opposite state of the
166  // current one, otherwise size() will not be correct.
167  void ChangeState(IntType i, bool should_be_inside);
168 
169  // Returns the current number of elements in the set.
170  // Note that this is not its maximum size n.
171  int size() const { return size_; }
172 
173  // Returns a superset of the current set of integers.
174  const std::vector<IntType>& Superset() const { return stack_; }
175 
176  // BacktrackOneLevel() backtracks to the state the class was in when the
177  // last AddBacktrackingLevel() was called. BacktrackAll() just restore the
178  // class to its state just after the last ClearAndResize().
179  void AddBacktrackingLevel();
180  void BacktrackOneLevel();
181  void BacktrackAll();
182 
183  private:
184  int size_;
185 
186  // Contains the elements whose status has been changed at least once.
187  std::vector<IntType> stack_;
188  std::vector<bool> in_stack_;
189 
190  // Used for backtracking. Contains the size_ and the stack_.size() at the time
191  // of each call to AddBacktrackingLevel() that is not yet backtracked over.
192  std::vector<int> saved_sizes_;
193  std::vector<int> saved_stack_sizes_;
194 };

196 // A simple and efficient class to hash a given set of integers in [0, n).
197 // It uses O(n) memory and produces a good hash (random linear function).
198 template <typename IntType>
199 class NonOrderedSetHasher {
200  public:
201  NonOrderedSetHasher() : random_("Random seed") {}
202 
203  // Initializes the NonOrderedSetHasher to hash sets of integer in [0, n).
204  void Initialize(int size) {
205  hashes_.resize(size);
206  for (IntType i(0); i < size; ++i) {
207  hashes_[i] = random_.Rand64();
208  }
209  }
210 
211  // Ignores the given set element in all subsequent hash computation. Note that
212  // this will be reset by the next call to Initialize().
213  void IgnoreElement(IntType e) { hashes_[e] = 0; }
214 
215  // Returns the hash of the given set. The hash is independent of the set
216  // order, but there must be no duplicate element in the set. This uses a
217  // simple random linear function which has really good hashing properties.
218  uint64 Hash(const std::vector<IntType>& set) const {
219  uint64 hash = 0;
220  for (const IntType i : set) hash ^= hashes_[i];
221  return hash;
222  }
223 
224  // The hash of a set is simply the XOR of all its elements. This allows
225  // to compute an hash incrementally or without the need of a std::vector<>.
226  uint64 Hash(IntType e) const { return hashes_[e]; }
227 
228  // Returns true if Initialize() has been called with a non-zero size.
229  bool IsInitialized() const { return !hashes_.empty(); }
230 
231  private:
232  MTRandom random_;
233  gtl::ITIVector<IntType, uint64> hashes_;
234 };

236 // This class is used to incrementally maintain an assignment and the
237 // feasibility of the constraints of a given LinearBooleanProblem.
238 //
239 // The current assignment is initialized using a feasible reference solution,
240 // i.e. the reference solution satisfies all the constraints of the problem.
241 // The current assignment is updated using the Assign() method.
242 //
243 // Note that the current assignment is not a solution in the sense that it
244 // might not be feasible, ie. violates some constraints.
245 //
246 // The assignment can be accessed at any time using Assignment().
247 // The set of infeasible constraints can be accessed at any time using
248 // PossiblyInfeasibleConstraints().
249 //
250 // Note that this class is reversible, i.e. it is possible to backtrack to
251 // previously added backtracking levels.
252 // levels. Consider for instance variable a, b, c, and d.
253 // Method called Assigned after method call
254 // 1- Assign({a, b}) a b
255 // 2- AddBacktrackingLevel() a b |
256 // 3- Assign({c}) a b | c
257 // 4- Assign({d}) a b | c d
258 // 5- BacktrackOneLevel() a b
259 // 6- Assign({c}) a b c
260 // 7- BacktrackOneLevel()
261 class AssignmentAndConstraintFeasibilityMaintainer {
262  public:
263  // Note that the constraint indices used in this class are not the same as
264  // the one used in the given LinearBooleanProblem here.
265  explicit AssignmentAndConstraintFeasibilityMaintainer(
266  const LinearBooleanProblem& problem);
267 
268  // When we construct the problem, we treat the objective as one constraint.
269  // This is the index of this special "objective" constraint.
270  static const ConstraintIndex kObjectiveConstraint;
271 
272  // Sets a new reference solution and reverts all internal structures to their
273  // initial state. Note that the reference solution has to be feasible.
274  void SetReferenceSolution(const BopSolution& reference_solution);
275 
276  // Behaves exactly like SetReferenceSolution() where the passed reference
277  // is the current assignment held by this class. Note that the current
278  // assignment must be feasible (i.e. IsFeasible() is true).
279  void UseCurrentStateAsReference();
280 
281  // Assigns all literals. That updates the assignment, the constraint values,
282  // and the infeasible constraints.
283  // Note that the assignment of those literals can be reverted thanks to
284  // AddBacktrackingLevel() and BacktrackOneLevel().
285  // Note that a variable can't be assigned twice, even for the same literal.
286  void Assign(const std::vector<sat::Literal>& literals);
287 
288  // Adds a new backtracking level to specify the state that will be restored
289  // by BacktrackOneLevel().
290  // See the example in the class comment.
291  void AddBacktrackingLevel();
292 
293  // Backtracks internal structures to the previous level defined by
294  // AddBacktrackingLevel(). As a consequence the state will be exactly as
295  // before the previous call to AddBacktrackingLevel().
296  // Note that backtracking the initial state has no effect.
297  void BacktrackOneLevel();
298  void BacktrackAll();
299 
300  // This returns the list of literal that appear in exactly all the current
301  // infeasible constraints (ignoring the objective) and correspond to a flip in
302  // a good direction for all the infeasible constraint. Performing this flip
303  // may repair the problem without any propagations.
304  //
305  // Important: The returned reference is only valid until the next
306  // PotentialOneFlipRepairs() call.
307  const std::vector<sat::Literal>& PotentialOneFlipRepairs();
308 
309  // Returns true if there is no infeasible constraint in the current state.
310  bool IsFeasible() const { return infeasible_constraint_set_.size() == 0; }
311 
312  // Returns the *exact* number of infeasible constraints.
313  // Note that PossiblyInfeasibleConstraints() will potentially return a larger
314  // number of constraints.
315  int NumInfeasibleConstraints() const {
316  return infeasible_constraint_set_.size();
317  }
318 
319  // Returns a superset of all the infeasible constraints in the current state.
320  const std::vector<ConstraintIndex>& PossiblyInfeasibleConstraints() const {
321  return infeasible_constraint_set_.Superset();
322  }
323 
324  // Returns the number of constraints of the problem, objective included,
325  // i.e. the number of constraint in the problem + 1.
326  size_t NumConstraints() const { return constraint_lower_bounds_.size(); }
327 
328  // Returns the value of the var in the assignment.
329  // As the assignment is initialized with the reference solution, if the
330  // variable has not been assigned through Assign(), the returned value is
331  // the value of the variable in the reference solution.
332  bool Assignment(VariableIndex var) const { return assignment_.Value(var); }
333 
334  // Returns the current assignment.
335  const BopSolution& reference() const { return reference_; }
336 
337  // Returns the lower bound of the constraint.
338  int64 ConstraintLowerBound(ConstraintIndex constraint) const {
339  return constraint_lower_bounds_[constraint];
340  }
341 
342  // Returns the upper bound of the constraint.
343  int64 ConstraintUpperBound(ConstraintIndex constraint) const {
344  return constraint_upper_bounds_[constraint];
345  }
346 
347  // Returns the value of the constraint. The value is computed using the
348  // variable values in the assignment. Note that a constraint is feasible iff
349  // its value is between its two bounds (inclusive).
350  int64 ConstraintValue(ConstraintIndex constraint) const {
351  return constraint_values_[constraint];
352  }
353 
354  // Returns true if the given constraint is currently feasible.
355  bool ConstraintIsFeasible(ConstraintIndex constraint) const {
356  const int64 value = ConstraintValue(constraint);
357  return value >= ConstraintLowerBound(constraint) &&
358  value <= ConstraintUpperBound(constraint);
359  }

361  std::string DebugString() const;

365  void InitializeConstraintSetHasher();
366 
367  // This is used by PotentialOneFlipRepairs(). It encodes a ConstraintIndex

416 // This class is an utility class used to select which infeasible constraint to
417 // repair and identify one variable to flip to actually repair the constraint.
418 // A constraint 'lb <= sum_i(w_i * x_i) <= ub', with 'lb' the lower bound,
419 // 'ub' the upper bound, 'w_i' the weight of the i-th term and 'x_i' the
420 // boolean variable appearing in the i-th term, is infeasible for a given
421 // assignment iff its value 'sum_i(w_i * x_i)' is outside of the bounds.
422 // Repairing-a-constraint-in-one-flip means making the constraint feasible by
423 // just flipping the value of one unassigned variable of the current assignment
424 // from the AssignmentAndConstraintFeasibilityMaintainer.
425 // For performance reasons, the pairs weight / variable (w_i, x_i) are stored
426 // in a sparse manner as a vector of terms (w_i, x_i). In the following the
427 // TermIndex term_index refers to the position of the term in the vector.
428 class OneFlipConstraintRepairer {
429  public:
430  // Note that the constraint indices used in this class follow the same
431  // convention as the one used in the
432  // AssignmentAndConstraintFeasibilityMaintainer.
433  //
434  // TODO(user): maybe merge the two classes? maintaining this implicit indices
435  // convention between the two classes sounds like a bad idea.
436  OneFlipConstraintRepairer(
437  const LinearBooleanProblem& problem,
438  const AssignmentAndConstraintFeasibilityMaintainer& maintainer,
439  const sat::VariablesAssignment& sat_assignment);
440 
441  static const ConstraintIndex kInvalidConstraint;
442  static const TermIndex kInitTerm;
443  static const TermIndex kInvalidTerm;
444 
445  // Returns the index of a constraint to repair. This will always return the
446  // index of a constraint that can be repaired in one flip if there is one.
447  // Note however that if there is only one possible candidate, it will be
448  // returned without checking that it can indeed be repaired in one flip.
449  // This is because the later check can be expensive, and is not needed in our
450  // context.
451  ConstraintIndex ConstraintToRepair() const;
452 
453  // Returns the index of the next term which repairs the constraint when the
454  // value of its variable is flipped. This method explores terms with an
455  // index strictly greater than start_term_index and then terms with an index
456  // smaller than or equal to init_term_index if any.
457  // Returns kInvalidTerm when no reparing terms are found.
458  //
459  // Note that if init_term_index == start_term_index, then all the terms will
460  // be explored. Both TermIndex arguments can take values in [-1, constraint
461  // size).
462  TermIndex NextRepairingTerm(ConstraintIndex ct_index,
463  TermIndex init_term_index,
464  TermIndex start_term_index) const;
465 
466  // Returns true if the constraint is infeasible and if flipping the variable
467  // at the given index will repair it.
468  bool RepairIsValid(ConstraintIndex ct_index, TermIndex term_index) const;
469 
470  // Returns the literal formed by the variable at the given constraint term and
471  // assigned to the opposite value of this variable in the current assignment.
472  sat::Literal GetFlip(ConstraintIndex ct_index, TermIndex term_index) const;
