39 const IntervalVariable kNoIntervalVariable(-1);

41 // This class maintains a set of intervals which correspond to three integer
42 // variables (start, end and size). It automatically registers with the
43 // PrecedencesPropagator the relation between the bounds of each interval and
44 // provides many helper functions to add precedences relation between intervals.
45 class IntervalsRepository {
46  public:
47  explicit IntervalsRepository(Model* model)
48  : integer_trail_(model->GetOrCreate<IntegerTrail>()),
49  precedences_(model->GetOrCreate<PrecedencesPropagator>()) {}
50 
51  // Returns the current number of intervals in the repository.
52  // The interval will always be identified by an integer in [0, num_intervals).
53  int NumIntervals() const { return start_vars_.size(); }
54 
55  // Functions to add a new interval to the repository.
56  // - If size == kNoIntegerVariable, then the size is fixed to fixed_size.
57  // - If is_present != kNoLiteralIndex, then this is an optional interval.
58  IntervalVariable CreateInterval(IntegerVariable start, IntegerVariable end,
59  IntegerVariable size, IntegerValue fixed_size,
60  LiteralIndex is_present);
61 
62  // Returns whether or not a interval is optional and the associated literal.
63  bool IsOptional(IntervalVariable i) const {
64  return is_present_[i] != kNoLiteralIndex;
65  }
66  Literal IsPresentLiteral(IntervalVariable i) const {
67  return Literal(is_present_[i]);
68  }
69 
70  // The 3 integer variables associated to a interval.
71  // Fixed size intervals will have a kNoIntegerVariable as size.
72  //
73  // Note: For an optional interval, the start/end variables are propagated
74  // asssuming the interval is present. Because of that, these variables can
75  // cross each other or have an empty domain. If any of this happen, then the
76  // IsPresentLiteral() of this interval will be propagated to false.
77  IntegerVariable SizeVar(IntervalVariable i) const { return size_vars_[i]; }
78  IntegerVariable StartVar(IntervalVariable i) const { return start_vars_[i]; }
79  IntegerVariable EndVar(IntervalVariable i) const { return end_vars_[i]; }
80 
81  // Return the minimum size of the given IntervalVariable.
82  IntegerValue MinSize(IntervalVariable i) const {
83  const IntegerVariable size_var = size_vars_[i];
84  if (size_var == kNoIntegerVariable) return fixed_sizes_[i];
85  return integer_trail_->LowerBound(size_var);
86  }
87 
88  // Return the maximum size of the given IntervalVariable.
89  IntegerValue MaxSize(IntervalVariable i) const {
90  const IntegerVariable size_var = size_vars_[i];
91  if (size_var == kNoIntegerVariable) return fixed_sizes_[i];
92  return integer_trail_->UpperBound(size_var);
93  }
94 
95  private:
96  // External classes needed.
97  IntegerTrail* integer_trail_;
98  PrecedencesPropagator* precedences_;
99 
100  // Literal indicating if the tasks is executed. Tasks that are always executed
101  // will have a kNoLiteralIndex entry in this vector.
102  gtl::ITIVector<IntervalVariable, LiteralIndex> is_present_;
103 
104  // The integer variables for each tasks.
105  gtl::ITIVector<IntervalVariable, IntegerVariable> start_vars_;
106  gtl::ITIVector<IntervalVariable, IntegerVariable> end_vars_;
107  gtl::ITIVector<IntervalVariable, IntegerVariable> size_vars_;
108  gtl::ITIVector<IntervalVariable, IntegerValue> fixed_sizes_;
109 
110  DISALLOW_COPY_AND_ASSIGN(IntervalsRepository);
111 };

113 // An helper struct to sort task by time. This is used by the
114 // SchedulingConstraintHelper but also by many scheduling propagators to sort
115 // tasks.
116 struct TaskTime {
117  int task_index;
118  IntegerValue time;
119  bool operator<(TaskTime other) const { return time < other.time; }
120  bool operator>(TaskTime other) const { return time > other.time; }
121 };

154  // Helpers for the current bounds on the current task time window.
155  // [(duration-min) ... (duration-min)]
156  // ^ ^ ^ ^
157  // start-min end-min start-max end-max
158  //
159  // Note that for tasks with variable durations, we don't necessarily have
160  // duration-min between the XXX-min and XXX-max value.

168  // In the presense of tasks with a variable duration, we do not necessarily
169  // have start_min + duration_min = end_min, we can instead have a situation
170  // like:
171  // | |<- duration-min ->|
172  // ^ ^ ^
173  // start-min | end-min
174  // |
175  // We define the "shifted start min" to be the right most time such that
176  // we known that we must have min-duration "energy" to the right of it if the
177  // task is present. Using it in our scheduling propagators allows to propagate
178  // more in the presence of tasks with variable duration (or optional task
179  // where we also do not necessarily have start_min + duration_min = end_min.
180  //
181  // To explain this shifted start min, one must use the AddEnergyAfterReason().
182  IntegerValue ShiftedStartMin(int t) const;
183 
184  bool StartIsFixed(int t) const;
185  bool EndIsFixed(int t) const;
186 
187  // Returns true if the corresponding fact is known for sure. A normal task is
188  // always present. For optional task for which the presence is still unknown,
189  // both of these function will return false.
190  bool IsOptional(int t) const;
191  bool IsPresent(int t) const;
192  bool IsAbsent(int t) const;
193 
194  // Sorts and returns the tasks in corresponding order at the time of the call.

196  // will be duplicate time values in these vectors.
197  //
198  // TODO(user): we could merge the first loop of IncrementalSort() with the
199  // loop that fill TaskTime.time at each call.
200  const std::vector<TaskTime>& TaskByIncreasingStartMin();
201  const std::vector<TaskTime>& TaskByIncreasingEndMin();
202  const std::vector<TaskTime>& TaskByDecreasingStartMax();
203  const std::vector<TaskTime>& TaskByDecreasingEndMax();
204  const std::vector<TaskTime>& TaskByIncreasingShiftedStartMin();

399  AddOtherReason(t);

405 inline void SchedulingConstraintHelper::AddDurationMinReason(int t) {
406  AddOtherReason(t);
407  if (duration_vars_[t] != kNoIntegerVariable) {
408  integer_reason_.push_back(
409  integer_trail_->LowerBoundAsLiteral(duration_vars_[t]));
410  }
411 }

413 inline void SchedulingConstraintHelper::AddDurationMinReason(
414  int t, IntegerValue lower_bound) {
415  AddOtherReason(t);
416  if (duration_vars_[t] != kNoIntegerVariable) {
417  DCHECK_GE(DurationMin(t), lower_bound);
418  integer_reason_.push_back(
419  IntegerLiteral::GreaterOrEqual(duration_vars_[t], lower_bound));
420  }
421 }

423 inline void SchedulingConstraintHelper::AddStartMinReason(
424  int t, IntegerValue lower_bound) {
425  DCHECK_GE(StartMin(t), lower_bound);
426  AddOtherReason(t);
427  integer_reason_.push_back(
428  IntegerLiteral::GreaterOrEqual(start_vars_[t], lower_bound));
429 }
430 
431 inline void SchedulingConstraintHelper::AddStartMaxReason(
432  int t, IntegerValue upper_bound) {
433  DCHECK_LE(StartMax(t), upper_bound);
434  AddOtherReason(t);
435  integer_reason_.push_back(
436  IntegerLiteral::LowerOrEqual(start_vars_[t], upper_bound));
437 }
438 
439 inline void SchedulingConstraintHelper::AddEndMinReason(
440  int t, IntegerValue lower_bound) {
441  DCHECK_GE(EndMin(t), lower_bound);
442  AddOtherReason(t);
443  integer_reason_.push_back(
444  IntegerLiteral::GreaterOrEqual(end_vars_[t], lower_bound));
445 }
446 
447 inline void SchedulingConstraintHelper::AddEndMaxReason(
448  int t, IntegerValue upper_bound) {
449  DCHECK_LE(EndMax(t), upper_bound);
450  AddOtherReason(t);
451  integer_reason_.push_back(
452  IntegerLiteral::LowerOrEqual(end_vars_[t], upper_bound));
453 }

455 inline void SchedulingConstraintHelper::AddEnergyAfterReason(
456  int t, IntegerValue energy_min, IntegerValue time) {
457  if (StartMin(t) >= time) {
458  AddStartMinReason(t, time);
459  } else {
460  AddEndMinReason(t, time + energy_min);
461  }
462  AddDurationMinReason(t, energy_min);
463 }

593 // This requires that all the alternatives are optional tasks.
594 inline std::function<void(Model*)> IntervalWithAlternatives(
595  IntervalVariable master, const std::vector<IntervalVariable>& members) {
596  return [=](Model* model) {
597  IntervalsRepository* intervals = model->GetOrCreate<IntervalsRepository>();
598 
599  std::vector<Literal> presences;
600  std::vector<IntegerValue> durations;
601 
602  // Create an "exactly one executed" constraint on the alternatives.
603  std::vector<LiteralWithCoeff> sat_ct;
604  for (const IntervalVariable member : members) {
605  CHECK(intervals->IsOptional(member));
606  const Literal is_present = intervals->IsPresentLiteral(member);
607  sat_ct.push_back({is_present, Coefficient(1)});
608  model->Add(
609  Equality(model->Get(StartVar(master)), model->Get(StartVar(member))));
610  model->Add(

638  ends.push_back(intervals->EndVar(member));
639  }
640  model->Add(PartialIsOneOfVar(intervals->EndVar(master), ends, presences));
