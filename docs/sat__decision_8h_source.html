31 // Implement the SAT branching policy responsible for deciding the next Boolean
32 // variable to branch on, and its polarity (true or false).
33 class SatDecisionPolicy {
34  public:
35  explicit SatDecisionPolicy(Model* model);
36 
37  // Notifies that more variables are now present. Note that currently this may
38  // change the current variable order because the priority queue need to be
39  // reconstructed.
40  void IncreaseNumVariables(int num_variables);
41 
42  // Reinitializes the decision heuristics (which variables to choose with which
43  // polarity) according to the current parameters. Note that this also resets
44  // the activity of the variables to 0. Note that this function is lazy, and
45  // the work will only happen on the first NextBranch() to cover the cases when
46  // this policy is not used at all.
47  void ResetDecisionHeuristic();
48 
49  // Returns next decision to branch upon. This shouldn't be called if all the
50  // variables are assigned.
51  Literal NextBranch();
52 
53  // Updates statistics about literal occurences in constraints.
54  // Input is a canonical linear constraint of the form (terms <= rhs).
55  void UpdateWeightedSign(const std::vector<LiteralWithCoeff>& terms,
56  Coefficient rhs);
57 
58  // Bumps the activity of all variables appearing in the conflict. All literals
59  // must be currently assigned. See VSIDS decision heuristic: Chaff:
60  // Engineering an Efficient SAT Solver. M.W. Moskewicz et al. ANNUAL ACM IEEE
61  // DESIGN AUTOMATION CONFERENCE 2001.
62  void BumpVariableActivities(const std::vector<Literal>& literals);
63 
64  // Updates the increment used for activity bumps. This is basically the same
65  // as decaying all the variable activities, but it is a lot more efficient.
66  void UpdateVariableActivityIncrement();
67 
68  // Called on Untrail() so that we can update the set of possible decisions.
69  void Untrail(int target_trail_index);
70 
71  // Called on a new conflict.
72  void OnConflict() {
73  if (parameters_.use_erwa_heuristic()) {
74  ++num_conflicts_;
75  num_conflicts_stack_.push_back({trail_->Index(), 1});
76  }
77  }
78 
79  // Gives a hint so the solver tries to find a solution with the given literal
80  // set to true. Currently this take precedence over the phase saving heuristic
81  // and a variable with a preference will always be branched on according to
82  // this preference.
83  //
84  // The weight is used as a tie-breaker between variable with the same
85  // activities. Larger weight will be selected first. A weight of zero is the
86  // default value for the other variables.
87  //
88  // Note(user): Having a lot of different weights may slow down the priority
89  // queue operations if there is millions of variables.
90  void SetAssignmentPreference(Literal literal, double weight);
91 
92  // Returns the vector of the current assignment preferences.
93  std::vector<std::pair<Literal, double>> AllPreferences() const;
94 
95  private:
96  // Computes an initial variable ordering.
97  void InitializeVariableOrdering();
98 
99  // Rescales activity value of all variables when one of them reached the max.
100  void RescaleVariableActivities(double scaling_factor);
101 
102  // Reinitializes the inital polarity of all the variables with an index
103  // greater than or equal to the given one.
104  void ResetInitialPolarity(int from);
105 
106  // Adds the given variable to var_ordering_ or updates its priority if it is
107  // already present.
108  void PqInsertOrUpdate(BooleanVariable var);
109 
110  // Singleton model objects.
111  const SatParameters& parameters_;
112  Trail* trail_;
113  random_engine_t* random_;
114 
115  // Variable ordering (priority will be adjusted dynamically). queue_elements_
116  // holds the elements used by var_ordering_ (it uses pointers).
117  //
118  // Note that we recover the variable that a WeightedVarQueueElement refers to
119  // by its position in the queue_elements_ vector, and we can recover the later
120  // using (pointer - &queue_elements_[0]).
121  struct WeightedVarQueueElement {
122  // Interface for the IntegerPriorityQueue.
123  int Index() const { return var.value(); }
124 
125  // Priority order. The IntegerPriorityQueue returns the largest element
126  // first.
127  //
128  // Note(user): We used to also break ties using the variable index, however
129  // this has two drawbacks:
130  // - On problem with many variables, this slow down quite a lot the priority
131  // queue operations (which do as little work as possible and hence benefit
132  // from having the majority of elements with a priority of 0).
133  // - It seems to be a bad heuristics. One reason could be that the priority
134  // queue will automatically diversify the choice of the top variables
135  // amongst the ones with the same priority.
136  //
137  // Note(user): For the same reason as explained above, it is probably a good
138  // idea not to have too many different values for the tie_breaker field. I
139  // am not even sure we should have such a field...
140  bool operator<(const WeightedVarQueueElement& other) const {
141  return weight < other.weight ||
142  (weight == other.weight && (tie_breaker < other.tie_breaker));
143  }
144 
145  BooleanVariable var;
146  float tie_breaker;
147 
148  // TODO(user): Experiment with float. In the rest of the code, we use
149  // double, but maybe we don't need that much precision. Using float here may
150  // save memory and make the PQ operations faster.
151  double weight;
152  };
153  COMPILE_ASSERT(sizeof(WeightedVarQueueElement) == 16,
154  ERROR_WeightedVarQueueElement_is_not_well_compacted);
155 
156  bool var_ordering_is_initialized_ = false;
157  IntegerPriorityQueue<WeightedVarQueueElement> var_ordering_;
158 
159  // This is used for the branching heuristic described in "Learning Rate Based
160  // Branching Heuristic for SAT solvers", J.H.Liang, V. Ganesh, P. Poupart,
161  // K.Czarnecki, SAT 2016.
162  //
163  // The entries are sorted by trail index, and one can get the number of
164  // conflicts during which a variable at a given trail index i was assigned by
165  // summing the entry.count for all entries with a trail index greater than i.
166  struct NumConflictsStackEntry {
167  int trail_index;
168  int64 count;
169  };
170  int64 num_conflicts_ = 0;
171  std::vector<NumConflictsStackEntry> num_conflicts_stack_;
172 
173  // Whether the priority of the given variable needs to be updated in
174  // var_ordering_. Note that this is only accessed for assigned variables and
175  // that for efficiency it is indexed by trail indices. If
176  // pq_need_update_for_var_at_trail_index_[trail_->Info(var).trail_index] is
177  // true when we untrail var, then either var need to be inserted in the queue,
178  // or we need to notify that its priority has changed.
179  BitQueue64 pq_need_update_for_var_at_trail_index_;
180 
181  // Increment used to bump the variable activities.
182  double variable_activity_increment_ = 1.0;
183 
184  // Stores variable activity and the number of time each variable was "bumped".
185  // The later is only used with the ERWA heuristic.
186  gtl::ITIVector<BooleanVariable, double> activities_;
187  gtl::ITIVector<BooleanVariable, double> tie_breakers_;
188  gtl::ITIVector<BooleanVariable, int64> num_bumps_;
189 
190  // Used by NextBranch() to choose the polarity of the next decision. For the
191  // phase saving, the last polarity is stored in trail_->Info(var).
192  gtl::ITIVector<BooleanVariable, bool> var_use_phase_saving_;
193  gtl::ITIVector<BooleanVariable, bool> var_polarity_;
194 
195  // Used in initial polarity computation.
196  gtl::ITIVector<BooleanVariable, double> weighted_sign_;
197 };
