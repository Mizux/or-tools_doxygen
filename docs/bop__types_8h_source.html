30 // Status of the solve of Bop.
31 enum class BopSolveStatus {
32  // The solver found the proven optimal solution.
33  OPTIMAL_SOLUTION_FOUND,
34 
35  // The solver found a solution, but it is not proven to be the optimal
36  // solution.
37  FEASIBLE_SOLUTION_FOUND,
38 
39  // The solver didn't find any solution.
40  NO_SOLUTION_FOUND,
41 
42  // The problem is infeasible.
43  INFEASIBLE_PROBLEM,
44 
45  // The problem is invalid.
46  INVALID_PROBLEM
47 };

49 inline std::string GetSolveStatusString(BopSolveStatus status) {
50  switch (status) {
51  case BopSolveStatus::OPTIMAL_SOLUTION_FOUND:
52  return "OPTIMAL_SOLUTION_FOUND";
53  case BopSolveStatus::FEASIBLE_SOLUTION_FOUND:
54  return "FEASIBLE_SOLUTION_FOUND";
55  case BopSolveStatus::NO_SOLUTION_FOUND:
56  return "NO_SOLUTION_FOUND";
57  case BopSolveStatus::INFEASIBLE_PROBLEM:
58  return "INFEASIBLE_PROBLEM";
59  case BopSolveStatus::INVALID_PROBLEM:
60  return "INVALID_PROBLEM";
61  }
62  // Fallback. We don't use "default:" so the compiler will return an error
63  // if we forgot one enum case above.
64  return "UNKNOWN Status";
65 }
66 inline std::ostream& operator<<(std::ostream& os, BopSolveStatus status) {
67  os << GetSolveStatusString(status);
68  return os;
69 }
