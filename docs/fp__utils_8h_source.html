109 // Tests whether x and y are close to one another using absolute and relative
110 // tolerances.
111 // Returns true if |x - y| <= a (with a being the absolute_tolerance).
112 // The above case is useful for values that are close to zero.
113 // Returns true if |x - y| <= max(|x|, |y|) * r. (with r being the relative
114 // tolerance.)
115 // The cases for infinities are treated separately to avoid generating NaNs.
116 template <typename FloatType>
117 bool AreWithinAbsoluteOrRelativeTolerances(FloatType x, FloatType y,
118  FloatType relative_tolerance,
119  FloatType absolute_tolerance) {
120  DCHECK_LE(0.0, relative_tolerance);
121  DCHECK_LE(0.0, absolute_tolerance);
122  DCHECK_GT(1.0, relative_tolerance);
123  if (IsPositiveOrNegativeInfinity(x) || IsPositiveOrNegativeInfinity(y)) {
124  return x == y;
125  }
126  const FloatType difference = fabs(x - y);
127  if (difference <= absolute_tolerance) {
128  return true;
129  }
130  const FloatType largest_magnitude = std::max(fabs(x), fabs(y));
131  return difference <= largest_magnitude * relative_tolerance;
132 }

134 // Tests whether x and y are close to one another using an absolute tolerance.
135 // Returns true if |x - y| <= a (with a being the absolute_tolerance).
136 // The cases for infinities are treated separately to avoid generating NaNs.
137 template <typename FloatType>
138 bool AreWithinAbsoluteTolerance(FloatType x, FloatType y,
139  FloatType absolute_tolerance) {
140  DCHECK_LE(0.0, absolute_tolerance);
141  if (IsPositiveOrNegativeInfinity(x) || IsPositiveOrNegativeInfinity(y)) {
142  return x == y;
143  }
144  return fabs(x - y) <= absolute_tolerance;
145 }

147 // Returns true if x is less than y or slighlty greater than y with the given
148 // absolute or relative tolerance.
149 template <typename FloatType>
150 bool IsSmallerWithinTolerance(FloatType x, FloatType y, FloatType tolerance) {
151  if (IsPositiveOrNegativeInfinity(y)) return x <= y;
152  return x <= y + tolerance * std::max(1.0, std::min(std::abs(x), std::abs(y)));
153 }

155 // Returns true if x is within tolerance of any integer. Always returns
156 // false for x equal to +/- infinity.
157 template <typename FloatType>
158 inline bool IsIntegerWithinTolerance(FloatType x, FloatType tolerance) {
159  DCHECK_LE(0.0, tolerance);
160  if (IsPositiveOrNegativeInfinity(x)) return false;
161  return std::abs(x - std::round(x)) <= tolerance;
162 }

179 // Given an array of doubles, this computes a positive scaling factor such that
180 // the scaled doubles can then be rounded to integers with little or no loss of
181 // precision, and so that the L1 norm of these integers is <= max_sum. More
182 // precisely, the following formulas will hold (x[i] is input[i], for brevity):
183 // - For all i, |round(factor * x[i]) / factor - x[i]| <= error * |x[i]|
184 // - The sum over i of |round(factor * x[i])| <= max_sum.
185 //
186 // The algorithm tries to minimize "error" (which is the relative error for one
187 // coefficient). Note however than in really broken cases, the error might be
188 // infinity and the factor zero.
189 //
190 // Note on the algorithm:
191 // - It only uses factors of the form 2^n (i.e. ldexp(1.0, n)) for simplicity.
192 // - The error will be zero in many practical instances. For example, if x
193 // contains only integers with low magnitude; or if x contains doubles whose
194 // exponents cover a small range.
195 // - It chooses the factor as high as possible under the given constraints, as
196 // a result the numbers produced may be large. To balance this, we recommend
197 // to divide the scaled integers by their gcd() which will result in no loss
198 // of precision and will help in many practical cases.
199 //
200 // TODO(user): incorporate the gcd computation here? The issue is that I am
201 // not sure if I just do factor /= gcd that round(x * factor) will be the same.
202 void GetBestScalingOfDoublesToInt64(const std::vector<double>& input,
203  int64 max_absolute_sum,
204  double* scaling_factor,
205  double* max_relative_coeff_error);

207 // Returns the scaling factor like above with the extra conditions:
208 // - The sum over i of min(0, round(factor * x[i])) >= -max_sum.
209 // - The sum over i of max(0, round(factor * x[i])) <= max_sum.
210 // For any possible values of the x[i] such that x[i] is in [lb[i], ub[i]].
211 double GetBestScalingOfDoublesToInt64(const std::vector<double>& input,
212  const std::vector<double>& lb,
213  const std::vector<double>& ub,
214  int64 max_absolute_sum);
215 // This computes:
216 //
217 // The max_relative_coeff_error, which is the maximum over all coeff of
