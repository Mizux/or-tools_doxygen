79 // [START solution_printer]
80 void PrintSolution(const DataModel& data, const RoutingIndexManager& manager,
81  const RoutingModel& routing, const Assignment& solution) {
82  int64 max_route_distance{0};
83  for (int vehicle_id = 0; vehicle_id < data.num_vehicles; ++vehicle_id) {
84  int64 index = routing.Start(vehicle_id);
85  LOG(INFO) << "Route for Vehicle " << vehicle_id << ":";
86  int64 route_distance{0};
87  std::stringstream route;
88  while (routing.IsEnd(index) == false) {
89  route << manager.IndexToNode(index).value() << " -> ";
90  int64 previous_index = index;
91  index = solution.Value(routing.NextVar(index));
92  route_distance += routing.GetArcCostForVehicle(previous_index, index,
93  int64{vehicle_id});
94  }
95  LOG(INFO) << route.str() << manager.IndexToNode(index).value();
96  LOG(INFO) << "Distance of the route: " << route_distance << "m";
97  max_route_distance = std::max(route_distance, max_route_distance);
98  }
99  LOG(INFO) << "Maximum of the route distances: " << max_route_distance << "m";
100  LOG(INFO) << "";
101  LOG(INFO) << "Problem solved in " << routing.solver()->wall_time() << "ms";
102 }

105 void VrpInitialRoutes() {
106  // Instantiate the data problem.
107  // [START data]
108  DataModel data;
109  // [END data]
110 
111  // Create Routing Index Manager
112  // [START index_manager]
113  RoutingIndexManager manager(data.distance_matrix.size(), data.num_vehicles,
114  data.depot);
115  // [END index_manager]
116 
117  // Create Routing Model.
118  // [START routing_model]
119  RoutingModel routing(manager);
120  // [END routing_model]
121 
122  // Create and register a transit callback.
123  // [START transit_callback]
124  const int transit_callback_index = routing.RegisterTransitCallback(
125  [&data, &manager](int64 from_index, int64 to_index) -> int64 {
126  // Convert from routing variable Index to distance matrix NodeIndex.
127  auto from_node = manager.IndexToNode(from_index).value();
128  auto to_node = manager.IndexToNode(to_index).value();
129  return data.distance_matrix[from_node][to_node];
130  });
131  // [END transit_callback]
132 
133  // Define cost of each arc.
134  // [START arc_cost]
135  routing.SetArcCostEvaluatorOfAllVehicles(transit_callback_index);
136  // [END arc_cost]
137 
138  // Add Distance constraint.
139  // [START distance_constraint]
140  routing.AddDimension(transit_callback_index, 0, 3000,
141  true, // start cumul to zero
142  "Distance");
143  routing.GetMutableDimension("Distance")->SetGlobalSpanCostCoefficient(100);
144  // [END distance_constraint]
145 
146  // Get initial solution from routes.
147  // [START print_initial_solution]
148  const Assignment* initial_solution =
149  routing.ReadAssignmentFromRoutes(data.initial_routes, true);
150  // Print initial solution on console.
151  LOG(INFO) << "Initial solution: ";
152  PrintSolution(data, manager, routing, *initial_solution);
153  // [END print_initial_solution]
154  // Setting first solution heuristic.
155  // [START parameters]
156  RoutingSearchParameters searchParameters = DefaultRoutingSearchParameters();
157  // [END parameters]
158 
159  // Solve from initial solution.
160  // [START solve]
161  const Assignment* solution = routing.SolveFromAssignmentWithParameters(
162  initial_solution, searchParameters);
163  // [START solve]
164 
165  // Print solution on console.
166  // [START print_solution]
167  LOG(INFO) << "";
168  LOG(INFO) << "Solution from search: ";
169  PrintSolution(data, manager, routing, *solution);
170  // [START print_solution]
171 }
