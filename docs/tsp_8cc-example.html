27 struct DataModel {
28  const std::vector<std::vector<int>> locations{
29  {4, 4}, {2, 0}, {8, 0}, {0, 1}, {1, 1}, {5, 2}, {7, 2}, {3, 3}, {6, 3},
30  {5, 5}, {8, 5}, {1, 6}, {2, 6}, {3, 7}, {6, 7}, {0, 8}, {7, 8},
31  };
32  const int num_vehicles = 1;
33  const RoutingIndexManager::NodeIndex depot{0};
34  DataModel() {
35  // Convert locations in meters using a city block dimension of 114m x 80m.
36  for (auto& it : const_cast<std::vector<std::vector<int>>&>(locations)) {
37  it[0] *= 114;
38  it[1] *= 80;
39  }
40  }

44 // [START manhattan_distance_matrix]
48 std::vector<std::vector<int64>> GenerateManhattanDistanceMatrix(
49  const std::vector<std::vector<int>>& locations) {
50  std::vector<std::vector<int64>> distances = std::vector<std::vector<int64>>(
51  locations.size(), std::vector<int64>(locations.size(), int64{0}));
52  for (int fromNode = 0; fromNode < locations.size(); fromNode++) {
53  for (int toNode = 0; toNode < locations.size(); toNode++) {
54  if (fromNode != toNode)
55  distances[fromNode][toNode] =
56  int64{std::abs(locations[toNode][0] - locations[fromNode][0]) +
57  std::abs(locations[toNode][1] - locations[fromNode][1])};
58  }
59  }
60  return distances;
61 }

64 // [START solution_printer]
69 void PrintSolution(const RoutingIndexManager& manager,
70  const RoutingModel& routing, const Assignment& solution) {
71  LOG(INFO) << "Objective: " << solution.ObjectiveValue();
72  // Inspect solution.
73  int64 index = routing.Start(0);
74  LOG(INFO) << "Route for Vehicle 0:";
75  int64 distance{0};
76  std::stringstream route;
77  while (routing.IsEnd(index) == false) {
78  route << manager.IndexToNode(index).value() << " -> ";
79  int64 previous_index = index;
80  index = solution.Value(routing.NextVar(index));
81  distance += routing.GetArcCostForVehicle(previous_index, index, int64{0});
82  }
83  LOG(INFO) << route.str() << manager.IndexToNode(index).value();
84  LOG(INFO) << "Distance of the route: " << distance << "m";
85  LOG(INFO) << "";
86  LOG(INFO) << "Advanced usage:";
87  LOG(INFO) << "Problem solved in " << routing.solver()->wall_time() << "ms";
88 }

91 void Tsp() {
92  // Instantiate the data problem.
93  // [START data]
94  DataModel data;
95  // [END data]
96 
97  // Create Routing Index Manager
98  // [START index_manager]
99  RoutingIndexManager manager(data.locations.size(), data.num_vehicles,
100  data.depot);
101  // [END index_manager]
102 
103  // Create Routing Model.
104  // [START routing_model]
105  RoutingModel routing(manager);
106  // [END routing_model]
107 
108  // Create and register a transit callback.
109  // [START transit_callback]
110  const auto distance_matrix = GenerateManhattanDistanceMatrix(data.locations);
111  const int transit_callback_index = routing.RegisterTransitCallback(
112  [&distance_matrix, &manager](int64 from_index, int64 to_index) -> int64 {
113  // Convert from routing variable Index to distance matrix NodeIndex.
114  auto from_node = manager.IndexToNode(from_index).value();
115  auto to_node = manager.IndexToNode(to_index).value();
116  return distance_matrix[from_node][to_node];
117  });
118  // [END transit_callback]
119 
120  // Define cost of each arc.
121  // [START arc_cost]
122  routing.SetArcCostEvaluatorOfAllVehicles(transit_callback_index);
123  // [END arc_cost]
124 
125  // Setting first solution heuristic.
126  // [START parameters]
127  RoutingSearchParameters searchParameters = DefaultRoutingSearchParameters();
128  searchParameters.set_first_solution_strategy(
129  FirstSolutionStrategy::PATH_CHEAPEST_ARC);
130  // [END parameters]
131 
132  // Solve the problem.
133  // [START solve]
134  const Assignment* solution = routing.SolveWithParameters(searchParameters);
135  // [END solve]
136 
137  // Print solution on console.
138  // [START print_solution]
139  PrintSolution(manager, routing, *solution);
140  // [END print_solution]
141 }
