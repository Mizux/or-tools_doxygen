46 // A simple class to enforce both an elapsed time limit and a deterministic time
47 // limit in the same thread as a program.
48 // The idea is to call LimitReached() as often as possible, until it returns
49 // false. The program should then abort as fast as possible.
50 //
51 // The deterministic limit is used to ensure reproductibility, as a consequence
52 // the deterministic time has to be advanced manually using the method
53 // AdvanceDeterministicTime().
54 //
55 // The instruction counter keeps track of number of executed cpu instructions.
56 // It uses Performance Monitoring Unit (PMU) counters to keep track of
57 // instruction count.
58 //
59 // The call itself is as fast as CycleClock::Now() + a few trivial instructions,
60 // unless the time_limit_use_instruction_count flag is set.
61 //
62 // The limit is very conservative: it returns true (i.e. the limit is reached)
63 // when current_time + max(T, ε) >= limit_time, where ε is a small constant (see
64 // TimeLimit::kSafetyBufferSeconds), and T is the maximum measured time interval
65 // between two consecutive calls to LimitReached() over the last kHistorySize
66 // calls (so that we only consider "recent" history).
67 // This is made so that the probability of actually exceeding the time limit is
68 // small, without aborting too early.
69 //
70 // The deterministic time limit can be logged at a more granular level: the
71 // method TimeLimit::AdvanceDeterministicTime takes an optional std::string
72 // argument: the name of a counter. In debug mode, the time limit object
73 // computes also the elapsed time for each named counter separately, and these
74 // values can be used to determine the coefficients for computing the
75 // deterministic duration from the number of operations. The values of the
76 // counters can be printed using TimeLimit::DebugString(). There is no API to
77 // access the values of the counters directly, because they do not exist in
78 // optimized mode.
79 //
80 // The basic steps for determining coefficients for the deterministic time are:
81 // 1. Run the code in debug mode to collect the values of the deterministic time
82 // counters. Unless the algorithm is different in optimized mode, the values
83 // of the deterministic counters in debug mode will be the same as in
84 // optimized mode.
85 // 2. Run the code in optimized mode to measure the real (CPU) time of the whole
86 // benchmark.
87 // 3. Determine the coefficients for deterministic time from the real time and
88 // the values of the deterministic counters, e. g. by solving the equations
89 // C_1*c_1 + C_2*c_2 + ... + C_N*c_N + Err = T
90 // where C_1 is the unknown coefficient for counter c_1, Err is the random
91 // measurement error and T is the measured real time. The equation can be
92 // solved e.g. using the least squares method.
93 //
94 // Note that in optimized mode, the counters are disabled for performance
95 // reasons, and calling AdvanceDeterministicTime(duration, counter_name) is
96 // equivalent to calling AdvanceDeterministicTime(duration).
97 //
98 // TODO(user): The expression "deterministic time" should be replaced with
99 // "number of operations" to avoid confusion with "real" time.

301 // Provides a way to nest time limits for algorithms where a certain part of
302 // the computation is bounded not just by the overall time limit, but also by a
303 // stricter time limit specific just for this particular part.
304 //
305 // This class takes a base time limit object (the overall time limit) and the
306 // part-specific time limit, and creates a new time limit object for the part.
307 // This new time limit object will expire when either the overall time limit
308 // expires or when the part-specific time limit expires.
309 //
310 // Example usage:
311 // TimeLimit overall_time_limit(...);
312 // NestedTimeLimit subalgorith_time_limit(&overall_time_limit,
313 // subalgorithm_limit_in_seconds,
314 // subalgorithm_deterministic_limit);
315 // RunTheSubalgorithm(subalgorithm_time_limit.GetTimeLimit());
316 //
317 // Note that remaining wall time in the base time limit is decreasing
318 // "automatically", but the deterministic time needs to be updated manually.
319 // This update is done only once, during the destruction of the nested time
320 // limit object. To track the deterministic time properly, the user must avoid
321 // modifying the base time limit object when a nested time limit exists.
322 //
323 // The nested time limits supports the external time limit condition in the
324 // sense, that if the overall time limit has an external boolean registered, the
325 // nested time limit object will use the same boolean value as an external time
326 // limit too.
327 class NestedTimeLimit {
328  public:
329  // Creates the nested time limit. Note that 'base_time_limit' must remain
330  // valid for the whole lifetime of the nested time limit object.
331  NestedTimeLimit(TimeLimit* base_time_limit, double limit_in_seconds,
332  double deterministic_limit);
333  // Updates elapsed deterministic time in the base time limit object.
334  ~NestedTimeLimit();
335 
336  // Creates a time limit object initialized from a base time limit and an
337  // object that provides methods max_time_in_seconds() and
338  // max_deterministic_time(). This method is designed specifically to work with
339  // solver parameter protos, e.g. BopParameters, MipParameters and
340  // SatParameters.
341  template <typename Parameters>
342  static std::unique_ptr<NestedTimeLimit> FromBaseTimeLimitAndParameters(
343  TimeLimit* time_limit, const Parameters& parameters) {
344  return absl::make_unique<NestedTimeLimit>(
345  time_limit, parameters.max_time_in_seconds(),
346  parameters.max_deterministic_time());
347  }
348 
349  // Returns a time limit object that represents the combination of the overall
350  // time limit and the part-specific time limit. The returned time limit object
351  // is owned by the nested time limit object that returns it, and it will
352  // remain valid until the nested time limit object is destroyed.
353  TimeLimit* GetTimeLimit() { return &time_limit_; }
354 
355  private:
356  TimeLimit* const base_time_limit_;
357  TimeLimit time_limit_;
358 
359  DISALLOW_COPY_AND_ASSIGN(NestedTimeLimit);
360 };

372 inline void TimeLimit::ResetTimers(double limit_in_seconds,
373  double deterministic_limit,
374  double instruction_limit) {
375  if (external_boolean_as_limit_ != nullptr) {
376  *external_boolean_as_limit_ = false;
377  }
378  elapsed_deterministic_time_ = 0.0;
379  deterministic_limit_ = deterministic_limit;
380  instruction_limit_ = instruction_limit;
381 
382  if (FLAGS_time_limit_use_usertime) {
383  user_timer_.Start();
384  limit_in_seconds_ = limit_in_seconds;
385  }
386 #ifdef HAS_PERF_SUBSYSTEM
387  if (FLAGS_time_limit_use_instruction_count) {
388  perf_subsystem_.CleanUp();
389  perf_subsystem_.AddEvent(GetInstructionRetiredEventName());
390  perf_subsystem_.StartCollecting();
391  }
392 #endif // HAS_PERF_SUBSYSTEM
393  start_ns_ = absl::GetCurrentTimeNanos();
394  last_ns_ = start_ns_;
395  limit_ns_ = limit_in_seconds >= 1e-9 * (kint64max - start_ns_)
396  ? kint64max
397  : static_cast<int64>(limit_in_seconds * 1e9) + start_ns_;
398 }

428 inline bool TimeLimit::LimitReached() {
429  if (external_boolean_as_limit_ != nullptr &&
430  external_boolean_as_limit_->load()) {
431  return true;
432  }
433 
434  if (GetDeterministicTimeLeft() <= 0.0) {
435  return true;
436  }
437 
438 #ifdef HAS_PERF_SUBSYSTEM
439  if (ReadInstructionCounter() >= instruction_limit_) {
440  return true;
441  }
442 #endif // HAS_PERF_SUBSYSTEM
443 
444  const int64 current_ns = absl::GetCurrentTimeNanos();
445  running_max_.Add(std::max(safety_buffer_ns_, current_ns - last_ns_));
446  last_ns_ = current_ns;
447  if (current_ns + running_max_.GetCurrentMax() >= limit_ns_) {
448  if (FLAGS_time_limit_use_usertime) {
449  // To avoid making many system calls, we only check the user time when
450  // the "absolute" time limit has been reached. Note that the user time
451  // should advance more slowly, so this is correct.
452  const double time_left_s = limit_in_seconds_ - user_timer_.Get();
453  if (time_left_s > kSafetyBufferSeconds) {
454  limit_ns_ = static_cast<int64>(time_left_s * 1e9) + last_ns_;
455  return false;
456  }
457  }
458 
459  // To ensure that future calls to LimitReached() will return true.
460  limit_ns_ = 0;
461  return true;
462  }
463  return false;
464 }

466 inline double TimeLimit::GetTimeLeft() const {
467  if (limit_ns_ == kint64max) return std::numeric_limits<double>::infinity();
468  const int64 delta_ns = limit_ns_ - absl::GetCurrentTimeNanos();
469  if (delta_ns < 0) return 0.0;
470  if (FLAGS_time_limit_use_usertime) {
471  return std::max(limit_in_seconds_ - user_timer_.Get(), 0.0);
472  } else {
473  return delta_ns * 1e-9;
474  }
475 }
