14 // This library solves 0-1 one-dimensional knapsack problems with fractional
15 // profits and weights using the branch and bound algorithm. Note that
16 // algorithms/knapsack_solver uses 'int64' for the profits and the weights.
17 // TODO(user): Merge this code with algorithms/knapsack_solver.
18 //
19 // Given n items, each with a profit and a weight and a knapsack of
20 // capacity c, the goal is to find a subset of the items which fits inside c
21 // and maximizes the total profit.
22 // Without loss of generality, profits and weights are assumed to be positive.
23 //
24 // From a mathematical point of view, the one-dimensional knapsack problem
25 // can be modeled by linear constraint:
26 // Sum(i:1..n)(weight_i * item_i) <= c,
27 // where item_i is a 0-1 integer variable.
28 // The goal is to maximize: Sum(i:1..n)(profit_i * item_i).
29 //
30 // Example Usage:
31 // std::vector<double> profits = {0, 0.5, 0.4, 1, 1, 1.1};
32 // std::vector<double> weights = {9, 6, 2, 1.5, 1.5, 1.5};

70 // ----- KnapsackItemForCuts -----
71 // KnapsackItemForCuts is a small struct to pair an item weight with its
72 // corresponding profit.
73 // The aim of the knapsack problem is to pack as many valuable items as
74 // possible. A straight forward heuristic is to take those with the greatest
75 // profit-per-unit-weight. This ratio is called efficiency in this
76 // implementation. So items will be grouped in vectors, and sorted by
77 // decreasing efficiency.
78 struct KnapsackItemForCuts {
79  KnapsackItemForCuts(int id, double weight, double profit)
80  : id(id), weight(weight), profit(profit) {}
81 
82  double GetEfficiency(double profit_max) const {
83  return (weight > 0) ? profit / weight : profit_max;
84  }
85 
86  // The 'id' field is used to retrieve the initial item in order to
87  // communicate with other propagators and state.
88  const int id;
89  const double weight;
90  const double profit;
91 };

142 // ----- KnapsackSearchPathForCuts -----
143 // KnapsackSearchPathForCuts is a small class used to represent the path between
144 // a node to another node in the search tree.
145 // As the solution state is not stored for each search node, the state should
146 // be rebuilt at each node. One simple solution is to apply all decisions
147 // between the node 'to' and the root. This can be computed in
148 // O(number_of_items).
149 //
150 // However, it is possible to achieve better average complexity. Two
151 // consecutively explored nodes are usually close enough (i.e., much less than
152 // number_of_items) to benefit from an incremental update from the node
153 // 'from' to the node 'to'.
154 //
155 // The 'via' field is the common parent of 'from' field and 'to' field.
156 // So the state can be built by reverting all decisions from 'from' to 'via'
157 // and then applying all decisions from 'via' to 'to'.
158 class KnapsackSearchPathForCuts {
159  public:
160  KnapsackSearchPathForCuts(const KnapsackSearchNodeForCuts* from,
161  const KnapsackSearchNodeForCuts* to);
162 
163  KnapsackSearchPathForCuts(const KnapsackSearchPathForCuts&) = delete;
164  KnapsackSearchPathForCuts& operator=(const KnapsackSearchPathForCuts&) =
165  delete;
166 
167  void Init();
168  const KnapsackSearchNodeForCuts& from() const { return *from_; }
169  const KnapsackSearchNodeForCuts& via() const { return *via_; }
170  const KnapsackSearchNodeForCuts& to() const { return *to_; }
171 
172  private:
173  const KnapsackSearchNodeForCuts* from_;
174  const KnapsackSearchNodeForCuts* via_; // Computed in 'Init'.
175  const KnapsackSearchNodeForCuts* to_;
176 };

213 // ----- KnapsackPropagatorForCuts -----
214 // KnapsackPropagatorForCuts is used to enforce a capacity constraint.
215 // It is supposed to compute profit lower and upper bounds, and get the next
216 // item to select, it can be seen as a 0-1 Knapsack solver. The most efficient
217 // way to compute the upper bound is to iterate on items in
218 // profit-per-unit-weight decreasing order. The break item is commonly defined
219 // as the first item for which there is not enough remaining capacity. Selecting
220 // this break item as the next-item-to-assign usually gives the best results
221 // (see Greenberg & Hegerich).
222 //
223 // This is exactly what is implemented in this class.
224 //
225 // It is possible to compute a better profit lower bound almost for free. During
226 // the scan to find the break element all unbound items are added just as if
227 // they were part of the current solution. This is used in both
228 // ComputeProfitBounds() and CopyCurrentSolution(). For incrementality reasons,
229 // the ith item should be accessible in O(1). That's the reason why the item
230 // vector has to be duplicated 'sorted_items_'.
231 class KnapsackPropagatorForCuts {
232  public:
233  explicit KnapsackPropagatorForCuts(const KnapsackStateForCuts* state);
234  ~KnapsackPropagatorForCuts();
235 
236  KnapsackPropagatorForCuts(const KnapsackPropagatorForCuts&) = delete;
237  KnapsackPropagatorForCuts& operator=(const KnapsackPropagatorForCuts&) =
238  delete;
239 
240  // Initializes the data structure and then calls InitPropagator.
241  void Init(const std::vector<double>& profits,
242  const std::vector<double>& weights, double capacity);
243 
244  // Updates data structure. Returns false on failure.
245  bool Update(bool revert, const KnapsackAssignmentForCuts& assignment);
246  // ComputeProfitBounds should set 'profit_lower_bound_' and
247  // 'profit_upper_bound_' which are constraint specific.
248  void ComputeProfitBounds();
249  // Returns the id of next item to assign.
250  // Returns kNoSelection when all items are bound.
251  int GetNextItemId() const { return break_item_id_; }
252 
253  double current_profit() const { return current_profit_; }
254  double profit_lower_bound() const { return profit_lower_bound_; }
255  double profit_upper_bound() const { return profit_upper_bound_; }
256 
257  // Copies the current state into 'solution'.
258  // All unbound items are set to false (i.e. not in the knapsack).
259  void CopyCurrentStateToSolution(std::vector<bool>* solution) const;
260 
261  // Initializes the propagator. This method is called by Init() after filling
262  // the fields defined in this class.
263  void InitPropagator();
264 
265  const KnapsackStateForCuts& state() const { return *state_; }
266  const std::vector<KnapsackItemForCutsPtr>& items() const { return items_; }
267 
268  void set_profit_lower_bound(double profit) { profit_lower_bound_ = profit; }
269  void set_profit_upper_bound(double profit) { profit_upper_bound_ = profit; }
270 
271  private:
272  // An obvious additional profit upper bound corresponds to the linear
273  // relaxation: remaining_capacity * efficiency of the break item.
274  // It is possible to do better in O(1), using Martello-Toth bound U2.
275  // The main idea is to enforce integrality constraint on the break item,
276  // i.e. either the break item is part of the solution, or it is not.
277  // So basically the linear relaxation is done on the item before the break
278  // item, or the one after the break item. This is what GetAdditionalProfit
279  // method implements.
280  double GetAdditionalProfitUpperBound(double remaining_capacity,
281  int break_item_id) const;
282 
283  double capacity_;
284  double consumed_capacity_;
285  int break_item_id_;
286  std::vector<KnapsackItemForCutsPtr> sorted_items_;
287  double profit_max_;
288  std::vector<KnapsackItemForCutsPtr> items_;
289  double current_profit_;
290  double profit_lower_bound_;
291  double profit_upper_bound_;
292  const KnapsackStateForCuts* const state_;
293 };

372  KnapsackStateForCuts state_;
