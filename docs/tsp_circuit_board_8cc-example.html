81 // [START distance_matrix]
82 // @brief Generate distance matrix.
83 std::vector<std::vector<int64>> ComputeEuclideanDistanceMatrix(
84  const std::vector<std::vector<int>>& locations) {
85  std::vector<std::vector<int64>> distances = std::vector<std::vector<int64>>(
86  locations.size(), std::vector<int64>(locations.size(), int64{0}));
87  for (int fromNode = 0; fromNode < locations.size(); fromNode++) {
88  for (int toNode = 0; toNode < locations.size(); toNode++) {
89  if (fromNode != toNode)
90  distances[fromNode][toNode] = static_cast<int64>(
91  std::hypot((locations[toNode][0] - locations[fromNode][0]),
92  (locations[toNode][1] - locations[fromNode][1])));
93  }
94  }
95  return distances;
96 }

99 // [START solution_printer]
104 void PrintSolution(const RoutingIndexManager& manager,
105  const RoutingModel& routing, const Assignment& solution) {
106  LOG(INFO) << "Objective: " << solution.ObjectiveValue();
107  // Inspect solution.
108  int64 index = routing.Start(0);
109  LOG(INFO) << "Route:";
110  int64 distance{0};
111  std::stringstream route;
112  while (routing.IsEnd(index) == false) {
113  route << manager.IndexToNode(index).value() << " -> ";
114  int64 previous_index = index;
115  index = solution.Value(routing.NextVar(index));
116  distance += routing.GetArcCostForVehicle(previous_index, index, int64{0});
117  }
118  LOG(INFO) << route.str() << manager.IndexToNode(index).value();
119  LOG(INFO) << "Route distance: " << distance << "miles";
120  LOG(INFO) << "";
121  LOG(INFO) << "Advanced usage:";
122  LOG(INFO) << "Problem solved in " << routing.solver()->wall_time() << "ms";
123 }

126 void Tsp() {
127  // Instantiate the data problem.
128  // [START data]
129  DataModel data;
130  // [END data]
131 
132  // Create Routing Index Manager
133  // [START index_manager]
134  RoutingIndexManager manager(data.locations.size(), data.num_vehicles,
135  data.depot);
136  // [END index_manager]
137 
138  // Create Routing Model.
139  // [START routing_model]
140  RoutingModel routing(manager);
141  // [END routing_model]
142 
143  // [START transit_callback]
144  const auto distance_matrix = ComputeEuclideanDistanceMatrix(data.locations);
145  const int transit_callback_index = routing.RegisterTransitCallback(
146  [&distance_matrix, &manager](int64 from_index, int64 to_index) -> int64 {
147  // Convert from routing variable Index to distance matrix NodeIndex.
148  auto from_node = manager.IndexToNode(from_index).value();
149  auto to_node = manager.IndexToNode(to_index).value();
150  return distance_matrix[from_node][to_node];
151  });
152  // [END transit_callback]
153 
154  // Define cost of each arc.
155  // [START arc_cost]
156  routing.SetArcCostEvaluatorOfAllVehicles(transit_callback_index);
157  // [END arc_cost]
158 
159  // Setting first solution heuristic.
160  // [START parameters]
161  RoutingSearchParameters searchParameters = DefaultRoutingSearchParameters();
162  searchParameters.set_first_solution_strategy(
163  FirstSolutionStrategy::PATH_CHEAPEST_ARC);
164  // [END parameters]
165 
166  // Solve the problem.
167  // [START solve]
168  const Assignment* solution = routing.SolveWithParameters(searchParameters);
169  // [END solve]
170 
171  // Print solution on console.
172  // [START print_solution]
173  PrintSolution(manager, routing, *solution);
174  // [END print_solution]
175 }
