55 // General method to read expected_num_records from a file. If
56 // expected_num_records is -1, then reads all records from the file. If not,
57 // dies if the file doesn't contain exactly expected_num_records.
58 template <typename Proto>
59 std::vector<Proto> ReadNumRecords(File* file, int expected_num_records) {
60  recordio::RecordReader reader(file);
61  std::vector<Proto> protos;
62  Proto proto;
63  int num_read = 0;
64  while (num_read != expected_num_records &&
65  reader.ReadProtocolMessage(&proto)) {
66  protos.push_back(proto);
67  ++num_read;
68  }
69 
70  if (expected_num_records >= 0) {
71  CHECK_EQ(num_read, expected_num_records)
72  << "There were less than the expected " << expected_num_records
73  << " in the file.";
74  }
75 
76  return protos;
77 }
