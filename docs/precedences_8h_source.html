35 // This class implement a propagator on simple inequalities between integer
36 // variables of the form (i1 + offset <= i2). The offset can be constant or
37 // given by the value of a third integer variable. Offsets can also be negative.
38 //
39 // The algorithm work by mapping the problem onto a graph where the edges carry
40 // the offset and the nodes correspond to one of the two bounds of an integer
41 // variable (lower_bound or -upper_bound). It then find the fixed point using an
42 // incremental variant of the Bellman-Ford(-Tarjan) algorithm.
43 //
44 // This is also known as an "integer difference logic theory" in the SMT world.
45 // Another word is "separation logic".
46 class PrecedencesPropagator : public SatPropagator, PropagatorInterface {
47  public:
48  explicit PrecedencesPropagator(Model* model)
49  : SatPropagator("PrecedencesPropagator"),
50  trail_(model->GetOrCreate<Trail>()),
51  integer_trail_(model->GetOrCreate<IntegerTrail>()),
52  watcher_(model->GetOrCreate<GenericLiteralWatcher>()),
53  watcher_id_(watcher_->Register(this)) {
54  model->GetOrCreate<SatSolver>()->AddPropagator(this);
55  integer_trail_->RegisterWatcher(&modified_vars_);
56  watcher_->SetPropagatorPriority(watcher_id_, 0);
57  }

92  // Finds all the IntegerVariable that are "after" at least two of the
93  // IntegerVariable in vars. Returns a vector of these precedences relation
94  // sorted by IntegerPrecedences.var so that it is efficient to find all the
95  // IntegerVariable "before" another one.
96  //
97  // Note that we only consider direct precedences here. Given our usage, it may
98  // be better to compute the full reachability in the precedence graph, but in
99  // pratice that may be too slow.
100  //
101  // Note that the IntegerVariable in the vector are also returned in
102  // topological order for a more efficient propagation in
103  // DisjunctivePrecedences::Propagate() where this is used.
104  //
105  // Important: For identical vars, the entry are sorted by index.
106  struct IntegerPrecedences {
107  int index; // position in vars.
108  IntegerVariable var; // An IntegerVariable that is >= to vars[index].
109  int arc_index; // Used by AddPrecedenceReason().
110  IntegerValue offset; // we have: vars[index] + offset <= var
111  };

114  void AddPrecedenceReason(int arc_index, IntegerValue min_offset,

181  // The core algorithm implementation is split in these functions. One must
182  // first call InitializeBFQueueWithModifiedNodes() that will push all the
183  // IntegerVariable whose lower bound has been modified since the last call.
184  // Then, BellmanFordTarjan() will take care of all the propagation and returns
185  // false in case of conflict. Internally, it uses DisassembleSubtree() which
186  // is the Tarjan variant to detect a possible positive cycle. Before exiting,
187  // it will call CleanUpMarkedArcsAndParents().
188  //
189  // The Tarjan version of the Bellam-Ford algorithm is really nice in our
190  // context because it was really easy to make it incremental. Moreover, it
191  // supports batch increment!
192  //
193  // This implementation is kind of unique because of our context and the fact
194  // that it is incremental, but a good reference is "Negative-cycle detection
195  // algorithms", Boris V. Cherkassky, Andrew V. Goldberg, 1996,
196  // http://people.cs.nctu.edu.tw/~tjshen/doc/ne.pdf
197  void InitializeBFQueueWithModifiedNodes();
198  bool BellmanFordTarjan(Trail* trail);
199  bool DisassembleSubtree(int source, int target,
200  std::vector<bool>* can_be_skipped);
201  void AnalyzePositiveCycle(ArcIndex first_arc, Trail* trail,
202  std::vector<Literal>* must_be_all_true,
203  std::vector<Literal>* literal_reason,
204  std::vector<IntegerLiteral>* integer_reason);
205  void CleanUpMarkedArcsAndParents();

246  gtl::ITIVector<IntegerVariable, int> var_to_last_index_;
247  struct SortedVar {
248  IntegerVariable var;

429 // is_le <=> (a + offset <= b).
430 inline std::function<void(Model*)> ReifiedLowerOrEqualWithOffset(
431  IntegerVariable a, IntegerVariable b, int64 offset, Literal is_le) {
432  return [=](Model* model) {
433  PrecedencesPropagator* p = model->GetOrCreate<PrecedencesPropagator>();
434  p->AddConditionalPrecedenceWithOffset(a, b, IntegerValue(offset), is_le);
435 
436  // The negation of (a + offset <= b) is (a + offset > b) which can be
437  // rewritten as (b + 1 - offset <= a).
438  p->AddConditionalPrecedenceWithOffset(b, a, IntegerValue(1 - offset),
439  is_le.Negated());
440  };
441 }

443 // is_eq <=> (a == b).
444 inline std::function<void(Model*)> ReifiedEquality(IntegerVariable a,
445  IntegerVariable b,
446  Literal is_eq) {
447  return [=](Model* model) {
448  // We creates two extra Boolean variables in this case.
449  //
450  // TODO(user): Avoid creating them if we already have some literal that
451  // have the same meaning. For instance if a client also wanted to know if
452  // a <= b, he would have called ReifiedLowerOrEqualWithOffset() directly.
453  const Literal is_le = Literal(model->Add(NewBooleanVariable()), true);
454  const Literal is_ge = Literal(model->Add(NewBooleanVariable()), true);
455  model->Add(ReifiedBoolAnd({is_le, is_ge}, is_eq));
456  model->Add(ReifiedLowerOrEqualWithOffset(a, b, 0, is_le));
457  model->Add(ReifiedLowerOrEqualWithOffset(b, a, 0, is_ge));
458  };
459 }

461 // is_eq <=> (a + offset == b).
462 inline std::function<void(Model*)> ReifiedEqualityWithOffset(IntegerVariable a,
463  IntegerVariable b,
464  int64 offset,
465  Literal is_eq) {
466  return [=](Model* model) {
467  // We creates two extra Boolean variables in this case.
468  //
469  // TODO(user): Avoid creating them if we already have some literal that
470  // have the same meaning. For instance if a client also wanted to know if
471  // a <= b, he would have called ReifiedLowerOrEqualWithOffset() directly.
472  const Literal is_le = Literal(model->Add(NewBooleanVariable()), true);
473  const Literal is_ge = Literal(model->Add(NewBooleanVariable()), true);
474  model->Add(ReifiedBoolAnd({is_le, is_ge}, is_eq));

480 // a != b.
481 inline std::function<void(Model*)> NotEqual(IntegerVariable a,
482  IntegerVariable b) {
483  return [=](Model* model) {
484  // We have two options (is_gt or is_lt) and one must be true.
485  const Literal is_lt = Literal(model->Add(NewBooleanVariable()), true);
486  const Literal is_gt = is_lt.Negated();
487  model->Add(ConditionalLowerOrEqualWithOffset(a, b, 1, is_lt));
488  model->Add(ConditionalLowerOrEqualWithOffset(b, a, 1, is_gt));
489  };
