73 // Returns the ith element of the strategy S^univ proposed by M. Luby et al. in
74 // Optimal Speedup of Las Vegas Algorithms, Information Processing Letters 1993.
75 // This is used to decide the number of conflicts allowed before the next
76 // restart. This method, used by most SAT solvers, is usually referenced as
77 // Luby.
78 // Returns 2^{k-1} when i == 2^k - 1
79 // and SUniv(i - 2^{k-1} + 1) when 2^{k-1} <= i < 2^k - 1.
80 // The sequence is defined for i > 0 and starts with:
81 // {1, 1, 2, 1, 1, 2, 4, 1, 1, 2, 1, 1, 2, 4, 8, ...}
82 inline int SUniv(int i) {
83  DCHECK_GT(i, 0);
84  while (i > 2) {
85  const int most_significant_bit_position =
86  MostSignificantBitPosition64(i + 1);
87  if ((1 << most_significant_bit_position) == i + 1) {
88  return 1 << (most_significant_bit_position - 1);
89  }
90  i -= (1 << most_significant_bit_position) - 1;
91  }
92  return 1;
93 }
