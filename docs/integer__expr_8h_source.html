33 // A really basic implementation of an upper-bounded sum of integer variables.
34 // The complexity is in O(num_variables) at each propagation.
35 //
36 // Note that we assume that there can be NO integer overflow. This must be
37 // checked at model validation time before this is even created.
38 //
39 // TODO(user): If one has many such constraint, it will be more efficient to
40 // propagate all of them at once rather than doing it one at the time.
41 //
42 // TODO(user): Explore tree structure to get a log(n) complexity.
43 //
44 // TODO(user): When the variables are Boolean, use directly the pseudo-Boolean
45 // constraint implementation. But we do need support for enforcement literals
46 // there.
47 class IntegerSumLE : public PropagatorInterface {
48  public:
49  // If refied_literal is kNoLiteralIndex then this is a normal constraint,
50  // otherwise we enforce the implication refied_literal => constraint is true.
51  // Note that we don't do the reverse implication here, it is usually done by
52  // another IntegerSumLE constraint on the negated variables.
53  IntegerSumLE(const std::vector<Literal>& enforcement_literals,
54  const std::vector<IntegerVariable>& vars,
55  const std::vector<IntegerValue>& coeffs,
56  IntegerValue upper_bound, Model* model);
57 
58  // We propagate:
59  // - If the sum of the individual lower-bound is > upper_bound, we fail.
60  // - For all i, upper-bound of i
61  // <= upper_bound - Sum {individual lower-bound excluding i).
62  bool Propagate() final;
63  void RegisterWith(GenericLiteralWatcher* watcher);
64 
65  private:
66  // Fills integer_reason_ with all the current lower_bounds. The real
67  // explanation may require removing one of them, but as an optimization, we
68  // always keep all the IntegerLiteral in integer_reason_, and swap them as
69  // needed just before pushing something.
70  void FillIntegerReason();
71 
72  const std::vector<Literal> enforcement_literals_;
73  const IntegerValue upper_bound_;
74 
75  Trail* trail_;
76  IntegerTrail* integer_trail_;
77  RevIntegerValueRepository* rev_integer_value_repository_;
78 
79  // Reversible sum of the lower bound of the fixed variables.
80  bool is_registered_ = false;
81  IntegerValue rev_lb_fixed_vars_;
82 
83  // Reversible number of fixed variables.
84  int rev_num_fixed_vars_;
85 
86  // Those vectors are shuffled during search to ensure that the variables
87  // (resp. coefficients) contained in the range [0, rev_num_fixed_vars_) of
88  // vars_ (resp. coeffs_) are fixed (resp. belong to fixed variables).
89  std::vector<IntegerVariable> vars_;
90  std::vector<IntegerValue> coeffs_;
91  std::vector<IntegerValue> max_variations_;
92 
93  std::vector<Literal> literal_reason_;
94 
95  // Parallel vectors.
96  std::vector<IntegerLiteral> integer_reason_;
97  std::vector<IntegerValue> reason_coeffs_;
98 
99  DISALLOW_COPY_AND_ASSIGN(IntegerSumLE);
100 };

102 // A min (resp max) contraint of the form min == MIN(vars) can be decomposed
103 // into two inequalities:
104 // 1/ min <= MIN(vars), which is the same as for all v in vars, "min <= v".
105 // This can be taken care of by the LowerOrEqual(min, v) constraint.
106 // 2/ min >= MIN(vars).
107 //
108 // And in turn, 2/ can be decomposed in:
109 // a) lb(min) >= lb(MIN(vars)) = MIN(lb(var));
110 // b) ub(min) >= ub(MIN(vars)) and we can't propagate anything here unless
111 // there is just one possible variable 'v' that can be the min:
112 // for all u != v, lb(u) > ub(min);
113 // In this case, ub(min) >= ub(v).
114 //
115 // This constraint take care of a) and b). That is:
116 // - If the min of the lower bound of the vars increase, then the lower bound of
117 // the min_var will be >= to it.
118 // - If there is only one candidate for the min, then if the ub(min) decrease,
119 // the ub of the only candidate will be <= to it.
120 //
121 // Complexity: This is a basic implementation in O(num_vars) on each call to
122 // Propagate(), which will happen each time one or more variables in vars_
123 // changed.
124 //
125 // TODO(user): Implement a more efficient algorithm when the need arise.
126 class MinPropagator : public PropagatorInterface {
127  public:
128  MinPropagator(const std::vector<IntegerVariable>& vars,
129  IntegerVariable min_var, IntegerTrail* integer_trail);
130 
131  bool Propagate() final;
132  void RegisterWith(GenericLiteralWatcher* watcher);
133 
134  private:
135  const std::vector<IntegerVariable> vars_;
136  const IntegerVariable min_var_;
137  IntegerTrail* integer_trail_;
138 
139  std::vector<IntegerLiteral> integer_reason_;
140 
141  DISALLOW_COPY_AND_ASSIGN(MinPropagator);
142 };

254 // Weighted sum <= constant.
255 template <typename VectorInt>
256 inline std::function<void(Model*)> WeightedSumLowerOrEqual(
257  const std::vector<IntegerVariable>& vars, const VectorInt& coefficients,
258  int64 upper_bound) {
259  // Special cases.
260  CHECK_GE(vars.size(), 1);
261  if (vars.size() == 1) {
262  const int64 c = coefficients[0];
263  CHECK_NE(c, 0);
264  if (c > 0) {
265  return LowerOrEqual(vars[0], upper_bound / c);
266  } else {
267  const int64 ceil_c = (upper_bound + c + 1) / c;
268  return GreaterOrEqual(vars[0], ceil_c);
269  }
270  }
271  if (vars.size() == 2 && (coefficients[0] == 1 || coefficients[0] == -1) &&
272  (coefficients[1] == 1 || coefficients[1] == -1)) {
273  return Sum2LowerOrEqual(
274  coefficients[0] == 1 ? vars[0] : NegationOf(vars[0]),
275  coefficients[1] == 1 ? vars[1] : NegationOf(vars[1]), upper_bound);
276  }
277  if (vars.size() == 3 && (coefficients[0] == 1 || coefficients[0] == -1) &&
278  (coefficients[1] == 1 || coefficients[1] == -1) &&
279  (coefficients[2] == 1 || coefficients[2] == -1)) {
280  return Sum3LowerOrEqual(
281  coefficients[0] == 1 ? vars[0] : NegationOf(vars[0]),
282  coefficients[1] == 1 ? vars[1] : NegationOf(vars[1]),
283  coefficients[2] == 1 ? vars[2] : NegationOf(vars[2]), upper_bound);
284  }
285 
286  return [=](Model* model) {
287  // We split large constraints into a square root number of parts.
288  // This is to avoid a bad complexity while propagating them since our
289  // algorithm is not in O(num_changes).
290  //
291  // TODO(user): Alternatively, we could use a O(num_changes) propagation (a
292  // bit tricky to implement), or a decomposition into a tree with more than
293  // one level. Both requires experimentations.
294  //
295  // TODO(user): If the initial constraint was an equalilty we will create
296  // the "intermediate" variable twice where we could have use the same for
297  // both direction. Improve?
298  const int num_vars = vars.size();
299  if (num_vars > 100) {
300  std::vector<IntegerVariable> bucket_sum_vars;
301 
302  std::vector<IntegerVariable> local_vars;
303  std::vector<IntegerValue> local_coeffs;
304 
305  int i = 0;
306  const int num_buckets = static_cast<int>(std::round(std::sqrt(num_vars)));
307  for (int b = 0; b < num_buckets; ++b) {
308  local_vars.clear();
309  local_coeffs.clear();
310  int64 bucket_lb = 0;
311  int64 bucket_ub = 0;
312  const int limit = num_vars * (b + 1);
313  for (; i * num_buckets < limit; ++i) {
314  local_vars.push_back(vars[i]);
315  local_coeffs.push_back(IntegerValue(coefficients[i]));
316  const int64 term1 = model->Get(LowerBound(vars[i])) * coefficients[i];
317  const int64 term2 = model->Get(UpperBound(vars[i])) * coefficients[i];
318  bucket_lb += std::min(term1, term2);
319  bucket_ub += std::max(term1, term2);
320  }
321 
322  const IntegerVariable bucket_sum =
323  model->Add(NewIntegerVariable(bucket_lb, bucket_ub));
324  bucket_sum_vars.push_back(bucket_sum);
325  local_vars.push_back(bucket_sum);
326  local_coeffs.push_back(IntegerValue(-1));
327  IntegerSumLE* constraint = new IntegerSumLE(
328  {}, local_vars, local_coeffs, IntegerValue(0), model);
329  constraint->RegisterWith(model->GetOrCreate<GenericLiteralWatcher>());
330  model->TakeOwnership(constraint);
331  }
332 
333  // Create the root-level sum.
334  local_vars.clear();
335  local_coeffs.clear();
336  for (const IntegerVariable var : bucket_sum_vars) {
337  local_vars.push_back(var);
338  local_coeffs.push_back(IntegerValue(1));
339  }
340  IntegerSumLE* constraint = new IntegerSumLE(
341  {}, local_vars, local_coeffs, IntegerValue(upper_bound), model);
342  constraint->RegisterWith(model->GetOrCreate<GenericLiteralWatcher>());
343  model->TakeOwnership(constraint);
344  return;
345  }
346 
347  IntegerSumLE* constraint = new IntegerSumLE(
348  {}, vars,
349  std::vector<IntegerValue>(coefficients.begin(), coefficients.end()),
350  IntegerValue(upper_bound), model);
351  constraint->RegisterWith(model->GetOrCreate<GenericLiteralWatcher>());
352  model->TakeOwnership(constraint);
353  };
354 }

356 // Weighted sum >= constant.
357 template <typename VectorInt>
358 inline std::function<void(Model*)> WeightedSumGreaterOrEqual(
359  const std::vector<IntegerVariable>& vars, const VectorInt& coefficients,
360  int64 lower_bound) {
361  // We just negate everything and use an <= constraints.
362  std::vector<int64> negated_coeffs(coefficients.begin(), coefficients.end());
363  for (int64& ref : negated_coeffs) ref = -ref;
364  return WeightedSumLowerOrEqual(vars, negated_coeffs, -lower_bound);
365 }

378 // enforcement_literals => sum <= upper_bound
379 template <typename VectorInt>
380 inline std::function<void(Model*)> ConditionalWeightedSumLowerOrEqual(
381  const std::vector<Literal>& enforcement_literals,
382  const std::vector<IntegerVariable>& vars, const VectorInt& coefficients,
383  int64 upper_bound) {
384  // Special cases.
385  CHECK_GE(vars.size(), 1);
386  if (vars.size() == 1) {
387  CHECK_NE(coefficients[0], 0);
388  if (coefficients[0] > 0) {
389  return Implication(
390  enforcement_literals,
391  IntegerLiteral::LowerOrEqual(
392  vars[0], IntegerValue(upper_bound / coefficients[0])));
393  } else {
394  return Implication(
395  enforcement_literals,
396  IntegerLiteral::GreaterOrEqual(
397  vars[0], IntegerValue(upper_bound / coefficients[0])));
398  }
399  }
400  if (vars.size() == 2 && (coefficients[0] == 1 || coefficients[0] == -1) &&
401  (coefficients[1] == 1 || coefficients[1] == -1)) {
402  return ConditionalSum2LowerOrEqual(
403  coefficients[0] == 1 ? vars[0] : NegationOf(vars[0]),
404  coefficients[1] == 1 ? vars[1] : NegationOf(vars[1]), upper_bound,
405  enforcement_literals);
406  }
407  if (vars.size() == 3 && (coefficients[0] == 1 || coefficients[0] == -1) &&
408  (coefficients[1] == 1 || coefficients[1] == -1) &&
409  (coefficients[2] == 1 || coefficients[2] == -1)) {
410  return ConditionalSum3LowerOrEqual(
411  coefficients[0] == 1 ? vars[0] : NegationOf(vars[0]),
412  coefficients[1] == 1 ? vars[1] : NegationOf(vars[1]),
413  coefficients[2] == 1 ? vars[2] : NegationOf(vars[2]), upper_bound,
414  enforcement_literals);
415  }
416  return [=](Model* model) {
417  IntegerSumLE* constraint = new IntegerSumLE(
418  enforcement_literals, vars,
419  std::vector<IntegerValue>(coefficients.begin(), coefficients.end()),
420  IntegerValue(upper_bound), model);
421  constraint->RegisterWith(model->GetOrCreate<GenericLiteralWatcher>());
422  model->TakeOwnership(constraint);
423  };
424 }

426 // enforcement_literals => sum >= lower_bound
427 template <typename VectorInt>
428 inline std::function<void(Model*)> ConditionalWeightedSumGreaterOrEqual(
429  const std::vector<Literal>& enforcement_literals,
430  const std::vector<IntegerVariable>& vars, const VectorInt& coefficients,
431  int64 lower_bound) {
432  // We just negate everything and use an <= constraint.
433  std::vector<int64> negated_coeffs(coefficients.begin(), coefficients.end());
434  for (int64& ref : negated_coeffs) ref = -ref;
435  return ConditionalWeightedSumLowerOrEqual(enforcement_literals, vars,
436  negated_coeffs, -lower_bound);

499 // Model-based function to create an IntegerVariable that corresponds to the
500 // given weighted sum of other IntegerVariables.
501 //
502 // Note that this is templated so that it can seamlessly accept std::vector<int>
503 // or std::vector<int64>.
504 //
505 // TODO(user): invert the coefficients/vars arguments.
506 template <typename VectorInt>
507 inline std::function<IntegerVariable(Model*)> NewWeightedSum(
508  const VectorInt& coefficients, const std::vector<IntegerVariable>& vars) {
509  return [=](Model* model) {
510  std::vector<IntegerVariable> new_vars = vars;
511  // To avoid overflow in the FixedWeightedSum() constraint, we need to
512  // compute the basic bounds on the sum.
513  //
514  // TODO(user): deal with overflow here too!
515  int64 sum_lb(0);
516  int64 sum_ub(0);
517  for (int i = 0; i < new_vars.size(); ++i) {
518  if (coefficients[i] > 0) {
519  sum_lb += coefficients[i] * model->Get(LowerBound(new_vars[i]));
520  sum_ub += coefficients[i] * model->Get(UpperBound(new_vars[i]));
521  } else {
522  sum_lb += coefficients[i] * model->Get(UpperBound(new_vars[i]));
523  sum_ub += coefficients[i] * model->Get(LowerBound(new_vars[i]));
524  }
525  }
526 
527  const IntegerVariable sum = model->Add(NewIntegerVariable(sum_lb, sum_ub));
528  new_vars.push_back(sum);
529  std::vector<int64> new_coeffs(coefficients.begin(), coefficients.end());
530  new_coeffs.push_back(-1);
531  model->Add(FixedWeightedSum(new_vars, new_coeffs, 0));
532  return sum;
533  };
534 }

619 // Adds the constraint: a * b = p.
620 inline std::function<void(Model*)> ProductConstraint(IntegerVariable a,
621  IntegerVariable b,
622  IntegerVariable p) {
623  return [=](Model* model) {
624  IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
625  if (a == b) {
626  if (model->Get(LowerBound(a)) >= 0) {
627  RegisterAndTransferOwnership(model,
628  new SquarePropagator(a, p, integer_trail));
629  } else if (model->Get(UpperBound(a)) <= 0) {
630  RegisterAndTransferOwnership(
631  model, new SquarePropagator(NegationOf(a), p, integer_trail));
632  } else {
633  LOG(FATAL) << "Not supported";
634  }
635  } else if (model->Get(LowerBound(a)) >= 0 &&
636  model->Get(LowerBound(b)) >= 0) {
637  RegisterAndTransferOwnership(
638  model, new PositiveProductPropagator(a, b, p, integer_trail));
639  } else if (model->Get(LowerBound(a)) >= 0 &&
640  model->Get(UpperBound(b)) <= 0) {
641  RegisterAndTransferOwnership(
642  model, new PositiveProductPropagator(a, NegationOf(b), NegationOf(p),
643  integer_trail));
644  } else if (model->Get(UpperBound(a)) <= 0 &&
645  model->Get(LowerBound(b)) >= 0) {
646  RegisterAndTransferOwnership(
647  model, new PositiveProductPropagator(NegationOf(a), b, NegationOf(p),
648  integer_trail));
649  } else if (model->Get(UpperBound(a)) <= 0 &&
650  model->Get(UpperBound(b)) <= 0) {
651  RegisterAndTransferOwnership(
652  model, new PositiveProductPropagator(NegationOf(a), NegationOf(b), p,
653  integer_trail));
654  } else {
655  LOG(FATAL) << "Not supported";
656  }
657  };
658 }
