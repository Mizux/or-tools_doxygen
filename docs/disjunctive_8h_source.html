52 // Helper class to compute the end-min of a set of tasks given their start-min
53 // and duration-min. In Petr Vilim's PhD "Global Constraints in Scheduling",
54 // this corresponds to his Theta-tree except that we use a O(n) implementation
55 // for most of the function here, not a O(log(n)) one.
56 class TaskSet {
57  public:
58  explicit TaskSet(int num_tasks) { sorted_tasks_.reserve(num_tasks); }
59 
60  struct Entry {
61  int task;
62  IntegerValue start_min;
63  IntegerValue duration_min;
64 
65  // Note that the tie-breaking is not important here.
66  bool operator<(Entry other) const { return start_min < other.start_min; }
67  };
68 
69  // Insertion and modification. These leave sorted_tasks_ sorted.
70  void Clear() {
71  sorted_tasks_.clear();
72  optimized_restart_ = 0;
73  }
74  void AddEntry(const Entry& e);
75  void RemoveEntryWithIndex(int index);
76 
77  // Advanced usage, if the entry is present, this assumes that its start_min is
78  // >= the end min without it, and update the datastructure accordingly.
79  void NotifyEntryIsNowLastIfPresent(const Entry& e);
80 
81  // Advanced usage. Instead of calling many AddEntry(), it is more efficient to
82  // call AddUnsortedEntry() instead, but then Sort() MUST be called just after
83  // the insertions. Nothing is checked here, so it is up to the client to do
84  // that properly.
85  void AddUnsortedEntry(const Entry& e) { sorted_tasks_.push_back(e); }
86  void Sort() { std::sort(sorted_tasks_.begin(), sorted_tasks_.end()); }
87 
88  // Returns the end-min for the task in the set. The time profile of the tasks
89  // packed to the left will always be a set of contiguous tasks separated by
90  // empty space:
91  //
92  // [Bunch of tasks] ... [Bunch of tasks] ... [critical tasks].
93  //
94  // We call "critical tasks" the last group. These tasks will be solely
95  // responsible for for the end-min of the whole set. The returned
96  // critical_index will be the index of the first critical task in
97  // SortedTasks().
98  //
99  // A reason for the min end is:
100  // - The duration-min of all the critical tasks.
101  // - The fact that all critical tasks have a start-min greater or equal to the
102  // first of them, that is SortedTasks()[critical_index].start_min.
103  //
104  // It is possible to behave like if one task was not in the set by setting
105  // task_to_ignore to the id of this task. This returns 0 if the set is empty
106  // in which case critical_index will be left unchanged.
107  IntegerValue ComputeEndMin(int task_to_ignore, int* critical_index) const;
108  const std::vector<Entry>& SortedTasks() const { return sorted_tasks_; }
109 
110  private:
111  std::vector<Entry> sorted_tasks_;
112  mutable int optimized_restart_ = 0;
113  DISALLOW_COPY_AND_ASSIGN(TaskSet);
114 };
