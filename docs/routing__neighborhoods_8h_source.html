23 // Relocate neighborhood which moves chains of neighbors.
24 // The operator starts by relocating a node n after a node m, then continues
25 // moving nodes which were after n as long as the "cost" added is less than
26 // the "cost" of the arc (m, n). If the new chain doesn't respect the domain of
27 // next variables, it will try reordering the nodes.
28 // Possible neighbors for path 1 -> A -> B -> C -> D -> E -> 2 (where (1, 2) are
29 // first and last nodes of the path and can therefore not be moved, A must
30 // be performed before B, and A, D and E are located at the same place):
31 // 1 -> A -> C -> [B] -> D -> E -> 2
32 // 1 -> A -> C -> D -> [B] -> E -> 2
33 // 1 -> A -> C -> D -> E -> [B] -> 2
34 // 1 -> A -> B -> D -> [C] -> E -> 2
35 // 1 -> A -> B -> D -> E -> [C] -> 2
36 // 1 -> A -> [D] -> [E] -> B -> C -> 2
37 // 1 -> A -> B -> [D] -> [E] -> C -> 2
38 // 1 -> A -> [E] -> B -> C -> D -> 2
39 // 1 -> A -> B -> [E] -> C -> D -> 2
40 // 1 -> A -> B -> C -> [E] -> D -> 2
41 // This operator is extremely useful to move chains of nodes which are located
42 // at the same place (for instance nodes part of a same stop).
43 // TODO(user): Consider merging with standard Relocate in local_search.cc.
44 class MakeRelocateNeighborsOperator : public PathWithPreviousNodesOperator {
45  public:
46  MakeRelocateNeighborsOperator(
47  const std::vector<IntVar*>& vars,
48  const std::vector<IntVar*>& secondary_vars,
49  std::function<int(int64)> start_empty_path_class,
50  RoutingTransitCallback2 arc_evaluator);
51  ~MakeRelocateNeighborsOperator() override {}
52 
53  bool MakeNeighbor() override;
54  std::string DebugString() const override { return "RelocateNeighbors"; }
55 
56  private:
57  // Moves a chain starting after 'before_chain' and ending at 'chain_end'
58  // after node 'destination'. Tries to repair the resulting solution by
59  // checking if the new arc created after 'destination' is compatible with
60  // NextVar domains, and moves the 'destination' down the path if the solution
61  // is inconsistent. Iterates the process on the new arcs created before
62  // the node 'destination' (if destination was moved).
63  bool MoveChainAndRepair(int64 before_chain, int64 chain_end,
64  int64 destination);
65 
66  // Moves node after 'before_to_move' down the path until a position is found
67  // where NextVar domains are not violated, if it exists. Stops when reaching
68  // position after 'up_to'.
69  // If the node was not moved (either because the current position does not
70  // violate any domains or because not such position could be found), returns
71  // -1. If the node was moved to a new position before up_to, returns up_to;
72  // if it was moved just after up_to returns the node which was after up_to.
73  int64 Reposition(int64 before_to_move, int64 up_to);
74 
75  RoutingTransitCallback2 arc_evaluator_;
76 };

459 template <bool swap_first>
460 void PairNodeSwapActiveOperator<swap_first>::OnNodeInitialization() {
461  for (int i = 0; i < pairs_.size(); ++i) {
462  if (IsInactive(pairs_[i].first[0]) && IsInactive(pairs_[i].second[0])) {
463  inactive_pair_ = i;

470 template <bool swap_first>
471 bool PairNodeSwapActiveOperator<swap_first>::MakeNextNeighbor(
472  Assignment* delta, Assignment* deltadelta) {
473  while (inactive_pair_ < pairs_.size()) {
474  if (!IsInactive(pairs_[inactive_pair_].first[0]) ||
475  !IsInactive(pairs_[inactive_pair_].second[0]) ||
476  !PathOperator::MakeNextNeighbor(delta, deltadelta)) {
477  ResetPosition();
478  ++inactive_pair_;
479  } else {
480  return true;
481  }
482  }
483  return false;
484 }

486 template <bool swap_first>
487 bool PairNodeSwapActiveOperator<swap_first>::MakeNeighbor() {
488  const int64 base = BaseNode(0);
489  if (IsPathEnd(base)) {
490  return false;
491  }
492  const int64 pair_first = pairs_[inactive_pair_].first[0];
493  const int64 pair_second = pairs_[inactive_pair_].second[0];
494  if (swap_first) {
495  return MakeActive(pair_second, BaseNode(1)) &&
496  MakeActive(pair_first, base) &&
497  MakeChainInactive(pair_first, Next(pair_first));
498  } else {
499  return MakeActive(pair_second, BaseNode(1)) &&
500  MakeActive(pair_first, base) &&
501  MakeChainInactive(pair_second, Next(pair_second));
502  }
503 }
