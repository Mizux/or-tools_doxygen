28 // This class holds a list of globally valid linear constraints and has some
29 // logic to decide which one should be part of the LP relaxation. We want more
30 // for a better relaxation, but for efficiency we do not want to have too much
31 // constraints while solving the LP.
32 //
33 // This class is meant to contain all the initial constraints of the LP
34 // relaxation and to get new cuts as they are generated. Thus, it can both
35 // manage cuts but also only add the initial constraints lazily if there is too
36 // many of them.
37 //
38 // TODO(user): Also store the LP objective there as it can be useful to decide
39 // which constraint should go into the current LP.
40 class LinearConstraintManager {
41  public:
42  explicit LinearConstraintManager(Model* model)
43  : sat_parameters_(*model->GetOrCreate<SatParameters>()),
44  integer_trail_(*model->GetOrCreate<IntegerTrail>()) {}
45  ~LinearConstraintManager();

47  // Add a new constraint to the manager. Note that we canonicalize constraints
48  // and merge the bounds of constraints with the same terms. We also perform
49  // basic preprocessing.
50  DEFINE_INT_TYPE(ConstraintIndex, int32);
51  ConstraintIndex Add(LinearConstraint ct);
52 
53  // Same as Add(), but logs some information about the newly added constraint.
54  // Cuts are also handled slightly differently than normal constraints.
55  void AddCut(LinearConstraint ct, std::string type_name,
56  const gtl::ITIVector<IntegerVariable, double>& lp_solution);
57 
58  // Heuristic to decides what LP is best solved next. The given lp_solution
59  // should usually be the optimal solution of the LP returned by GetLp() before
60  // this call, but is just used as an heuristic.
61  //
62  // Returns true iff LpConstraints() will return a different LP than before.
63  bool ChangeLp(const gtl::ITIVector<IntegerVariable, double>& lp_solution);
64 
65  // This can be called initially to add all the current constraint to the LP
66  // returned by GetLp().
67  void AddAllConstraintsToLp();
