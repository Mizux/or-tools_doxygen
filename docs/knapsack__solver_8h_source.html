186 // The following code defines needed classes for the KnapsackGenericSolver
187 // class which is the entry point to extend knapsack with new constraints such
188 // as conflicts between items.
189 //
190 // Constraints are enforced using KnapsackPropagator objects, in the current
191 // code there is one propagator per dimension (KnapsackCapacityPropagator).
192 // One of those propagators, named master propagator, is used to guide the
193 // search, i.e. decides which item should be assigned next.
194 // Roughly speaking the search algorithm is:
195 // - While not optimal
196 // - Select next search node to expand
197 // - Select next item_i to assign (using master propagator)
198 // - Generate a new search node where item_i is in the knapsack
199 // - Check validity of this new partial solution (using propagators)
200 // - If valid, add this new search node to the search
201 // - Generate a new search node where item_i is not in the knapsack
202 // - Check validity of this new partial solution (using propagators)
203 // - If valid, add this new search node to the search
204 //
205 // TODO(user): Add a new propagator class for conflict constraint.
206 // TODO(user): Add a new propagator class used as a guide when the problem has
207 // several dimensions.

358 // ----- KnapsackPropagator -----
359 // KnapsackPropagator is the base class for modeling and propagating a
360 // constraint given an assignment.
361 //
362 // When some work has to be done both by the base and the derived class,
363 // a protected pure virtual method ending by 'Propagator' is defined.
364 // For instance, 'Init' creates a vector of items, and then calls
365 // 'InitPropagator' to let the derived class perform its own initialization.
366 class KnapsackPropagator {
367  public:
368  explicit KnapsackPropagator(const KnapsackState& state);
369  virtual ~KnapsackPropagator();
370 
371  // Initializes data structure and then calls InitPropagator.
372  void Init(const std::vector<int64>& profits,
373  const std::vector<int64>& weights);
374 
375  // Updates data structure and then calls UpdatePropagator.
376  // Returns false when failure.
377  bool Update(bool revert, const KnapsackAssignment& assignment);
378  // ComputeProfitBounds should set 'profit_lower_bound_' and
379  // 'profit_upper_bound_' which are constraint specific.
380  virtual void ComputeProfitBounds() = 0;
381  // Returns the id of next item to assign.
382  // Returns kNoSelection when all items are bound.
383  virtual int GetNextItemId() const = 0;
384 
385  int64 current_profit() const { return current_profit_; }
386  int64 profit_lower_bound() const { return profit_lower_bound_; }
387  int64 profit_upper_bound() const { return profit_upper_bound_; }
388 
389  // Copies the current state into 'solution'.
390  // All unbound items are set to false (i.e. not in the knapsack).
391  // When 'has_one_propagator' is true, CopyCurrentSolutionPropagator is called
392  // to have a better solution. When there is only one propagator
393  // there is no need to check the solution with other propagators, so the
394  // partial solution can be smartly completed.
395  void CopyCurrentStateToSolution(bool has_one_propagator,
396  std::vector<bool>* solution) const;
397 
398  protected:
399  // Initializes data structure. This method is called after initialization
400  // of KnapsackPropagator data structure.
401  virtual void InitPropagator() = 0;
402 
403  // Updates internal data structure incrementally. This method is called
404  // after update of KnapsackPropagator data structure.
405  virtual bool UpdatePropagator(bool revert,
406  const KnapsackAssignment& assignment) = 0;
407 
408  // Copies the current state into 'solution'.
409  // Only unbound items have to be copied as CopyCurrentSolution was already
410  // called with current state.
411  // This method is useful when a propagator is able to find a better solution
412  // than the blind instantiation to false of unbound items.
413  virtual void CopyCurrentStateToSolutionPropagator(
414  std::vector<bool>* solution) const = 0;
415 
416  const KnapsackState& state() const { return state_; }
417  const std::vector<KnapsackItemPtr>& items() const { return items_; }
418 
419  void set_profit_lower_bound(int64 profit) { profit_lower_bound_ = profit; }
420  void set_profit_upper_bound(int64 profit) { profit_upper_bound_ = profit; }
421 
422  private:
423  std::vector<KnapsackItemPtr> items_;
424  int64 current_profit_;
425  int64 profit_lower_bound_;
426  int64 profit_upper_bound_;
427  const KnapsackState& state_;
428 
429  DISALLOW_COPY_AND_ASSIGN(KnapsackPropagator);
430 };

453  public:
454  KnapsackCapacityPropagator(const KnapsackState& state, int64 capacity);
455  ~KnapsackCapacityPropagator() override;
456  void ComputeProfitBounds() override;
457  int GetNextItemId() const override { return break_item_id_; }

459  protected:
460  // Initializes KnapsackCapacityPropagator (e.g., sort items in decreasing
461  // order).
462  void InitPropagator() override;
463  // Updates internal data structure incrementally (i.e., 'consumed_capacity_')
464  // to avoid a O(number_of_items) scan.

467  void CopyCurrentStateToSolutionPropagator(
468  std::vector<bool>* solution) const override;
469 
470  private:
471  // An obvious additional profit upper bound corresponds to the linear
472  // relaxation: remaining_capacity * efficiency of the break item.
473  // It is possible to do better in O(1), using Martello-Toth bound U2.
474  // The main idea is to enforce integrality constraint on the break item,
475  // ie. either the break item is part of the solution, either it is not.
476  // So basically the linear relaxation is done on the item before the break
477  // item, or the one after the break item.
478  // This is what GetAdditionalProfit method implements.
479  int64 GetAdditionalProfit(int64 remaining_capacity, int break_item_id) const;
480 
481  const int64 capacity_;
482  int64 consumed_capacity_;
483  int break_item_id_;
484  std::vector<KnapsackItemPtr> sorted_items_;
485  int64 profit_max_;
486 
487  DISALLOW_COPY_AND_ASSIGN(KnapsackCapacityPropagator);
