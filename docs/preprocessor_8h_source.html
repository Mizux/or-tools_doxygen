35 // --------------------------------------------------------
36 // Preprocessor
37 // --------------------------------------------------------
38 // This is the base class for preprocessors.
39 //
40 // TODO(user): On most preprocessors, calling Run() more than once will not work
41 // as expected. Fix? or document and crash in debug if this happens.
42 class Preprocessor {
43  public:
44  explicit Preprocessor(const GlopParameters* parameters);
45  Preprocessor(const Preprocessor&) = delete;
46  Preprocessor& operator=(const Preprocessor&) = delete;
47  virtual ~Preprocessor();
48 
49  // Runs the preprocessor by modifying the given linear program. Returns true
50  // if a postsolve step will be needed (i.e. RecoverSolution() is not the
51  // identity function). Also updates status_ to something different from
52  // ProblemStatus::INIT if the problem was solved (including bad statuses
53  // like ProblemStatus::ABNORMAL, ProblemStatus::INFEASIBLE, etc.).
54  virtual bool Run(LinearProgram* lp) = 0;
55 
56  // Stores the optimal solution of the linear program that was passed to
57  // Run(). The given solution needs to be set to the optimal solution of the
58  // linear program "modified" by Run().
59  virtual void RecoverSolution(ProblemSolution* solution) const = 0;
60 
61  // Returns the status of the preprocessor.
62  // A status different from ProblemStatus::INIT means that the problem is
63  // solved and there is not need to call subsequent preprocessors.
64  ProblemStatus status() const { return status_; }
65 
66  // Some preprocessors only need minimal changes when used with integer
67  // variables in a MIP context. Setting this to true allows to consider integer
68  // variables as integer in these preprocessors.
69  //
70  // Not all preprocessors handle integer variables correctly, calling this
71  // function on them will cause a LOG(FATAL).
72  virtual void UseInMipContext() { in_mip_context_ = true; }
73 
74  void SetTimeLimit(TimeLimit* time_limit) { time_limit_ = time_limit; }
75 
76  protected:
77  // Returns true if a is less than b (or slighlty greater than b with a given
78  // tolerance).
79  bool IsSmallerWithinFeasibilityTolerance(Fractional a, Fractional b) const {
80  return ::operations_research::IsSmallerWithinTolerance(
81  a, b, parameters_.solution_feasibility_tolerance());
82  }
83  bool IsSmallerWithinPreprocessorZeroTolerance(Fractional a,
84  Fractional b) const {
85  // TODO(user): use an absolute tolerance here to be even more defensive?
86  return ::operations_research::IsSmallerWithinTolerance(
87  a, b, parameters_.preprocessor_zero_tolerance());
88  }
89 
90  ProblemStatus status_;
91  const GlopParameters& parameters_;
92  bool in_mip_context_;
93  TimeLimit* time_limit_;
94 };

494 // --------------------------------------------------------
495 // ForcingAndImpliedFreeConstraintPreprocessor
496 // --------------------------------------------------------
497 // This preprocessor computes for each constraint row the bounds that are
498 // implied by the variable bounds and applies one of the following reductions:
499 //
500 // * If the intersection of the implied bounds and the current constraint bounds
501 // is empty (modulo some tolerance), the problem is INFEASIBLE.
502 //
503 // * If the intersection of the implied bounds and the current constraint bounds
504 // is a singleton (modulo some tolerance), then the constraint is said to be
505 // forcing and all the variables that appear in it can be fixed to one of their
506 // bounds. All these columns and the constraint row is removed.
507 //
508 // * If the implied bounds are included inside the current constraint bounds
509 // (modulo some tolerance) then the constraint is said to be redundant or
510 // implied free. Its bounds are relaxed and the constraint will be removed
511 // later by the FreeConstraintPreprocessor.
512 //
513 // * Otherwise, wo do nothing.
514 class ForcingAndImpliedFreeConstraintPreprocessor : public Preprocessor {
515  public:
516  explicit ForcingAndImpliedFreeConstraintPreprocessor(
517  const GlopParameters* parameters)
518  : Preprocessor(parameters) {}
519  ForcingAndImpliedFreeConstraintPreprocessor(
520  const ForcingAndImpliedFreeConstraintPreprocessor&) = delete;
521  ForcingAndImpliedFreeConstraintPreprocessor& operator=(
522  const ForcingAndImpliedFreeConstraintPreprocessor&) = delete;
523  ~ForcingAndImpliedFreeConstraintPreprocessor() final {}
524  bool Run(LinearProgram* lp) final;
525  void RecoverSolution(ProblemSolution* solution) const final;
526 
527  private:
528  bool lp_is_maximization_problem_;
529  SparseMatrix deleted_columns_;
530  DenseRow costs_;
531  DenseBooleanColumn is_forcing_up_;
532  ColumnDeletionHelper column_deletion_helper_;
533  RowDeletionHelper row_deletion_helper_;

536 // --------------------------------------------------------
537 // ImpliedFreePreprocessor
538 // --------------------------------------------------------
539 // It is possible to compute "implied" bounds on a variable from the bounds of
540 // all the other variables and the constraints in which this variable take
541 // place. If such "implied" bounds are inside the variable bounds, then the
542 // variable bounds can be relaxed and the variable is said to be "implied free".
543 //
544 // This preprocessor detects the implied free variables and make as many as
545 // possible free with a priority towards low-degree columns. This transformation
546 // will make the simplex algorithm more efficient later, but will also make it
547 // possible to reduce the problem by applying subsequent transformations:
548 //
549 // * The SingletonPreprocessor already deals with implied free singleton
550 // variables and removes the columns and the rows in which they appear.
551 //
552 // * Any multiple of the column of a free variable can be added to any other
553 // column without changing the linear program solution. This is the dual
554 // counterpart of the fact that any multiple of an equality row can be added to
555 // any row.
556 //
557 // TODO(user): Only process doubleton columns so we have more chance in the
558 // later passes to create more doubleton columns? Such columns lead to a smaller
559 // problem thanks to the DoubletonFreeColumnPreprocessor.
560 class ImpliedFreePreprocessor : public Preprocessor {
561  public:
562  explicit ImpliedFreePreprocessor(const GlopParameters* parameters)
563  : Preprocessor(parameters) {}
564  ImpliedFreePreprocessor(const ImpliedFreePreprocessor&) = delete;
565  ImpliedFreePreprocessor& operator=(const ImpliedFreePreprocessor&) = delete;
566  ~ImpliedFreePreprocessor() final {}
567  bool Run(LinearProgram* lp) final;
568  void RecoverSolution(ProblemSolution* solution) const final;
569 
570  private:
571  // This preprocessor adds fixed offsets to some variables. We remember those
572  // here to un-offset them in RecoverSolution().
573  DenseRow variable_offsets_;
574 
575  // This preprocessor causes some variables who would normally be
576  // AT_{LOWER,UPPER}_BOUND to be VariableStatus::FREE. We store the restore

578  // variable actually turns out to be VariableStatus::FREE.

582 // --------------------------------------------------------
583 // DoubletonFreeColumnPreprocessor
584 // --------------------------------------------------------
585 // This preprocessor removes one of the two rows in which a doubleton column of
586 // a free variable appears. Since we can add any multiple of such a column to
587 // any other column, the way this works is that we can always remove all the
588 // entries on one row.
589 //
590 // Actually, we can remove all the entries except the one of the free column.
591 // But we will be left with a singleton row that we can delete in the same way
592 // as what is done in SingletonPreprocessor. That is by reporting the constraint
593 // bounds into the one of the originally free variable. After this operation,
594 // the doubleton free column will become a singleton and may or may not be
595 // removed later by the SingletonPreprocessor.
596 //
597 // Note that this preprocessor can be seen as the dual of the
598 // DoubletonEqualityRowPreprocessor since when taking the dual, an equality row
599 // becomes a free variable and vice versa.
600 //
601 // Note(user): As far as I know, this doubleton free column procedure is more
602 // general than what can be found in the research papers or in any of the linear
603 // solver open source codes as of July 2013. All of them only process such
604 // columns if one of the two rows is also an equality which is not actually
605 // required. Most probably, commercial solvers do use it though.
606 class DoubletonFreeColumnPreprocessor : public Preprocessor {
607  public:
608  explicit DoubletonFreeColumnPreprocessor(const GlopParameters* parameters)
609  : Preprocessor(parameters) {}
610  DoubletonFreeColumnPreprocessor(const DoubletonFreeColumnPreprocessor&) =
611  delete;
612  DoubletonFreeColumnPreprocessor& operator=(
613  const DoubletonFreeColumnPreprocessor&) = delete;
614  ~DoubletonFreeColumnPreprocessor() final {}
615  bool Run(LinearProgram* lp) final;
616  void RecoverSolution(ProblemSolution* solution) const final;
617 
618  private:
619  enum RowChoice {
620  DELETED = 0,
