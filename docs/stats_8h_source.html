14 // Helper classes to track statistics of a program component.
15 //
16 // Usage example:
17 // // Suppose you have a class that contains a factorization of a matrix B and
18 // // a Solve() function to solve the linear system B.x = a.
19 //
20 // // You will hold your stats in a Stats stats_ class member:
21 // struct Stats : public StatsGroup {
22 // Stats() : StatsGroup("BasisFactorization"),
23 // solve_time("solve_time", this),
24 // input_vector_density("input_vector_density", this),
25 // estimated_accuracy("estimated_accuracy", this) {}
26 //
27 // TimeDistribution solve_time;
28 // RatioDistribution input_vector_density;
29 //
30 // // Values of a few components of B.x - a, updated on each solve.
31 // DoubleDistribution estimated_accuracy;
32 // }
33 //
34 // // You then add a few lines to your Solve() function:
35 // void Solve() {
36 // stats_.solve_time.StartTimer();
37 // stats_.input_vector_density.Add(ComputeDensity());
38 // ... // Do the work.
39 // stats_.estimated_accuracy.Add(EstimateAccuracy());
40 // stats_.solve_time.StopTimerAndAddElapsedTime();
41 // }
42 //
43 // // Now, calling stats_.StatString() will give you a summary of your stats:
44 // BasisFactorization {
45 // solve_time : num [min, max] average std_deviation total
46 // input_vector_density : num [min, max] average std_deviation
47 // estimated_accuracy : num [min, max] average std_deviation
48 // }
49 //
50 // For measuring time, another alternative is to use the SCOPED_TIME_STAT macro.
51 // In our example above, you don't need to define the solve_time distribution
52 // and you can just do:
53 //
54 // void Solve() {
55 // SCOPED_TIME_STAT(&stats_);
56 // ...
57 // }
58 //
59 // This automatically adds a TimeDistribution with name "Solve" to stats_ and
60 // times your function calls!
61 //
62 // IMPORTANT: The SCOPED_TIME_STAT() macro only does something if OR_STATS is
63 // defined, so you need to build your code with blaze build --copt='-DOR_STATS'.
64 // The idea is that by default the instrumentation is off. You can also use the
65 // macro IF_STATS_ENABLED() that does nothing if OR_STATS is not defined or just
66 // translates to its argument otherwise.

90 // Base class for a statistic that can be pretty-printed.
91 class Stat {
92  public:
93  explicit Stat(const std::string& name) : name_(name) {}
94 
95  // Also add this stat to the given group.
96  Stat(const std::string& name, StatsGroup* group);
97  virtual ~Stat() {}
98 
99  // Only used for display purposes.
100  std::string Name() const { return name_; }
101 
102  // Returns a human-readable formatted line of the form "name:
103  // ValueAsString()".
104  std::string StatString() const;
105 
106  // At display, stats are displayed by decreasing priority, then decreasing
107  // Sum(), then alphabetical order.
108  // Used to group the stats per category (timing, ratio, etc..,).
109  virtual int Priority() const { return 0; }
110 
111  // By default return 0 for the sum. This makes it possible to sort stats by
112  // decreasing total time.
113  virtual double Sum() const { return 0; }
114 
115  // Prints information about this statistic.
116  virtual std::string ValueAsString() const = 0;
117 
118  // Is this stat worth printing? Usually false if nothing was measured.
119  virtual bool WorthPrinting() const = 0;
120 
121  // Reset this statistic to the same state as if it was newly created.
122  virtual void Reset() = 0;
123 
124  private:
125  const std::string name_;
126 };

128 // Base class to print a nice summary of a group of statistics.
129 class StatsGroup {
130  public:
131  enum PrintOrder {
132  SORT_BY_PRIORITY_THEN_VALUE = 0,
133  SORT_BY_NAME = 1,
134  };
135 
136  explicit StatsGroup(const std::string& name)
137  : name_(name), stats_(), time_distributions_() {}
138  ~StatsGroup();
139 
140  // Registers a Stat, which will appear in the std::string returned by
141  // StatString(). The Stat object must live as long as this StatsGroup.
142  void Register(Stat* stat);
143 
144  // Returns this group name, followed by one line per Stat registered with this
145  // group (this includes the ones created by LookupOrCreateTimeDistribution()).
146  // Note that only the stats WorthPrinting() are printed.
147  std::string StatString() const;
148 
149  // Changes the print ordering (will affect the order in which the stats
150  // registered with this group are printed via StatString()).
151  void SetPrintOrder(PrintOrder print_order) { print_order_ = print_order; }
152 
153  // Returns and if needed creates and registers a TimeDistribution with the
154  // given name. Note that this involve a map lookup and his thus slower than
155  // directly accessing a TimeDistribution variable.
156  TimeDistribution* LookupOrCreateTimeDistribution(std::string name);
157 
158  // Calls Reset() on all the statistics registered with this group.
159  void Reset();
160 
161  private:
162  std::string name_;
163  PrintOrder print_order_ = SORT_BY_PRIORITY_THEN_VALUE;
164  std::vector<Stat*> stats_;
165  std::map<std::string, TimeDistribution*> time_distributions_;
166 
167  DISALLOW_COPY_AND_ASSIGN(StatsGroup);
168 };

170 // Base class to track and compute statistics about the distribution of a
171 // sequence of double. We provide a few sub-classes below that differ in the way
172 // the values are added to the sequence and in the way the stats are printed.
173 class DistributionStat : public Stat {
174  public:
175  explicit DistributionStat(const std::string& name);
176  DistributionStat(const std::string& name, StatsGroup* group);
177  ~DistributionStat() override {}
178  void Reset() override;
179  bool WorthPrinting() const override { return num_ != 0; }
180 
181  // Implemented by the subclasses.
182  std::string ValueAsString() const override = 0;
183 
184  // Trivial statistics on all the values added so far.
185  double Sum() const override { return sum_; }
186  double Max() const { return max_; }
187  double Min() const { return min_; }
188  int64 Num() const { return num_; }
189 
190  // Get the average of the distribution or 0.0 if empty.
191  double Average() const;
192 
193  // Get the standard deviation of the distribution or 0.0 if empty.
194  // We use the on-line algorithm of Welford described at
195  // http://en.wikipedia.org/wiki/Algorithms_for_calculating_variance
196  // TODO(user): We could also use on top the Kahan summation algorithm to be
197  // even more precise but a bit slower too.
198  double StdDeviation() const;
199 
200  protected:
201  // Adds a value to this sequence and updates the stats.
202  void AddToDistribution(double value);
203  double sum_;
204  double average_;
205  double sum_squares_from_average_;
206  double min_;
207  double max_;
208  int64 num_;
209 };

211 // Statistic on the distribution of a sequence of running times.
212 // Also provides some facility to measure such time with the CPU cycle counter.
213 //
214 // TODO(user): Since we inherit from DistributionStat, we currently store the
215 // sum of CPU cycles as a double internally. A better option is to use int64
216 // because with the 53 bits of precision of a double, we will run into an issue
217 // if the sum of times reaches 52 days for a 2GHz processor.
218 class TimeDistribution : public DistributionStat {
219  public:
220  explicit TimeDistribution(const std::string& name)
221  : DistributionStat(name), timer_() {}
222  TimeDistribution(const std::string& name, StatsGroup* group)
223  : DistributionStat(name, group), timer_() {}
224  std::string ValueAsString() const override;
225 
226  // Time distributions have a high priority to be displayed first.
227  int Priority() const override { return 100; }
228 
229  // Internaly the TimeDistribution stores CPU cycles (to do a bit less work
230  // on each StopTimerAndAddElapsedTime()). Use this function to convert
231  // all the statistics of DistributionStat into seconds.
232  static double CyclesToSeconds(double num_cycles);
233 
234  // Adds a time in seconds to this distribution.
235  void AddTimeInSec(double seconds);
236 
237  // Adds a time in CPU cycles to this distribution.
238  void AddTimeInCycles(double cycles);
239 
240  // Starts the timer in preparation of a StopTimerAndAddElapsedTime().
241  inline void StartTimer() { timer_.Restart(); }
242 
243  // Adds the elapsed time since the last StartTimer() to the distribution and
244  // returns this time in CPU cycles.
245  inline double StopTimerAndAddElapsedTime() {
246  const double cycles = static_cast<double>(timer_.GetCycles());
247  AddToDistribution(cycles);
248  return cycles;
249  }
250 
251  private:
252  // Converts and prints a number of cycles in an human readable way using the
253  // proper time unit depending on the value (ns, us, ms, s, m or h).
254  static std::string PrintCyclesAsTime(double cycles);
255  CycleTimer timer_;
256 };

291 // Helper classes to time a block of code and add the result to a
292 // TimeDistribution. Calls StartTimer() on creation and
293 // StopTimerAndAddElapsedTime() on destruction.
294 //
295 // There are three classes with the same interface:
296 // * EnabledScopedTimeDistributionUpdater always collects the time stats of the
297 // scope in which it is defined. This class is used for stats that are always
298 // collected.
299 // * ScopedTimeDistributionUpdater collects the time stats only when OR_STATS is
300 // defined. This symbol should be used for collecting stats in places where
301 // the overhead of collecting the stats may hurt the performance of the
302 // algorithm.
303 // * DisabledScopedTimeDistributionUpdater is used to implement
304 // ScopedTimeDistributionUpdater when OR_STATS is not defined.
305 class EnabledScopedTimeDistributionUpdater {
306  public:
307  // Note that this does not take ownership of the given stat.
308  explicit EnabledScopedTimeDistributionUpdater(TimeDistribution* stat)
309  : stat_(stat), also_update_(nullptr) {
310  stat->StartTimer();
311  }
312  ~EnabledScopedTimeDistributionUpdater() {
313  const double cycles = stat_->StopTimerAndAddElapsedTime();
314  if (also_update_ != nullptr) {
315  also_update_->AddTimeInCycles(cycles);
316  }
317  }
318 
319  // Updates another TimeDistribution on destruction. This is useful to split
320  // a total time measurement in different categories:
321  //
322  // EnabledScopedTimeDistributionUpdater timer(&total_timer);
323  // ...
324  // switch (type) {
325  // case TypeA : timer.AlsoUpdate(&typeA_timer); break;
326  // case TypeB : timer.AlsoUpdate(&typeB_timer); break;
327  // }
328  void AlsoUpdate(TimeDistribution* also_update) { also_update_ = also_update; }
329 
330  private:
331  TimeDistribution* stat_;
332  TimeDistribution* also_update_;
333  DISALLOW_COPY_AND_ASSIGN(EnabledScopedTimeDistributionUpdater);
334 };
335

338  explicit DisabledScopedTimeDistributionUpdater(TimeDistribution* stat) {}
339  void AlsoUpdate(TimeDistribution* also_update) {}
340 
341  private:
342  DISALLOW_COPY_AND_ASSIGN(DisabledScopedTimeDistributionUpdater);
