25 // Union-Find algorithm to maintain "representative" for relations of the form:
26 // x = coeff * y + offset, where "coeff" and "offset" are integers. The variable
27 // x and y are represented by non-negative integer indices. The idea is to
28 // express variables in affine relation using as little different variables as
29 // possible (the representatives).
30 //
31 // IMPORTANT: If there are relations with std::abs(coeff) != 1, then some
32 // relations might be ignored. See TryAdd() for more details.
33 //
34 // TODO(user): it might be possible to do something fancier and drop less
35 // relations if all the affine relations are given before hand.
36 class AffineRelation {
37  public:
38  AffineRelation() : num_relations_(0) {}
39 
40  // Returns the number of relations added to the class and not ignored.
41  int NumRelations() const { return num_relations_; }
42 
43  // Adds the relation x = coeff * y + offset to the class.
44  // Returns true if it wasn't ignored.
45  //
46  // This relation will only be taken into account if the representative of x
47  // and the one of y are different and if the relation can be transformed into
48  // a similar relation with integer coefficient between the two
49  // representatives.
50  //
51  // That is, given that:
52  // - x = coeff_x * representative_x + offset_x
53  // - y = coeff_y * representative_y + offset_y
54  // we have:
55  // coeff_x * representative_x + offset_x =
56  // coeff * coeff_y * representative_y + coeff * offset_y + offset.
57  // Which can be simplified with the introduction of new variables to:
58  // coeff_x * representative_x = new_coeff * representative_y + new_offset.
59  // And we can merge the two if:
60  // - new_coeff and new_offset are divisible by coeff_x.
61  // - OR coeff_x and new_offset are divisible by new_coeff.
62  //
63  // Checked preconditions: x >=0, y >= 0, coeff != 0 and x != y.
64  //
65  // IMPORTANT: we do not check for integer overflow, but that could be added
66  // if it is needed.
67  bool TryAdd(int x, int y, int64 coeff, int64 offset);
68 
69  // Same as TryAdd() with the option to disallow the use of a given
70  // representative.
71  bool TryAdd(int x, int y, int64 coeff, int64 offset, bool allow_rep_x,
72  bool allow_rep_y);
73 
74  // Returns a valid relation of the form x = coeff * representative + offset.
75  // Note that this can return x = x. Non-const because of path-compression.
76  struct Relation {
77  int representative;
78  int64 coeff;
79  int64 offset;
80  Relation(int r, int64 c, int64 o)
81  : representative(r), coeff(c), offset(o) {}
82  const bool operator==(const Relation& other) const {
83  return representative == other.representative && coeff == other.coeff &&
84  offset == other.offset;
85  }
86  };
87  Relation Get(int x) const;
88 
89  // Returns the size of the class of x.
90  int ClassSize(int x) const {
91  if (x >= representative_.size()) return 1;
92  return size_[Get(x).representative];
93  }
94 
95  private:
96  void IncreaseSizeOfMemberVectors(int new_size) {
97  if (new_size <= representative_.size()) return;
98  for (int i = representative_.size(); i < new_size; ++i) {
99  representative_.push_back(i);
100  }
101  offset_.resize(new_size, 0);
102  coeff_.resize(new_size, 1);
103  size_.resize(new_size, 1);
104  }
105 
106  void CompressPath(int x) const {
107  DCHECK_GE(x, 0);
108  DCHECK_LT(x, representative_.size());
109  tmp_path_.clear();
110  int parent = x;
111  while (parent != representative_[parent]) {
112  tmp_path_.push_back(parent);
113  parent = representative_[parent];
114  }
115  for (const int var : ::gtl::reversed_view(tmp_path_)) {
116  const int old_parent = representative_[var];
117  offset_[var] += coeff_[var] * offset_[old_parent];
118  coeff_[var] *= coeff_[old_parent];
119  representative_[var] = parent;
120  }
121  }
122 
123  int num_relations_;
124 
125  // The equivalence class representative for each variable index.
126  mutable std::vector<int> representative_;
127 
128  // The offset and coefficient such that
129  // variable[index] = coeff * variable[representative_[index]] + offset;
130  mutable std::vector<int64> coeff_;
131  mutable std::vector<int64> offset_;
132 
133  // The size of each representative "tree", used to get a good complexity when
134  // we have the choice of which tree to merge into the other.
135  //
136  // TODO(user): Using a "rank" might be faster, but because we sometimes
137  // need to merge the bad subtree into the better one, it is trickier to
138  // maintain than in the classic union-find algorihtm.
139  std::vector<int> size_;
140 
141  // Used by CompressPath() to maintain the coeff/offset during compression.
142  mutable std::vector<int> tmp_path_;
143 };

149 inline bool AffineRelation::TryAdd(int x, int y, int64 coeff, int64 offset,
150  bool allow_rep_x, bool allow_rep_y) {
151  CHECK_NE(coeff, 0);
152  CHECK_NE(x, y);
153  CHECK_GE(x, 0);
154  CHECK_GE(y, 0);
155  IncreaseSizeOfMemberVectors(std::max(x, y) + 1);
156  CompressPath(x);
157  CompressPath(y);
158  const int rep_x = representative_[x];
159  const int rep_y = representative_[y];
160  if (rep_x == rep_y) return false;
161 
162  // TODO(user): It should be possible to optimize this code block a bit, for
163  // instance depending on the magnitude of new_coeff vs coeff_x, we may already
164  // know that one of the two merge is not possible.
165  const int64 coeff_x = coeff_[x];
166  const int64 new_coeff = coeff * coeff_[y];
167  const int64 new_offset = coeff * offset_[y] + offset - offset_[x];
168  const bool condition1 =
169  allow_rep_y && (new_coeff % coeff_x == 0) && (new_offset % coeff_x == 0);
170  const bool condition2 = allow_rep_x && (coeff_x % new_coeff == 0) &&
171  (new_offset % new_coeff == 0);
172  if (condition1 && (!condition2 || size_[x] <= size_[y])) {
173  representative_[rep_x] = rep_y;
174  size_[rep_y] += size_[rep_x];
175  coeff_[rep_x] = new_coeff / coeff_x;
176  offset_[rep_x] = new_offset / coeff_x;
177  } else if (condition2) {
178  representative_[rep_y] = rep_x;
179  size_[rep_x] += size_[rep_y];
180  coeff_[rep_y] = coeff_x / new_coeff;
181  offset_[rep_y] = -new_offset / new_coeff;
182  } else {
183  return false;
184  }
185  ++num_relations_;
186  return true;
187 }

189 inline AffineRelation::Relation AffineRelation::Get(int x) const {
190  if (x >= representative_.size() || representative_[x] == x) return {x, 1, 0};
191  CompressPath(x);
192  return {representative_[x], coeff_[x], offset_[x]};
193 }
