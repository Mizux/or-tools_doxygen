26 // An LU-Factorization class encapsulating the LU factorization data and
27 // algorithms. The actual algorithm is in markowitz.h and .cc. This class holds
28 // all the Solve() functions that deal with the permutations and the L and U
29 // factors once they are computed.
30 class LuFactorization {
31  public:
32  LuFactorization();
33 
34  // Returns true if the LuFactorization is a factorization of the identity
35  // matrix. In this state, all the Solve() functions will work for any
36  // vector dimension.
37  bool IsIdentityFactorization() { return is_identity_factorization_; }

39  // Clears internal data structure and reset this class to the factorization
40  // of an identity matrix.
41  void Clear();

43  // Computes an LU-decomposition for a given matrix B. If for some reason,
44  // there was an error, then the factorization is reset to the one of the
45  // identity matrix, and an error is reported.
46  //
47  // Note(user): Since a client must use the result, there is little chance of
48  // it being confused by this revert to identity factorization behavior. The
49  // reason behind it is that this way, calling any public function of this
50  // class will never cause a crash of the program.
51  ABSL_MUST_USE_RESULT Status ComputeFactorization(const MatrixView& matrix);
52 
53  // Returns the column permutation used by the LU factorization.
54  const ColumnPermutation& GetColumnPermutation() const { return col_perm_; }
55 
56  // Sets the column permutation to the identity permutation. The idea is that
57  // the column permutation can be incorporated in the basis RowToColMapping,
58  // and once this is done, then a client can call this and effectively remove
59  // the need for a column permutation on each solve.
60  void SetColumnPermutationToIdentity() {
61  col_perm_.clear();
62  inverse_col_perm_.clear();
63  }
64 
65  // Solves 'B.x = b', x initially contains b, and is replaced by 'B^{-1}.b'.
66  // Since P.B.Q^{-1} = L.U, we have B = P^{-1}.L.U.Q.
67  // 1/ Solve P^{-1}.y = b for y by computing y = P.b,
68  // 2/ solve L.z = y for z,
69  // 3/ solve U.t = z for t,
70  // 4/ finally solve Q.x = t, by computing x = Q^{-1}.t.
71  void RightSolve(DenseColumn* x) const;
72 
73  // Solves 'y.B = r', y initially contains r, and is replaced by r.B^{-1}.
74  // Internally, it takes x = y^T, b = r^T and solves B^T.x = b.
75  // We have P.B.Q^{-1} = P.B.Q^T = L.U, thus (L.U)^T = Q.B^T.P^T.
76  // Therefore B^T = Q^{-1}.U^T.L^T.P^T.P^{-1} = Q^{-1}.U^T.L^T.P
77  // The procedure is thus:
78  // 1/ Solve Q^{-1}.y = b for y, by computing y = Q.b,
79  // 2/ solve U^T.z = y for z,
80  // 3/ solve L^T.t = z for t,
81  // 4/ finally, solve P.x = t for x by computing x = P^{-1}.t.
82  void LeftSolve(DenseRow* y) const;
83 
84  // More fine-grained right/left solve functions that may exploit the initial
85  // non-zeros of the input vector if non-empty. Note that a solve involving L
86  // actually solves P^{-1}.L and a solve involving U actually solves U.Q. To
87  // solve a system with the initial matrix B, one needs to call:
88  // - RightSolveL() and then RightSolveU() for a right solve (B.x = initial x).
89  // - LeftSolveU() and then LeftSolveL() for a left solve (y.B = initial y).
90  void RightSolveLWithNonZeros(ScatteredColumn* x) const;
91  void RightSolveUWithNonZeros(ScatteredColumn* x) const;
92  void LeftSolveUWithNonZeros(ScatteredRow* y) const;
93 
94  // Specialized version of LeftSolveL() that may exploit the initial non_zeros
95  // of y if it is non empty. Moreover, if result_before_permutation is not
96  // NULL, it might be filled with the result just before row_perm_ is applied
97  // to it and true is returned. If result_before_permutation is not filled,
98  // then false is returned.
99  bool LeftSolveLWithNonZeros(ScatteredRow* y,
100  ScatteredColumn* result_before_permutation) const;
101 
102  // Specialized version of RightSolveLWithNonZeros() that takes a SparseColumn
103  // or a ScatteredColumn as input. non_zeros will either be cleared or set to
104  // the non zeros of the result. Important: the output x must be of the correct
105  // size and all zero.
106  void RightSolveLForColumnView(const CompactSparseMatrix::ColumnView& b,
107  ScatteredColumn* x) const;
108  void RightSolveLForScatteredColumn(const ScatteredColumn& b,
109  ScatteredColumn* x) const;
110 
111  // Specialized version of RightSolveLWithNonZeros() where x is originaly equal
112  // to 'a' permuted by row_perm_. Note that 'a' is only used for DCHECK.
113  void RightSolveLWithPermutedInput(const DenseColumn& a, DenseColumn* x) const;
114 
115  // Specialized version of LeftSolveU() for an unit right-hand side.
116  // non_zeros will either be cleared or set to the non zeros of the results.
117  // It also returns the value of col permuted by Q (which is the position
118  // of the unit-vector rhs in the solve system: y.U = rhs).
119  // Important: the output y must be of the correct size and all zero.
120  ColIndex LeftSolveUForUnitRow(ColIndex col, ScatteredRow* y) const;
121 
122  // Returns the given column of U.
123  // It will only be valid until the next call to GetColumnOfU().
124  const SparseColumn& GetColumnOfU(ColIndex col) const;
125 
126  // Returns the norm of B^{-1}.a
127  Fractional RightSolveSquaredNorm(const SparseColumn& a) const;
128 
129  // Returns the norm of (B^T)^{-1}.e_row where e is an unit vector.
130  Fractional DualEdgeSquaredNorm(RowIndex row) const;
131 
132  // The fill-in of the LU-factorization is defined as the sum of the number
133  // of entries of both the lower- and upper-triangular matrices L and U minus
134  // the number of entries in the initial matrix B.
135  //
136  // This returns the number of entries in lower + upper as the percentage of
137  // the number of entries in B.
138  double GetFillInPercentage(const MatrixView& matrix) const;
139 
140  // Returns the number of entries in L + U.
141  // If the factorization is the identity, this returns 0.
142  EntryIndex NumberOfEntries() const;
143 
144  // Computes the determinant of the input matrix B.
145  // Since P.B.Q^{-1} = L.U, det(P) * det(B) * det(Q^{-1}) = det(L) * det(U).
146  // det(L) = 1 since L is a lower-triangular matrix with 1 on the diagonal.
147  // det(P) = +1 or -1 (by definition it is the sign of the permutation P)
148  // det(Q^{-1}) = +1 or -1 (the sign of the permutation Q^{-1})
149  // Finally det(U) = product of the diagonal elements of U, since U is an
150  // upper-triangular matrix.
151  // Taking all this into account:
152  // det(B) = sign(P) * sign(Q^{-1}) * prod_i u_ii .
153  Fractional ComputeDeterminant() const;
154 
155  // Computes the 1-norm of the inverse of the input matrix B.
156  // For this we iteratively solve B.x = e_j, where e_j is the jth unit vector.
157  // The result of this computation is the jth column of B^-1.
158  // The 1-norm |B| is defined as max_j sum_i |a_ij|
159  // http://en.wikipedia.org/wiki/Matrix_norm
160  Fractional ComputeInverseOneNorm() const;
161 
162  // Computes the infinity-norm of the inverse of the input matrix B.
163  // The infinity-norm |B| is defined as max_i sum_j |a_ij|
164  // http://en.wikipedia.org/wiki/Matrix_norm
165  Fractional ComputeInverseInfinityNorm() const;
166 
167  // Computes the condition number of the input matrix B.
168  // For a given norm, this is the matrix norm times the norm of its inverse.
169  //
170  // Note that because the LuFactorization class does not keep the
171  // non-factorized matrix in memory, it needs to be passed to these functions.
172  // It is up to the client to pass exactly the same matrix as the one used
173  // for ComputeFactorization().
174  //
175  // TODO(user): separate this from LuFactorization.
176  Fractional ComputeOneNormConditionNumber(const MatrixView& matrix) const;
177  Fractional ComputeInfinityNormConditionNumber(const MatrixView& matrix) const;
178  Fractional ComputeInverseInfinityNormUpperBound() const;

192  // TODO(user): avoid the matrix conversion by multiplying TriangularMatrix
193  // directly.
194  void ComputeLowerTimesUpper(SparseMatrix* product) const {
195  SparseMatrix temp_lower, temp_upper;
196  lower_.CopyToSparseMatrix(&temp_lower);
197  upper_.CopyToSparseMatrix(&temp_upper);
