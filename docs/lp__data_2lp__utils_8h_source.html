43 // Returns the scalar product between u and v.
44 // The precise versions use KahanSum and are about two times slower.
45 template <class DenseRowOrColumn1, class DenseRowOrColumn2>
46 Fractional ScalarProduct(const DenseRowOrColumn1& u,
47  const DenseRowOrColumn2& v) {
48  DCHECK_EQ(u.size().value(), v.size().value());
49  Fractional sum(0.0);
50  typename DenseRowOrColumn1::IndexType i(0);
51  typename DenseRowOrColumn2::IndexType j(0);
52  const size_t num_blocks = u.size().value() / 4;
53  for (size_t block = 0; block < num_blocks; ++block) {
54  // Computing the sum of 4 elements at once may allow the compiler to
55  // generate more efficient code, e.g. using SIMD and checking the loop
56  // condition much less frequently.
57  //
58  // This produces different results from the case where each multiplication
59  // is added to sum separately. An extreme example of this can be derived
60  // using the fact that 1e11 + 2e-6 == 1e11, but 1e11 + 8e-6 > 1e11.
61  //
62  // While the results are different, they aren't necessarily better or worse.
63  // Typically, sum will be of larger magnitude than any individual
64  // multiplication, so one might expect, in practice, this method to yield
65  // more accurate results. However, if accuracy is vital, use the precise
66  // version.
67  sum += (u[i++] * v[j++]) + (u[i++] * v[j++]) + (u[i++] * v[j++]) +
68  (u[i++] * v[j++]);
69  }
70  while (i < u.size()) {
71  sum += u[i++] * v[j++];
72  }
73  return sum;
74 }

76 // Note: This version is heavily used in the pricing.
77 // TODO(user): Optimize this more (SSE or unroll with two sums). Another
78 // option is to skip the u[col] that are 0.0 rather than fetching the coeff
79 // and doing a Fractional multiplication.

81 Fractional ScalarProduct(const DenseRowOrColumn& u, const SparseColumn& v) {

90 template <class DenseRowOrColumn, class DenseRowOrColumn2>
91 Fractional PreciseScalarProduct(const DenseRowOrColumn& u,
92  const DenseRowOrColumn2& v) {
93  DCHECK_EQ(u.size().value(), v.size().value());
94  KahanSum sum;
95  for (typename DenseRowOrColumn::IndexType i(0); i < u.size(); ++i) {
96  sum.Add(u[i] * v[typename DenseRowOrColumn2::IndexType(i.value())]);
97  }
98  return sum.Value();
99 }

112 template <class DenseRowOrColumn>
113 Fractional PreciseScalarProduct(const DenseRowOrColumn& u,
114  const ScatteredColumn& v) {
115  DCHECK_EQ(u.size().value(), v.values.size().value());
116  if (ShouldUseDenseIteration(v)) {
117  return PreciseScalarProduct(u, v.values);
118  }
119  KahanSum sum;
120  for (const RowIndex row : v.non_zeros) {
121  sum.Add(u[typename DenseRowOrColumn::IndexType(row.value())] * v[row]);
122  }
123  return sum.Value();
124 }

126 // Computes a scalar product for entries with index not greater than max_index.
127 template <class DenseRowOrColumn>
128 Fractional PartialScalarProduct(const DenseRowOrColumn& u,
129  const SparseColumn& v, int max_index) {

133  return sum;

231 // Permutes the given dense vector. It uses for this an all zero scratchpad.
232 template <typename IndexType, typename PermutationIndexType>
233 inline void PermuteWithScratchpad(
234  const Permutation<PermutationIndexType>& permutation,
235  StrictITIVector<IndexType, Fractional>* zero_scratchpad,
236  StrictITIVector<IndexType, Fractional>* input_output) {
237  DCHECK(IsAllZero(*zero_scratchpad));
238  const IndexType size = input_output->size();
239  zero_scratchpad->swap(*input_output);
240  input_output->resize(size, 0.0);
241  for (IndexType index(0); index < size; ++index) {
242  const Fractional value = (*zero_scratchpad)[index];
243  if (value != 0.0) {
244  const IndexType permuted_index(
245  permutation[PermutationIndexType(index.value())].value());
246  (*input_output)[permuted_index] = value;
247  }
248  }
249  zero_scratchpad->assign(size, 0.0);
250 }

252 // Same as PermuteAndComputeNonZeros() except that we assume that the given
253 // non-zeros are the initial non-zeros positions of output.
254 template <typename IndexType>
255 inline void PermuteWithKnownNonZeros(
256  const Permutation<IndexType>& permutation,
257  StrictITIVector<IndexType, Fractional>* zero_scratchpad,
258  StrictITIVector<IndexType, Fractional>* output,
259  std::vector<IndexType>* non_zeros) {
260  DCHECK(IsAllZero(*zero_scratchpad));
261  zero_scratchpad->swap(*output);
262  output->resize(zero_scratchpad->size(), 0.0);
263  for (IndexType& index_ref : *non_zeros) {
264  const Fractional value = (*zero_scratchpad)[index_ref];
265  (*zero_scratchpad)[index_ref] = 0.0;
266  const IndexType permuted_index(permutation[index_ref]);
267  (*output)[permuted_index] = value;
268  index_ref = permuted_index;
269  }
270 }

272 // Sets a dense vector for which the non zeros are known to be non_zeros.
273 template <typename IndexType, typename ScatteredRowOrCol>
274 inline void ClearAndResizeVectorWithNonZeros(IndexType size,
275  ScatteredRowOrCol* v) {
276  // Only use the sparse version if there is less than 5% non-zeros positions
277  // compared to the wanted size. Note that in most cases the vector will
278  // already be of the correct size.
279  const double kSparseThreshold = 0.05;
280  if (!v->non_zeros.empty() &&
281  v->non_zeros.size() < kSparseThreshold * size.value()) {
282  for (const IndexType index : v->non_zeros) {
283  DCHECK_LT(index, v->values.size());
284  (*v)[index] = 0.0;
285  }
286  v->values.resize(size, 0.0);
287  DCHECK(IsAllZero(v->values));
288  } else {
289  v->values.AssignToZero(size);
290  }
291  v->non_zeros.clear();
292 }

303 // Given N Fractional elements, this class maintains their sum and can
304 // provide, for each element X, the sum of all elements except X.
305 // The subtelty is that it works well with infinities: for example, if there is
306 // exactly one infinite element X, then SumWithout(X) will be finite.
307 //
308 // Two flavors of this class are provided: SumWithPositiveInfiniteAndOneMissing
309 // supports calling Add() with normal numbers and positive infinities (and will
310 // DCHECK() that), and SumWithNegativeInfiniteAndOneMissing does the same with
311 // negative infinities.
312 //
313 // The numerical accuracy suffers however. If X is 1e100 and SumWithout(X)
314 // should be 1e-100, then the value actually returned by SumWithout(X) is likely
315 // to be wrong (by up to std::numeric_limits<Fractional>::epsilon() ^ 2).
316 template <bool supported_infinity_is_positive>
317 class SumWithOneMissing {
318  public:
319  SumWithOneMissing() : num_infinities_(0), sum_() {}
320 
321  void Add(Fractional x) {
322  if (num_infinities_ > 1) return;
323  if (IsFinite(x)) {
324  sum_.Add(x);
325  return;
326  }
327  DCHECK_EQ(Infinity(), x);
328  ++num_infinities_;
329  }
330 
331  Fractional Sum() const {
332  if (num_infinities_ > 0) return Infinity();
333  return sum_.Value();
334  }
335 
336  Fractional SumWithout(Fractional x) const {
337  if (IsFinite(x)) {
338  if (num_infinities_ > 0) return Infinity();
339  return sum_.Value() - x;
340  }
341  DCHECK_EQ(Infinity(), x);
342  if (num_infinities_ > 1) return Infinity();
343  return sum_.Value();
344  }
345 
346  private:
347  Fractional Infinity() const {
348  return supported_infinity_is_positive ? kInfinity : -kInfinity;
349  }
350 
351  // Count how many times Add() was called with an infinite value. The count is
352  // stopped at 2 to be a bit faster.
353  int num_infinities_;
354  KahanSum sum_; // stripped of all the infinite values.
355 };
356 typedef SumWithOneMissing<true> SumWithPositiveInfiniteAndOneMissing;
357 typedef SumWithOneMissing<false> SumWithNegativeInfiniteAndOneMissing;
