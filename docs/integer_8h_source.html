49 // Value type of an integer variable. An integer variable is always bounded
50 // on both sides, and this type is also used to store the bounds [lb, ub] of the
51 // range of each integer variable.
52 //
53 // Note that both bounds are inclusive, which allows to write many propagation
54 // algorithms for just one of the bound and apply it to the negated variables to
55 // get the symmetric algorithm for the other bound.
56 DEFINE_INT_TYPE(IntegerValue, int64);
57 
58 // The max range of an integer variable is [kMinIntegerValue, kMaxIntegerValue].
59 //
60 // It is symmetric so the set of possible ranges stays the same when we take the
61 // negation of a variable. Moreover, we need some IntegerValue that fall outside
62 // this range on both side so that we can usally take care of integer overflow
63 // by simply doing "saturated arithmetic" and if one of the bound overflow, the
64 // two bounds will "cross" each others and we will get an empty range.
65 constexpr IntegerValue kMaxIntegerValue(
66  std::numeric_limits<IntegerValue::ValueType>::max() - 1);
67 constexpr IntegerValue kMinIntegerValue(-kMaxIntegerValue);
68 
69 inline double ToDouble(IntegerValue value) {
70  const double kInfinity = std::numeric_limits<double>::infinity();
71  if (value >= kMaxIntegerValue) return kInfinity;
72  if (value <= kMinIntegerValue) return -kInfinity;
73  return static_cast<double>(value.value());
74 }

81 inline IntegerValue Subtract(IntegerValue a, IntegerValue b) {
82  const int64 result = CapSub(a.value(), b.value());
83  if (result == kint64min || IntegerValue(result) <= kMinIntegerValue) {
84  return kMinIntegerValue;
85  }
86  if (result == kint64max || IntegerValue(result) >= kMaxIntegerValue) {
87  return kMaxIntegerValue;
88  }
89  return IntegerValue(result);
90 }

92 inline IntegerValue CeilRatio(IntegerValue dividend,
93  IntegerValue positive_divisor) {
94  CHECK_GT(positive_divisor, 0);
95  const IntegerValue result = dividend / positive_divisor;
96  const IntegerValue adjust =
97  static_cast<IntegerValue>(result * positive_divisor < dividend);
98  return result + adjust;

103  CHECK_GT(positive_divisor, 0);

106  static_cast<IntegerValue>(result * positive_divisor > dividend);

110 // Computes result += a * b, and return false iff there is an overflow.
111 inline bool AddProductTo(IntegerValue a, IntegerValue b, IntegerValue* result) {
112  const int64 prod = CapProd(a.value(), b.value());
113  if (prod == kint64min || prod == kint64max) return false;
114  const int64 add = CapAdd(prod, result->value());
115  if (add == kint64min || add == kint64max) return false;
116  *result = IntegerValue(add);
117  return true;
118 }

143 // The integer equivalent of a literal.
144 // It represents an IntegerVariable and an upper/lower bound on it.
145 //
146 // Overflow: all the bounds below kMinIntegerValue and kMaxIntegerValue are
147 // treated as kMinIntegerValue - 1 and kMaxIntegerValue + 1.
148 struct IntegerLiteral {
149  // Because IntegerLiteral should never be created at a bound less constrained
150  // than an existing IntegerVariable bound, we don't allow GreaterOrEqual() to
151  // have a bound lower than kMinIntegerValue, and LowerOrEqual() to have a
152  // bound greater than kMaxIntegerValue. The other side is not constrained
153  // to allow for a computed bound to overflow. Note that both the full initial
154  // domain and the empty domain can always be represented.
155  static IntegerLiteral GreaterOrEqual(IntegerVariable i, IntegerValue bound);
156  static IntegerLiteral LowerOrEqual(IntegerVariable i, IntegerValue bound);
157 
158  // Clients should prefer the static construction methods above.
159  IntegerLiteral() : var(-1), bound(0) {}
160  IntegerLiteral(IntegerVariable v, IntegerValue b) : var(v), bound(b) {
161  DCHECK_GE(bound, kMinIntegerValue);
162  DCHECK_LE(bound, kMaxIntegerValue + 1);
163  }
164 
165  // The negation of x >= bound is x <= bound - 1.
166  IntegerLiteral Negated() const;
167 
168  bool operator==(IntegerLiteral o) const {
169  return var == o.var && bound == o.bound;
170  }
171  bool operator!=(IntegerLiteral o) const {
172  return var != o.var || bound != o.bound;
173  }
174 
175  std::string DebugString() const {
176  return VariableIsPositive(var)
177  ? absl::StrCat("I", var.value() / 2, ">=", bound.value())
178  : absl::StrCat("I", var.value() / 2, "<=", -bound.value());
179  }
180 
181  // Note that bound should be in [kMinIntegerValue, kMaxIntegerValue + 1].
182  IntegerVariable var;
183  IntegerValue bound;
184 };

205 // Each integer variable x will be associated with a set of literals encoding
206 // (x >= v) for some values of v. This class maintains the relationship between
207 // the integer variables and such literals which can be created by a call to
208 // CreateAssociatedLiteral().
209 //
210 // The advantage of creating such Boolean variables is that the SatSolver which
211 // is driving the search can then take this variable as a decision and maintain
212 // these variables activity and so on. These variables can also be propagated
213 // directly by the learned clauses.
214 //
215 // This class also support a non-lazy full domain encoding which will create one
216 // literal per possible value in the domain. See FullyEncodeVariable(). This is
217 // meant to be called by constraints that directly work on the variable values
218 // like a table constraint or an all-diff constraint.
219 //
220 // TODO(user): We could also lazily create precedences Booleans between two
221 // arbitrary IntegerVariable. This is better done in the PrecedencesPropagator
222 // though.

234  // Fully encode a variable using its current initial domain.
235  // This can be called only once.
236  //

238  // 1) num_values for the literals X == value. Except when there is just
239  // two value in which case only one variable is created.

242  // literals for the two extreme points.

244  // The encoding for NegationOf(var) is automatically created too. It reuses
245  // the same Boolean variable as the encoding of var.
