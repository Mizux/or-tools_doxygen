164  // Solves the problem and returns its status.
165  // An empty problem is considered to be SAT.
166  //
167  // Note that the conflict limit applies only to this function and starts
168  // counting from the time it is called.
169  //
170  // This will restart from the current solver configuration. If a previous call
171  // to Solve() was interrupted by a conflict or time limit, calling this again
172  // will resume the search exactly as it would have continued.
173  //
174  // Note that this will use the TimeLimit singleton, so the time limit
175  // will be counted since the last time TimeLimit was reset, not from
176  // the start of this function.
177  enum Status {
178  ASSUMPTIONS_UNSAT,
179  INFEASIBLE,
180  FEASIBLE,

189  // Simple interface to solve a problem under the given assumptions. This
190  // simply ask the solver to solve a problem given a set of variables fixed to
191  // a given value (the assumptions). Compared to simply calling AddUnitClause()
192  // and fixing the variables once and for all, this allow to backtrack over the
193  // assumptions and thus exploit the incrementally between subsequent solves.
194  //
195  // This function backtrack over all the current decision, tries to enqueue the
196  // given assumptions, sets the assumption level accordingly and finally calls
197  // Solve().
198  //
199  // If, given these assumptions, the model is UNSAT, this returns the
200  // ASSUMPTIONS_UNSAT status. INFEASIBLE is reserved for the case where the
201  // model is proven to be unsat without any assumptions.
202  //
203  // If ASSUMPTIONS_UNSAT is returned, it is possible to get a "core" of unsat
204  // assumptions by calling GetLastIncompatibleDecisions().
205  Status ResetAndSolveWithGivenAssumptions(
206  const std::vector<Literal>& assumptions);

210  // heuristics, like the LBD value of a clause.
211  void SetAssumptionLevel(int assumption_level);

213  // Returns the current assumption level. Note that if a solve was done since
214  // the last SetAssumptionLevel(), then the returned level may be lower than
215  // the one that was set. This is because some assumptions may now be

217  int AssumptionLevel() const { return assumption_level_; }

229  // Takes a new decision (the given true_literal must be unassigned) and
230  // propagates it. Returns the trail index of the first newly propagated
231  // literal. If there is a conflict and the problem is detected to be UNSAT,
232  // returns kUnsatTrailIndex.
233  //
234  // A client can determine if there is a conflict by checking if the
235  // CurrentDecisionLevel() was increased by 1 or not.
236  //
237  // If there is a conflict, the given decision is not applied and:
238  // - The conflict is learned.
239  // - The decisions are potentially backtracked to the first decision that
240  // propagates more variables because of the newly learned conflict.
241  // - The returned value is equal to trail_->Index() after this backtracking
242  // and just before the new propagation (due to the conflict) which is also
243  // performed by this function.
244  int EnqueueDecisionAndBackjumpOnConflict(Literal true_literal);
245 
246  // This function starts by calling EnqueueDecisionAndBackjumpOnConflict(). If
247  // there is no conflict, it stops there. Otherwise, it tries to reapply all
248  // the decisions that were backjumped over until the first one that can't be
249  // taken because it is incompatible. Note that during this process, more
250  // conflicts may happen and the trail may be backtracked even further.
251  //
252  // In any case, the new decisions stack will be the largest valid "prefix"
253  // of the old stack. Note that decisions that are now consequence of the ones
254  // before them will no longer be decisions.
255  //
256  // Note(user): This function can be called with an already assigned literal.
257  int EnqueueDecisionAndBacktrackOnConflict(Literal true_literal);
258 
259  // Tries to enqueue the given decision and performs the propagation.
260  // Returns true if no conflict occurred. Otherwise, returns false and restores
261  // the solver to the state just before this was called.
262  //
263  // Note(user): With this function, the solver doesn't learn anything.
264  bool EnqueueDecisionIfNotConflicting(Literal true_literal);
265 
266  // Restores the state to the given target decision level. The decision at that
267  // level and all its propagation will not be undone. But all the trail after
268  // this will be cleared. Calling this with 0 will revert all the decisions and
269  // only the fixed variables will be left on the trail.
270  void Backtrack(int target_level);
271 
272  // Advanced usage. This is meant to restore the solver to a "proper" state
273  // after a solve was interupted due to a limit reached.
274  //
275  // Without assumption (i.e. if AssumptionLevel() is 0), this will revert all
276  // decisions and make sure that all the fixed literals are propagated. In
277  // presence of assumptions, this will either backtrack to the assumption level
278  // or re-enqueue any assumptions that may have been backtracked over due to
279  // conflits resolution. In both cases, the propagation is finished.
280  //
281  // Note that this may prove the model to be UNSAT or ASSUMPTION_UNSAT in which
282  // case it will return false.
283  bool RestoreSolverToAssumptionLevel();
284 
285  // Advanced usage. Finish the progation if it was interupted. Note that this

306  // Extract the current problem clauses. The Output type must support the two
307  // functions:
308  // - void AddBinaryClause(Literal a, Literal b);
309  // - void AddClause(absl::Span<const Literal> clause);
310  //
311  // TODO(user): also copy the removable clauses?
312  template <typename Output>
313  void ExtractClauses(Output* out) {
314  CHECK(!IsModelUnsat());
315  Backtrack(0);
316  out->SetNumVariables(NumVariables());
317 
318  // It is important to process the newly fixed variables, so they are not
319  // present in the clauses we export.
320  if (num_processed_fixed_variables_ < trail_->Index()) {
321  ProcessNewlyFixedVariables();
322  }
323  clauses_propagator_->DeleteDetachedClauses();
324 
325  // Note(user): Putting the binary clauses first help because the presolver
326  // currently process the clauses in order.
327  binary_implication_graph_->ExtractAllBinaryClauses(out);
328  for (SatClause* clause : clauses_propagator_->AllClausesInCreationOrder()) {
329  if (!clauses_propagator_->IsRemovable(clause)) {
330  out->AddClause(clause->AsSpan());
331  }
332  }
333  }

568  // learning in a boolean satisfiability solver" Proceedings of the 2001
569  // IEEE/ACM international conference on Computer-aided design, Pages 279-285.
570  // http://www.cs.tau.ac.il/~msagiv/courses/ATP/iccad2001_final.pdf

989 // This can be used to enumerate all the solutions. After each SAT call to
990 // Solve(), calling this will reset the solver and exclude the current solution
991 // so that the next call to Solve() will give a new solution or UNSAT is there
992 // is no more new solutions.

994  return [=](Model* model) {

996
