64 // The main class to search for a solution in a flatzinc model. It is
65 // responsible for parsing the search annotations, setting up the
66 // search state and performing the actual search.
67 class Solver {
68  public:
69  explicit Solver(const Model& model)
70  : model_(model),
71  statistics_(model),
72  data_(model.name()),
73  objective_var_(nullptr),
74  objective_monitor_(nullptr),
75  default_phase_(nullptr) {
76  solver_ = data_.solver();
77  }
78 
79  // Searches for for solutions in the model passed at construction
80  // time. The exact search context (search for optimal solution, for
81  // n solutions, for the first solution) is specified in the
82  // parameters.
83  // The parallel context (sequential, multi-threaded) is encapsulated
84  // in the search reporting interface.
85  void Solve(FlatzincParameters p, SearchReportingInterface* report);
86 
87  // Extraction support.
88  bool Extract();
89 
90  // String output for the minizinc interface.
91  std::string SolutionString(const SolutionOutputSpecs& output) const;
92 
93  // Query the value of the variable. This must be called during search, when
94  // a solution is found.
95  int64 SolutionValue(IntegerVariable* var) const;
96 
97  // Programmatic interface to read the solutions.
98 
99  // Returns the number of solutions stored. You need to set store_all_solution
100  // to true in the parameters, otherwise this method will always return 0.
101  int NumStoredSolutions() const { return stored_values_.size(); }
102  // Returns the stored value for the given variable in the solution_index'th
103  // stored solution.
104  // A variable is stored only if it appears in the output part of the model.
105  int64 StoredValue(int solution_index, IntegerVariable* var) {
106  CHECK_GE(solution_index, 0);
107  CHECK_LT(solution_index, stored_values_.size());
108  CHECK(gtl::ContainsKey(stored_values_[solution_index], var));
109  return stored_values_[solution_index][var];
110  }
