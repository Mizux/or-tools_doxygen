51 // A literal is used to represent a variable or its negation. If it represents
52 // the variable it is said to be positive. If it represent its negation, it is
53 // said to be negative. We support two representations as an integer.
54 //
55 // The "signed" encoding of a literal is convenient for input/output and is used
56 // in the cnf file format. For a 0-based variable index x, (x + 1) represent the
57 // variable x and -(x + 1) represent its negation. The signed value 0 is an
58 // undefined literal and this class can never contain it.
59 //
60 // The "index" encoding of a literal is convenient as an index to an array
61 // and is the one used internally for efficiency. It is always positive or zero,
62 // and for a 0-based variable index x, (x << 1) encode the variable x and the
63 // same number XOR 1 encode its negation.
64 class Literal {
65  public:
66  // Not explicit for tests so we can write:
67  // std::vector<literal> literal = {+1, -3, +4, -9};
68  Literal(int signed_value) // NOLINT
69  : index_(signed_value > 0 ? ((signed_value - 1) << 1)
70  : ((-signed_value - 1) << 1) ^ 1) {
71  CHECK_NE(signed_value, 0);
72  }
73 
74  Literal() {}
75  explicit Literal(LiteralIndex index) : index_(index.value()) {}
76  Literal(BooleanVariable variable, bool is_positive)
77  : index_(is_positive ? (variable.value() << 1)
78  : (variable.value() << 1) ^ 1) {}
79 
80  BooleanVariable Variable() const { return BooleanVariable(index_ >> 1); }
81  bool IsPositive() const { return !(index_ & 1); }
82  bool IsNegative() const { return (index_ & 1); }
83 
84  LiteralIndex Index() const { return LiteralIndex(index_); }
85  LiteralIndex NegatedIndex() const { return LiteralIndex(index_ ^ 1); }
86 
87  int SignedValue() const {
88  return (index_ & 1) ? -((index_ >> 1) + 1) : ((index_ >> 1) + 1);
89  }
90 
91  Literal Negated() const { return Literal(NegatedIndex()); }
92 
93  std::string DebugString() const {
94  return absl::StrFormat("%+d", SignedValue());
95  }
96  bool operator==(Literal other) const { return index_ == other.index_; }
97  bool operator!=(Literal other) const { return index_ != other.index_; }
98 
99  bool operator<(const Literal& literal) const {
100  return Index() < literal.Index();
101  }
102 
103  private:
104  int index_;
105 };

108  os << literal.DebugString();
109  return os;

113  absl::Span<const Literal> literals) {

120 // Holds the current variable assignment of the solver.
121 // Each variable can be unassigned or be assigned to true or false.
122 class VariablesAssignment {
123  public:
124  VariablesAssignment() {}
125  explicit VariablesAssignment(int num_variables) { Resize(num_variables); }
126  void Resize(int num_variables) {
127  assignment_.Resize(LiteralIndex(num_variables << 1));
128  }
129 
130  // Makes the given literal true by assigning its underlying variable to either
131  // true or false depending on the literal sign. This can only be called on an
132  // unassigned variable.
133  void AssignFromTrueLiteral(Literal literal) {
134  DCHECK(!VariableIsAssigned(literal.Variable()));
135  assignment_.Set(literal.Index());
136  }
137 
138  // Unassign the variable corresponding to the given literal.
139  // This can only be called on an assigned variable.
140  void UnassignLiteral(Literal literal) {
141  DCHECK(VariableIsAssigned(literal.Variable()));
142  assignment_.ClearTwoBits(literal.Index());
143  }
144 
145  // Literal getters. Note that both can be false, in which case the
146  // corresponding variable is not assigned.
147  bool LiteralIsFalse(Literal literal) const {
148  return assignment_.IsSet(literal.NegatedIndex());
149  }
150  bool LiteralIsTrue(Literal literal) const {
151  return assignment_.IsSet(literal.Index());
152  }
153  bool LiteralIsAssigned(Literal literal) const {
154  return assignment_.AreOneOfTwoBitsSet(literal.Index());
155  }
156 
157  // Returns true iff the given variable is assigned.
158  bool VariableIsAssigned(BooleanVariable var) const {
159  return assignment_.AreOneOfTwoBitsSet(LiteralIndex(var.value() << 1));
160  }
161 
162  // Returns the literal of the given variable that is assigned to true.
163  // That is, depending on the variable, it can be the positive literal or the
164  // negative one. Only call this on an assigned variable.
165  Literal GetTrueLiteralForAssignedVariable(BooleanVariable var) const {
166  DCHECK(VariableIsAssigned(var));
167  return Literal(var, assignment_.IsSet(LiteralIndex(var.value() << 1)));
168  }
169 
170  int NumberOfVariables() const { return assignment_.size().value() / 2; }
171 
172  private:
173  // The encoding is as follows:
174  // - assignment_.IsSet(literal.Index()) means literal is true.
175  // - assignment_.IsSet(literal.Index() ^ 1]) means literal is false.
176  // - If both are false, then the variable (and the literal) is unassigned.
177  Bitset64<LiteralIndex> assignment_;
178 
179  DISALLOW_COPY_AND_ASSIGN(VariablesAssignment);
180 };

186 // Information about a variable assignment.
187 struct AssignmentInfo {
188  // The decision level at which this assignment was made. This starts at 0 and
189  // increases each time the solver takes a search decision.
190  //
191  // TODO(user): We may be able to get rid of that for faster enqueues. Most of
192  // the code only need to know if this is 0 or the highest level, and for the

194  // so we could do it fairly efficiently.
195  //
196  // TODO(user): We currently don't support more than 134M decision levels. That

207 
208  // The index of this assignment in the trail.
209  int32 trail_index;

249  // Enqueues the assignment that make the given literal true on the trail. This
250  // should only be called on unassigned variables.
251  void Enqueue(Literal true_literal, int propagator_id) {
252  DCHECK(!assignment_.VariableIsAssigned(true_literal.Variable()));
253  trail_[current_info_.trail_index] = true_literal;
254  current_info_.last_polarity = true_literal.IsPositive();
255  current_info_.type = propagator_id;
256  info_[true_literal.Variable()] = current_info_;
257  assignment_.AssignFromTrueLiteral(true_literal);
258  ++current_info_.trail_index;
259  }
260 
261  // Specific Enqueue() version for the search decision.
262  void EnqueueSearchDecision(Literal true_literal) {
263  Enqueue(true_literal, AssignmentType::kSearchDecision);
264  }
265 
266  // Specific Enqueue() version for a fixed variable.
267  void EnqueueWithUnitReason(Literal true_literal) {
268  Enqueue(true_literal, AssignmentType::kUnitReason);
269  }

271  // Some constraints propagate a lot of literals at once. In these cases, it is
272  // more efficient to have all the propagated literals except the first one
273  // referring to the reason of the first of them.
274  void EnqueueWithSameReasonAs(Literal true_literal,
275  BooleanVariable reference_var) {
276  reference_var_with_same_reason_as_[true_literal.Variable()] = reference_var;
277  Enqueue(true_literal, AssignmentType::kSameReasonAs);
278  }
279 
280  // Enqueues the given literal using the current content of
281  // GetEmptyVectorToStoreReason() as the reason. This API is a bit more
282  // leanient and does not require the literal to be unassigned. If it is
283  // already assigned to false, then MutableConflict() will be set appropriately
284  // and this will return false otherwise this will enqueue the literal and
285  // returns true.
286  ABSL_MUST_USE_RESULT bool EnqueueWithStoredReason(Literal true_literal) {
287  if (assignment_.LiteralIsTrue(true_literal)) return true;
288  if (assignment_.LiteralIsFalse(true_literal)) {
289  *MutableConflict() = reasons_repository_[Index()];
290  MutableConflict()->push_back(true_literal);
291  return false;
292  }
293 
294  Enqueue(true_literal, AssignmentType::kCachedReason);
295  const BooleanVariable var = true_literal.Variable();
296  reasons_[var] = reasons_repository_[info_[var].trail_index];
297  old_type_[var] = info_[var].type;
298  info_[var].type = AssignmentType::kCachedReason;
299  return true;
300  }
301 
302  // Returns the reason why this variable was assigned.
303  //
304  // Note that this shouldn't be called on a variable at level zero, because we
305  // don't cleanup the reason data for these variables but the underlying
306  // clauses may have been deleted.
307  absl::Span<const Literal> Reason(BooleanVariable var) const;
308 
309  // Returns the "type" of an assignment (see AssignmentType). Note that this
310  // function never returns kSameReasonAs or kCachedReason, it instead returns
311  // the initial type that caused this assignment. As such, it is different
312  // from Info(var).type and the latter should not be used outside this class.
313  int AssignmentType(BooleanVariable var) const;
314 
315  // If a variable was propagated with EnqueueWithSameReasonAs(), returns its
316  // reference variable. Otherwise return the given variable.
317  BooleanVariable ReferenceVarWithSameReason(BooleanVariable var) const;
318 
319  // This can be used to get a location at which the reason for the literal
320  // at trail_index on the trail can be stored. This clears the vector before
321  // returning it.
322  std::vector<Literal>* GetEmptyVectorToStoreReason(int trail_index) const {
323  if (trail_index >= reasons_repository_.size()) {
324  reasons_repository_.resize(trail_index + 1);
325  }
326  reasons_repository_[trail_index].clear();
327  return &reasons_repository_[trail_index];
328  }
329 
330  // Shortcut for GetEmptyVectorToStoreReason(Index()).
331  std::vector<Literal>* GetEmptyVectorToStoreReason() const {
332  return GetEmptyVectorToStoreReason(Index());
333  }
334 
335  // Reverts the trail and underlying assignment to the given target trail
336  // index. Note that we do not touch the assignment info.
337  void Untrail(int target_trail_index) {
338  const int index = Index();
339  num_untrailed_enqueues_ += index - target_trail_index;
340  for (int i = target_trail_index; i < index; ++i) {
341  assignment_.UnassignLiteral(trail_[i]);
342  }
343  current_info_.trail_index = target_trail_index;
344  }
345  void Dequeue() { Untrail(Index() - 1); }
346 
347  // Changes the decision level used by the next Enqueue().
348  void SetDecisionLevel(int level) { current_info_.level = level; }
349  int CurrentDecisionLevel() const { return current_info_.level; }
350 
351  // Generic interface to set the current failing clause.
352  //
353  // Returns the address of a vector where a client can store the current
354  // conflict. This vector will be returned by the FailingClause() call.
355  std::vector<Literal>* MutableConflict() {
356  failing_sat_clause_ = nullptr;
357  return &conflict_;
358  }
359 
360  // Returns the last conflict.
361  absl::Span<const Literal> FailingClause() const { return conflict_; }
362 
363  // Specific SatClause interface so we can update the conflict clause activity.
364  // Note that MutableConflict() automatically sets this to nullptr, so we can
365  // know whether or not the last conflict was caused by a clause.
366  void SetFailingSatClause(SatClause* clause) { failing_sat_clause_ = clause; }
367  SatClause* FailingSatClause() const { return failing_sat_clause_; }
368 
369  // Getters.
370  int NumVariables() const { return trail_.size(); }
371  int64 NumberOfEnqueues() const { return num_untrailed_enqueues_ + Index(); }
372  int Index() const { return current_info_.trail_index; }
373  const Literal operator[](int index) const { return trail_[index]; }
374  const VariablesAssignment& Assignment() const { return assignment_; }
375  const AssignmentInfo& Info(BooleanVariable var) const {
376  DCHECK_GE(var, 0);
377  DCHECK_LT(var, info_.size());
378  return info_[var];
379  }
380 
381  // Print the current literals on the trail.
382  std::string DebugString() {
383  std::string result;
384  for (int i = 0; i < current_info_.trail_index; ++i) {
385  if (!result.empty()) result += " ";
386  result += trail_[i].DebugString();
387  }
388  return result;
389  }
390 
391  void SetLastPolarity(BooleanVariable var, bool polarity) {
392  info_[var].last_polarity = polarity;
393  }
394 
395  private:
396  int64 num_untrailed_enqueues_ = 0;
397  AssignmentInfo current_info_;
398  VariablesAssignment assignment_;
399  std::vector<Literal> trail_;
400  std::vector<Literal> conflict_;
401  gtl::ITIVector<BooleanVariable, AssignmentInfo> info_;
402  SatClause* failing_sat_clause_;
403 
404  // Data used by EnqueueWithSameReasonAs().
405  gtl::ITIVector<BooleanVariable, BooleanVariable>
406  reference_var_with_same_reason_as_;
407 
408  // Reason cache. Mutable since we want the API to be the same whether the
409  // reason are cached or not.
410  //
411  // When a reason is computed for the first time, we change the type of the
412  // variable assignment to kCachedReason so that we know that if it is needed
413  // again the reason can just be retrieved by a direct access to reasons_. The
414  // old type is saved in old_type_ and can be retrieved by
415  // AssignmentType().
416  //
417  // Note(user): Changing the type is not "clean" but it is efficient. The idea
418  // is that it is important to do as little as possible when pushing/popping
419  // literals on the trail. Computing the reason happens a lot less often, so it
420  // is okay to do slightly more work then. Note also, that we don't need to
421  // do anything on "untrail", the kCachedReason type will be overwritten when
422  // the same variable is assigned again.
423  //
424  // TODO(user): An alternative would be to change the sign of the type. This
425  // would remove the need for a separate old_type_ vector, but it requires
426  // more bits for the type filed in AssignmentInfo.
427  //
428  // Note that we use a deque for the reason repository so that if we add
429  // variables, the memory address of the vectors (kept in reasons_) are still
430  // valid.
431  mutable std::deque<std::vector<Literal>> reasons_repository_;
432  mutable gtl::ITIVector<BooleanVariable, absl::Span<const Literal>> reasons_;
433  mutable gtl::ITIVector<BooleanVariable, int> old_type_;
434 
435  // This is used by RegisterPropagator() and Reason().
436  std::vector<SatPropagator*> propagators_;
437 
438  DISALLOW_COPY_AND_ASSIGN(Trail);

441 // Base class for all the SAT constraints.
442 class SatPropagator {
443  public:
444  explicit SatPropagator(const std::string& name)
445  : name_(name), propagator_id_(-1), propagation_trail_index_(0) {}
446  virtual ~SatPropagator() {}
447 
448  // Sets/Gets this propagator unique id.
449  void SetPropagatorId(int id) { propagator_id_ = id; }
450  int PropagatorId() const { return propagator_id_; }
451 
452  // Inspects the trail from propagation_trail_index_ until at least one literal
453  // is propagated. Returns false iff a conflict is detected (in which case
454  // trail->SetFailingClause() must be called).
455  //
456  // This must update propagation_trail_index_ so that all the literals before
457  // it have been propagated. In particular, if nothing was propagated, then
458  // PropagationIsDone() must return true.
459  virtual bool Propagate(Trail* trail) = 0;
460 
461  // Reverts the state so that all the literals with a trail index greater or
462  // equal to the given one are not processed for propagation. Note that the
463  // trail current decision level is already reverted before this is called.
464  //
465  // TODO(user): Currently this is called at each Backtrack(), but we could
466  // bundle the calls in case multiple conflict one after the other are detected
467  // even before the Propagate() call of a SatPropagator is called.
468  //
469  // TODO(user): It is not yet 100% the case, but this can be guaranteed to be
470  // called with a trail index that will always be the start of a new decision
471  // level.
472  virtual void Untrail(const Trail& trail, int trail_index) {
473  propagation_trail_index_ = std::min(propagation_trail_index_, trail_index);
474  }
475 
476  // Explains why the literal at given trail_index was propagated by returning a
477  // reason for this propagation. This will only be called for literals that are
478  // on the trail and were propagated by this class.
479  //
480  // The interpretation is that because all the literals of a reason were
481  // assigned to false, we could deduce the assignement of the given variable.
482  //
483  // The returned Span has to be valid until the literal is untrailed. A client
484  // can use trail_.GetEmptyVectorToStoreReason() if it doesn't have a memory
485  // location that already contains the reason.
486  virtual absl::Span<const Literal> Reason(const Trail& trail,
487  int trail_index) const {
488  LOG(FATAL) << "Not implemented.";
489  return {};
490  }
491 
492  // Returns true if all the preconditions for Propagate() are satisfied.
493  // This is just meant to be used in a DCHECK.
494  bool PropagatePreconditionsAreSatisfied(const Trail& trail) const;
495 
496  // Returns true iff all the trail was inspected by this propagator.
497  bool PropagationIsDone(const Trail& trail) const {
498  return propagation_trail_index_ == trail.Index();
499  }
500 
501  protected:
502  const std::string name_;
503  int propagator_id_;
504  int propagation_trail_index_;
505 
506  private:
507  DISALLOW_COPY_AND_ASSIGN(SatPropagator);
508 };

514 inline bool SatPropagator::PropagatePreconditionsAreSatisfied(
515  const Trail& trail) const {
516  if (propagation_trail_index_ > trail.Index()) {
517  LOG(INFO) << "Issue in '" << name_ << ":"
518  << " propagation_trail_index_=" << propagation_trail_index_
519  << " trail_.Index()=" << trail.Index();
520  return false;
521  }
522  if (propagation_trail_index_ < trail.Index() &&
523  trail.Info(trail[propagation_trail_index_].Variable()).level !=
524  trail.CurrentDecisionLevel()) {
525  LOG(INFO) << "Issue in '" << name_ << "':"
526  << " propagation_trail_index_=" << propagation_trail_index_
527  << " trail_.Index()=" << trail.Index()
528  << " level_at_propagation_index="
529  << trail.Info(trail[propagation_trail_index_].Variable()).level
530  << " current_decision_level=" << trail.CurrentDecisionLevel();
531  return false;
532  }
533  return true;
534 }

557 inline BooleanVariable Trail::ReferenceVarWithSameReason(

569 inline int Trail::AssignmentType(BooleanVariable var) const {
570  if (info_[var].type == AssignmentType::kSameReasonAs) {
571  var = reference_var_with_same_reason_as_[var];
572  DCHECK_NE(info_[var].type, AssignmentType::kSameReasonAs);
573  }
574  const int type = info_[var].type;
575  return type != AssignmentType::kCachedReason ? type : old_type_[var];

578 inline absl::Span<const Literal> Trail::Reason(BooleanVariable var) const {
579  // Special case for AssignmentType::kSameReasonAs to avoid a recursive call.
580  var = ReferenceVarWithSameReason(var);
581 
582  // Fast-track for cached reason.
583  if (info_[var].type == AssignmentType::kCachedReason) return reasons_[var];
584 
585  const AssignmentInfo& info = info_[var];
586  if (info.type == AssignmentType::kUnitReason ||
587  info.type == AssignmentType::kSearchDecision) {
588  reasons_[var] = {};
589  } else {
590  DCHECK_LT(info.type, propagators_.size());
591  DCHECK(propagators_[info.type] != nullptr) << info.type;
592  reasons_[var] = propagators_[info.type]->Reason(*this, info.trail_index);
593  }
594  old_type_[var] = info.type;
595  info_[var].type = AssignmentType::kCachedReason;
596  return reasons_[var];
597 }
