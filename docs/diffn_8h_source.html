149 // Enforces that the boxes with corners in (x, y), (x + dx, y), (x, y + dy)
150 // and (x + dx, y + dy) do not overlap.
151 // If strict is true, and if one box has a zero dimension, it still cannot
152 // intersect another box.
153 inline std::function<void(Model*)> NonOverlappingRectangles(
154  const std::vector<IntervalVariable>& x,
155  const std::vector<IntervalVariable>& y, bool is_strict) {
156  return [=](Model* model) {
157  SchedulingConstraintHelper* x_helper =
158  new SchedulingConstraintHelper(x, model);
159  SchedulingConstraintHelper* y_helper =
160  new SchedulingConstraintHelper(y, model);
161  model->TakeOwnership(x_helper);
162  model->TakeOwnership(y_helper);
163 
164  NonOverlappingRectanglesEnergyPropagator* energy_constraint =
165  new NonOverlappingRectanglesEnergyPropagator(x_helper, y_helper);
166  GenericLiteralWatcher* const watcher =
167  model->GetOrCreate<GenericLiteralWatcher>();
168  watcher->SetPropagatorPriority(energy_constraint->RegisterWith(watcher), 3);
169  model->TakeOwnership(energy_constraint);
170 
171  NonOverlappingRectanglesDisjunctivePropagator* constraint =
172  new NonOverlappingRectanglesDisjunctivePropagator(is_strict, x_helper,
173  y_helper, model);
174  constraint->Register(/*fast_priority=*/3, /*slow_priority=*/4);
175  model->TakeOwnership(constraint);
176 
177  AddCumulativeRelaxation(x, x_helper, y_helper, model);
178  AddCumulativeRelaxation(y, y_helper, x_helper, model);
179  };
180 }
