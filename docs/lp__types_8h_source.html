99 // Different statuses for a given problem.
100 enum class ProblemStatus : int8 {
101  // The problem has been solved to optimality. Both the primal and dual have
102  // a feasible solution.
103  OPTIMAL,
104 
105  // The problem has been proven primal-infeasible. Note that the problem is not
106  // necessarily DUAL_UNBOUNDED (See Chvatal p.60). The solver does not have a
107  // dual unbounded ray in this case.
108  PRIMAL_INFEASIBLE,
109 
110  // The problem has been proven dual-infeasible. Note that the problem is not
111  // necessarily PRIMAL_UNBOUNDED (See Chvatal p.60). The solver does
112  // note have a primal unbounded ray in this case,
113  DUAL_INFEASIBLE,
114 
115  // The problem is either INFEASIBLE or UNBOUNDED (this applies to both the
116  // primal and dual algorithms). This status is only returned by the presolve
117  // step and means that a primal or dual unbounded ray was found during
118  // presolve. Note that because some presolve techniques assume that a feasible
119  // solution exists to simplify the problem further, it is difficult to
120  // distinguish between infeasibility and unboundedness.
121  //
122  // If a client needs to distinguish, it is possible to run the primal
123  // algorithm on the same problem with a 0 objective function to know if the
124  // problem was PRIMAL_INFEASIBLE.
125  INFEASIBLE_OR_UNBOUNDED,
126 
127  // The problem has been proven feasible and unbounded. That means that the
128  // problem is DUAL_INFEASIBLE and that the solver has a primal unbounded ray.
129  PRIMAL_UNBOUNDED,
130 
131  // The problem has been proven dual-feasible and dual-unbounded. That means
132  // the problem is PRIMAL_INFEASIBLE and that the solver has a dual unbounded
133  // ray to prove it.
134  DUAL_UNBOUNDED,
135 
136  // All the statuses below correspond to a case where the solver was
137  // interrupted. This can happen because of a timeout, an iteration limit or an
138  // error.
139 
140  // The solver didn't had a chance to prove anything.
141  INIT,
142 
143  // The problem has been proven primal-feasible but may still be
144  // PRIMAL_UNBOUNDED.
145  PRIMAL_FEASIBLE,
146 
147  // The problem has been proven dual-feasible, but may still be DUAL_UNBOUNDED.
148  // That means that if the primal is feasible, then it has a finite optimal
149  // solution.
150  DUAL_FEASIBLE,
151 
152  // An error occurred during the solving process.
153  ABNORMAL,
154 
155  // The input problem was invalid (see LinearProgram.IsValid()).
156  INVALID_PROBLEM,
157 
158  // The problem was solved to a feasible status, but the solution checker found
159  // the primal and/or dual infeasibilities too important for the specified
160  // parameters.
161  IMPRECISE,
162 };

189 // Different variables statuses.
190 // If a solution is OPTIMAL or FEASIBLE, then all the properties described here
191 // should be satisfied. These properties should also be true during the
192 // execution of the revised simplex algorithm, except that because of
193 // bound-shifting, the variable may not be at their exact bounds until the
194 // shifts are removed.
195 enum class VariableStatus : int8 {
196  // The basic status is special and takes precedence over all the other
197  // statuses. It means that the variable is part of the basis.
198  BASIC,
199  // Only possible status of a FIXED_VARIABLE not in the basis. The variable
200  // value should be exactly equal to its bounds (which are the same).
201  FIXED_VALUE,
202  // Only possible statuses of a non-basic variable which is not UNCONSTRAINED
203  // or FIXED. The variable value should be at its exact specified bound (which
204  // must be finite).
205  AT_LOWER_BOUND,
206  AT_UPPER_BOUND,
207  // Only possible status of an UNCONSTRAINED non-basic variable.
208  // Its value should be zero.
209  FREE,
210 };

220 // Different constraints statuses.
221 // The meaning is the same for the constraint activity relative to its bounds as
222 // it is for a variable value relative to its bounds. Actually, this is the
223 // VariableStatus of the slack variable associated to a constraint modulo a
224 // change of sign. The difference is that because of precision error, a
225 // constraint activity cannot exactly be equal to one of its bounds or to zero.
226 enum class ConstraintStatus : int8 {
227  BASIC,
228  FIXED_VALUE,
229  AT_LOWER_BOUND,
230  AT_UPPER_BOUND,

245 // Wrapper around an ITIVector to allow (and enforce) creation/resize/assign
246 // to use the index type for the size.
247 //
248 // TODO(user): This should probably move into ITIVector, but note that this
249 // version is more strict and does not allow any other size types.
250 template <typename IntType, typename T>
251 class StrictITIVector : public gtl::ITIVector<IntType, T> {
252  public:
253  typedef IntType IndexType; // g++ 4.8.1 needs this.
254  typedef gtl::ITIVector<IntType, T> ParentType;
255 // This allows for brace initialization, which is really useful in tests.
256 // It is not 'explicit' by design, so one can do vector = {...};
257 #if !defined(__ANDROID__) && (!defined(_MSC_VER) || (_MSC_VER >= 1800))
258  StrictITIVector(std::initializer_list<T> init_list) // NOLINT
259  : ParentType(init_list.begin(), init_list.end()) {}
260 #endif
261  StrictITIVector() : ParentType() {}
262  explicit StrictITIVector(IntType size) : ParentType(size.value()) {}
263  StrictITIVector(IntType size, const T& v) : ParentType(size.value(), v) {}
264  template <typename InputIteratorType>
265  StrictITIVector(InputIteratorType first, InputIteratorType last)
266  : ParentType(first, last) {}
267 
268  void resize(IntType size) { ParentType::resize(size.value()); }
269  void resize(IntType size, const T& v) { ParentType::resize(size.value(), v); }
270 
271  void reserve(IntType size) { ParentType::reserve(size.value()); }
272 
273  void assign(IntType size, const T& v) { ParentType::assign(size.value(), v); }
274 
275  IntType size() const { return IntType(ParentType::size()); }
276 
277  IntType capacity() const { return IntType(ParentType::capacity()); }
278 
279  // Since calls to resize() must use a default value, we introduce a new
280  // function for convenience to reduce the size of a vector.
281  void resize_down(IntType size) {
282  DCHECK_GE(size, IntType(0));
283  DCHECK_LE(size, IntType(ParentType::size()));
284  ParentType::resize(size.value());
285  }
286 
287  // This function can be up to 4 times faster than calling assign(size, 0).
288  // Note that it only works with StrictITIVector of basic types.
289  void AssignToZero(IntType size) {
290  resize(size, 0);
291  memset(ParentType::data(), 0, size.value() * sizeof(T));
292  }
293 };

346 // Returns true if it is more advantageous to use a dense iteration rather than
347 // using the non-zeros positions.
348 //
349 // TODO(user): The constant should depend on what algorithm is used. Clearing a
350 // dense vector is a lot more efficient than doing more complex stuff. Clean
351 // this up by extracting all the currently used constants in one place with
352 // meaningful names.
353 template <typename ScatteredRowOrCol>
354 bool ShouldUseDenseIteration(const ScatteredRowOrCol& v) {
355  if (v.non_zeros.empty()) return true;
356  const double kThresholdForUsingDenseRepresentation = 0.8;
357  return static_cast<double>(v.non_zeros.size()) >
358  kThresholdForUsingDenseRepresentation *
359  static_cast<double>(v.values.size().value());
360 }

362 // A simple struct that contains a DenseVector and its non-zeros indices.
363 template <typename Index>
364 struct ScatteredVector {
365  StrictITIVector<Index, Fractional> values;
366 
367  // This can be left empty in which case we just have the dense representation
368  // above. Otherwise, it should always be a subset of the actual non-zeros.
369  bool non_zeros_are_sorted = false;
370  std::vector<Index> non_zeros;
371 
372  // Temporary vector used in some sparse computation on the ScatteredColumn.
373  // True indicate a possible non-zero value. Note that its state is not always
374  // consistent.
375  StrictITIVector<Index, bool> is_non_zero;
376 
377  Fractional operator[](Index index) const { return values[index]; }
378  Fractional& operator[](Index index) { return values[index]; }
379 
380  // Sorting the non-zeros is not always needed, but it allows us to have
381  // exactly the same behavior while using a sparse iteration or a dense one. So
382  // we always do it after a Solve().
383  void SortNonZerosIfNeeded() {
384  if (!non_zeros_are_sorted) {
385  std::sort(non_zeros.begin(), non_zeros.end());
386  non_zeros_are_sorted = true;
387  }
388  }
389 };
