45 // This is how the SatSolver stores a clause. A clause is just a disjunction of
46 // literals. In many places, we just use std::vector<literal> to encode one. But
47 // in the critical propagation code, we use this class to remove one memory
48 // indirection.
49 class SatClause {
50  public:
51  // Creates a sat clause. There must be at least 2 literals. Smaller clause are
52  // treated separatly and never constructed. In practice, we do use
53  // BinaryImplicationGraph for the clause of size 2, so this is mainly used for
54  // size at least 3.
55  static SatClause* Create(absl::Span<const Literal> literals);
56 
57  // Non-sized delete because this is a tail-padded class.
58  void operator delete(void* p) {
59  ::operator delete(p); // non-sized delete
60  }
61 
62  // Number of literals in the clause.
63  int Size() const { return size_; }
64 
65  // Allows for range based iteration: for (Literal literal : clause) {}.
66  const Literal* const begin() const { return &(literals_[0]); }
67  const Literal* const end() const { return &(literals_[size_]); }
68 
69  // Returns the first and second literals. These are always the watched
70  // literals if the clause is attached in the LiteralWatchers.
71  Literal FirstLiteral() const { return literals_[0]; }
72  Literal SecondLiteral() const { return literals_[1]; }
73 
74  // Returns the literal that was propagated to true. This only works for a
75  // clause that just propagated this literal. Otherwise, this will just returns
76  // a literal of the clause.
77  Literal PropagatedLiteral() const { return literals_[0]; }
78 
79  // Returns the reason for the last unit propagation of this clause. The
80  // preconditions are the same as for PropagatedLiteral(). Note that we don't
81  // need to include the propagated literal.
82  absl::Span<const Literal> PropagationReason() const {
83  return absl::Span<const Literal>(&(literals_[1]), size_ - 1);
84  }
85 
86  // Returns a Span<> representation of the clause.
87  absl::Span<const Literal> AsSpan() const {
88  return absl::Span<const Literal>(&(literals_[0]), size_);
89  }
90 
91  // Removes literals that are fixed. This should only be called at level 0
92  // where a literal is fixed iff it is assigned. Aborts and returns true if
93  // they are not all false.
94  //
95  // Note that the removed literal can still be accessed in the portion [size,
96  // old_size) of literals().
97  bool RemoveFixedLiteralsAndTestIfTrue(const VariablesAssignment& assignment);
98 
99  // Rewrites a clause with another shorter one. Note that the clause shouldn't
100  // be attached when this is called.
101  void Rewrite(absl::Span<const Literal> new_clause) {
102  size_ = 0;
103  for (const Literal l : new_clause) literals_[size_++] = l;
104  }
105 
106  // Returns true if the clause is satisfied for the given assignment. Note that
107  // the assignment may be partial, so false does not mean that the clause can't
108  // be satisfied by completing the assignment.
109  bool IsSatisfied(const VariablesAssignment& assignment) const;
110 
111  // Returns true if the clause is attached to a LiteralWatchers.
112  bool IsAttached() const { return size_ > 0; }
113 
114  std::string DebugString() const;
115 
116  private:
117  // LiteralWatchers need to permute the order of literals in the clause and
118  // call LazyDetach().
119  friend class LiteralWatchers;
120 
121  Literal* literals() { return &(literals_[0]); }
122 
123  // Marks the clause so that the next call to CleanUpWatchers() can identify it
124  // and actually detach it. We use size_ = 0 for this since the clause will
125  // never be used afterwards.
126  void LazyDetach() { size_ = 0; }
127 
128  int32 size_;
129 
130  // This class store the literals inline, and literals_ mark the starts of the
131  // variable length portion.
132  Literal literals_[0];
133 
134  DISALLOW_COPY_AND_ASSIGN(SatClause);
135 };

226  // Really basic algorithm to return a clause to try to minimize. We simply
227  // loop over the clause that we keep forever, in creation order. This starts
228  // by the problem clauses and then the learned one that we keep forever.
229  SatClause* NextClauseToMinimize() {
230  for (; to_minimize_index_ < clauses_.size(); ++to_minimize_index_) {
231  if (!clauses_[to_minimize_index_]->IsAttached()) continue;
232  if (!IsRemovable(clauses_[to_minimize_index_])) {
233  return clauses_[to_minimize_index_++];
234  }
235  }
236  return nullptr;
237  }
238 
239  // Restart the scan in NextClauseToMinimize() from the first problem clause.
240  void ResetToMinimizeIndex() { to_minimize_index_ = 0; }

318 // A binary clause. This is used by BinaryClauseManager.
319 struct BinaryClause {
320  BinaryClause(Literal _a, Literal _b) : a(_a), b(_b) {}
321  bool operator==(BinaryClause o) const { return a == o.a && b == o.b; }
322  bool operator!=(BinaryClause o) const { return a != o.a || b != o.b; }

327 // A simple class to manage a set of binary clauses.
328 class BinaryClauseManager {
329  public:
330  BinaryClauseManager() {}
331  int NumClauses() const { return set_.size(); }
332 
333  // Adds a new binary clause to the manager and returns true if it wasn't
334  // already present.
335  bool Add(BinaryClause c) {
336  std::pair<int, int> p(c.a.SignedValue(), c.b.SignedValue());
337  if (p.first > p.second) std::swap(p.first, p.second);
338  if (set_.find(p) == set_.end()) {
339  set_.insert(p);
340  newly_added_.push_back(c);
341  return true;
342  }
343  return false;
344  }
345 
346  // Returns the newly added BinaryClause since the last ClearNewlyAdded() call.
347  const std::vector<BinaryClause>& newly_added() const { return newly_added_; }
348  void ClearNewlyAdded() { newly_added_.clear(); }
349 
350  private:
351  absl::flat_hash_set<std::pair<int, int>> set_;
352  std::vector<BinaryClause> newly_added_;
353  DISALLOW_COPY_AND_ASSIGN(BinaryClauseManager);
354 };

356 // Special class to store and propagate clauses of size 2 (i.e. implication).
357 // Such clauses are never deleted. Together, they represent the 2-SAT part of
358 // the problem. Note that 2-SAT satisfiability is a polynomial problem, but
359 // W2SAT (weighted 2-SAT) is NP-complete.
360 //
361 // TODO(user): All the variables in a strongly connected component are
362 // equivalent and can be thus merged as one. This is relatively cheap to compute
363 // from time to time (linear complexity). We will also get contradiction (a <=>
364 // not a) this way.
365 //
366 // TODO(user): An implication (a => not a) implies that a is false. I am not
367 // sure it is worth detecting that because if the solver assign a to true, it
368 // will learn that right away. I don't think we can do it faster.
369 //
370 // TODO(user): The implication graph can be pruned. This is called the
371 // transitive reduction of a graph. For instance If a => {b,c} and b => {c},
372 // then there is no need to store a => {c}. The transitive reduction is unique
373 // on an acyclic graph. Computing it will allow for a faster propagation and
374 // memory reduction. It is however not cheap. Maybe simple lazy heuristics to
375 // remove redundant arcs are better. Note that all the learned clauses we add
376 // will never be redundant (but they could introduce cycles).
377 //
378 // TODO(user): Add a preprocessor to remove duplicates in the implication lists.
379 // Note that all the learned clauses we had will never create duplicates.
380 //
381 // References for most of the above TODO and more:
382 // - Brafman RI, "A simplifier for propositional formulas with many binary
383 // clauses", IEEE Trans Syst Man Cybern B Cybern. 2004 Feb;34(1):52-9.
384 // http://citeseerx.ist.psu.edu/viewdoc/summary?doi=10.1.1.28.4911
385 // - Marijn J. H. Heule, Matti Jrvisalo, Armin Biere, "Efficient CNF
386 // Simplification Based on Binary Implication Graphs", Theory and Applications
387 // of Satisfiability Testing - SAT 2011, Lecture Notes in Computer Science
388 // Volume 6695, 2011, pp 201-215
389 // http://www.cs.helsinki.fi/u/mjarvisa/papers/heule-jarvisalo-biere.sat11.pdf

511  // Extract all the binary clauses managed by this class. The Output type must
512  // support an AddBinaryClause(Literal a, Literal b) function.
513  template <typename Output>
514  void ExtractAllBinaryClauses(Output* out) const {
515  for (LiteralIndex i(0); i < implications_.size(); ++i) {
516  const Literal a = Literal(i).Negated();
517  for (const Literal b : implications_[i]) {
518  // Because we store implications, the clause will actually appear twice
519  // as (a, b) and (b, a). We output only one.
520  if (a < b) out->AddBinaryClause(a, b);
521  }
522  }
523  }
