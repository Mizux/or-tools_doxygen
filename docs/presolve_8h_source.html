30 // The Presolver "pre-solves" a Model by applying some iterative
31 // transformations to it, which may simplify and/or shrink the model.
32 //
33 // TODO(user): Error reporting of unfeasible models.
34 class Presolver {
35  public:
36  // Recursively apply all the pre-solve rules to the model, until exhaustion.
37  // The reduced model will:
38  // - Have some unused variables.
39  // - Have some unused constraints (marked as inactive).
40  // - Have some modified constraints (for example, they will no longer
41  // refer to unused variables).
42  //
43  // TODO(user): compute on the fly, and add an API to access the set of
44  // unused variables.
45  //
46  // This method returns true iff some transformations were applied to the
47  // model.
48  // TODO(user): Returns the number of rules applied instead.
49  bool Run(Model* model);

61  enum RuleStatus {
62  NOT_CHANGED = 0, // Constraint has not changed.
63  CONTEXT_CHANGED, // The constraint has not changed, but some mapping,
64  // or some variables have been updated.
65  CONSTRAINT_REWRITTEN, // The constraint has been rewritten.
66  CONSTRAINT_ALWAYS_FALSE, // The constraint is always false.
67  CONSTRAINT_ALWAYS_TRUE, // The constraint is always true, and now inactive.
68  };

70  // This struct stores the affine mapping of one variable:
71  // it represents new_var = var * coefficient + offset. It also stores the
72  // constraint that defines this mapping.
73  struct AffineMapping {
74  IntegerVariable* variable;
75  int64 coefficient;
76  int64 offset;
77  Constraint* constraint;
78 
79  AffineMapping()
80  : variable(nullptr), coefficient(0), offset(0), constraint(nullptr) {}
81  AffineMapping(IntegerVariable* v, int64 c, int64 o, Constraint* ct)
82  : variable(v), coefficient(c), offset(o), constraint(ct) {}
83  };

85  // This struct stores the mapping of two index variables (of a 2D array; not
86  // included here) onto a single index variable (of the flattened 1D array).
87  // The original 2D array could be trimmed in the process; so we also need an
88  // offset.
89  // Eg. new_index_var = index_var1 * int_coeff + index_var2 + int_offset
90  struct Array2DIndexMapping {
91  IntegerVariable* variable1;
92  int64 coefficient;
93  IntegerVariable* variable2;
94  int64 offset;
95  Constraint* constraint;
96 
97  Array2DIndexMapping()
98  : variable1(nullptr),
99  coefficient(0),
100  variable2(nullptr),
101  offset(0),
102  constraint(nullptr) {}
103  Array2DIndexMapping(IntegerVariable* v1, int64 c, IntegerVariable* v2,
104  int64 o, Constraint* ct)
105  : variable1(v1),
106  coefficient(c),
107  variable2(v2),
108  offset(o),
109  constraint(ct) {}
110  };
