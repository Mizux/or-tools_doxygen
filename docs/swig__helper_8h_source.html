30 // Base class for SWIG director based on solution callbacks.
31 // See http://www.swig.org/Doc3.0/SWIGDocumentation.html#CSharp_directors.
32 class SolutionCallback {
33  public:
34  virtual ~SolutionCallback() {}

36  virtual void OnSolutionCallback() const = 0;
37 
38  void Run(const operations_research::sat::CpSolverResponse& response) const {
39  response_ = response;
40  has_response_ = true;
41  OnSolutionCallback();
42  }
43 
44  int64 NumBooleans() const { return response_.num_booleans(); }
45 
46  int64 NumBranches() const { return response_.num_branches(); }

48  int64 NumConflicts() const { return response_.num_conflicts(); }

55  return response_.num_integer_propagations();

60  double UserTime() const { return response_.user_time(); }
61 
62  double DeterministicTime() const { return response_.deterministic_time(); }
63 
64  double ObjectiveValue() const { return response_.objective_value(); }
65 
66  double BestObjectiveBound() const { return response_.best_objective_bound(); }
67 
68  int64 SolutionIntegerValue(int index) {
69  return index >= 0 ? response_.solution(index)
70  : -response_.solution(-index - 1);
71  }
72 
73  bool SolutionBooleanValue(int index) {
74  return index >= 0 ? response_.solution(index) != 0
75  : response_.solution(-index - 1) == 0;
76  }
77 
78  void StopSearch() { stopped_ = true; }
79 
80  std::atomic<bool>* stopped() const { return &stopped_; }
81 
82  operations_research::sat::CpSolverResponse Response() const {
83  return response_;
84  }
85 
86  bool HasResponse() const { return has_response_; }
87 
88  private:
89  mutable CpSolverResponse response_;
90  mutable bool has_response_ = false;
91  mutable std::atomic<bool> stopped_;
92 };

94 class SatHelper {
95  public:
96  // The arguments of the functions defined below must follow these rules
97  // to be wrapped by swig correctly:
98  // 1) Their types must include the full operations_research::sat::
99  // namespace.
100  // 2) Their names must correspond to the ones declared in the .i
101  // file (see the python/ and java/ subdirectories).
102  // 3) String variations of the parameters have been added for C# as
103  // C# protobufs do not support proto2.
104  static operations_research::sat::CpSolverResponse Solve(
105  const operations_research::sat::CpModelProto& model_proto) {
106  FixFlagsAndEnvironmentForSwig();
107  Model model;
108  std::atomic<bool> stopped(false);
109  model.GetOrCreate<TimeLimit>()->RegisterExternalBooleanAsLimit(&stopped);
110  model.GetOrCreate<SigintHandler>()->Register(
111  [&stopped]() { stopped = true; });
112 
113  return operations_research::sat::SolveCpModel(model_proto, &model);
114  }
115 
116  static operations_research::sat::CpSolverResponse SolveWithParameters(
117  const operations_research::sat::CpModelProto& model_proto,
118  const operations_research::sat::SatParameters& parameters) {
119  FixFlagsAndEnvironmentForSwig();

121  model.Add(NewSatParameters(parameters));
122  std::atomic<bool> stopped(false);
123  model.GetOrCreate<TimeLimit>()->RegisterExternalBooleanAsLimit(&stopped);
124  model.GetOrCreate<SigintHandler>()->Register(
125  [&stopped]() { stopped = true; });
126 
127  return SolveCpModel(model_proto, &model);
128  }

144  static operations_research::sat::CpSolverResponse
145  SolveWithParametersAndSolutionCallback(
146  const operations_research::sat::CpModelProto& model_proto,
147  const operations_research::sat::SatParameters& parameters,
148  const SolutionCallback& callback) {
149  FixFlagsAndEnvironmentForSwig();
150  Model model;
151  model.Add(NewSatParameters(parameters));
152  model.Add(NewFeasibleSolutionObserver(
153  [&callback](const CpSolverResponse& r) { return callback.Run(r); }));
154  model.GetOrCreate<TimeLimit>()->RegisterExternalBooleanAsLimit(
155  callback.stopped());
156  model.GetOrCreate<SigintHandler>()->Register(
157  [&callback]() { *callback.stopped() = true; });
158 
159  return SolveCpModel(model_proto, &model);
160  }

162  static operations_research::sat::CpSolverResponse
163  SolveWithStringParametersAndSolutionCallback(
164  const operations_research::sat::CpModelProto& model_proto,
165  const std::string& parameters, const SolutionCallback& callback) {
166  FixFlagsAndEnvironmentForSwig();

169  model.Add(NewFeasibleSolutionObserver(

171  model.GetOrCreate<TimeLimit>()->RegisterExternalBooleanAsLimit(
172  callback.stopped());
173  model.GetOrCreate<SigintHandler>()->Register(
174  [&callback]() { *callback.stopped() = true; });
175 
176  return SolveCpModel(model_proto, &model);
177  }
