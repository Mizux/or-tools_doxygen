28 // Represents a closed interval [start, end]. We must have start <= end.
29 struct ClosedInterval {
30  int64 start; // Inclusive.
31  int64 end; // Inclusive.
32 
33  std::string DebugString() const;
34  bool operator==(const ClosedInterval& other) const {
35  return start == other.start && end == other.end;
36  }
37 
38  // Because we mainly manipulate vector of disjoint intervals, we only need to
39  // sort by the start. We do not care about the order in which interval with
40  // the same start appear since they will always be merged into one interval.
41  bool operator<(const ClosedInterval& other) const {
42  return start < other.start;
43  }

57 // We call "domain" any subset of Int64 = [kint64min, kint64max].
58 //
59 // This class can be used to represent such set efficiently as a sorted and
60 // non-adjacent list of intervals. This is efficient as long as the size of such
61 // list stays reasonable.

64 // Note that all the functions are safe with respect to integer overflow.

139  // Returns a super-set of MultiplicationBy() to avoid the explosion in the
140  // representation size. This behaves as if we replace the set D of
141  // non-adjacent integer intervals by the set of floating-point element in the
142  // same intervals.
143  //
144  // For instance, [1, 100] * 2 will be transformed in [2, 200] and not in
145  // [2][4][6]...[200] like in MultiplicationBy(). Note that this would be
146  // similar to a InverseDivisionBy(), but not quite the same because if we

156  // Returns {x ∈ Int64, ∃ e ∈ D, x * coeff = e}.

158  // For instance Domain(1, 7).InverseMultiplicationBy(2) == Domain(1, 3).

161  // Advanced usage. Given some "implied" information on this domain that is
162  // assumed to be always true (i.e. only values in the intersection with
163  // implied domain matter), this function will simplify the current domain
164  // without changing the set of "possible values".
165  //
166  // More precisely, this will:
167  // - Take the intersection with implied_domain.
168  // - Minimize the number of intervals. That is, if the
169  // domain is like [1,2][4] and implied is [1][4], then the domain can be
170  // relaxed to [1, 4] to simplify its complexity without changing the set
171  // of admissible value assuming only implied values can be seen.
172  // - Restrict as much as possible the bounds of the remaining intervals.
173  // I.e if the input is [1,2] and implied is [0,4], then the domain will
174  // not be changed.
175  //
176  // Note that domain.SimplifyUsingImpliedDomain(domain) will just return
177  // [domain.Min(), domain.Max()]. This is meant to be applied to the rhs of a
178  // constraint to make its propagation more efficient.
179  Domain SimplifyUsingImpliedDomain(const Domain& implied_domain) const;

233 // This class represents a sorted list of disjoint, closed intervals. When an
234 // interval is inserted, all intervals that overlap it or that are even adjacent
235 // to it are merged into one. I.e. [0,14] and [15,30] will be merged to [0,30].
236 //
237 // Iterators returned by this class are invalidated by non-const operations.
238 //
239 // TODO(user): Templatize the class on the type of the bounds.
240 class SortedDisjointIntervalList {
241  public:
242  struct IntervalComparator {
243  bool operator()(const ClosedInterval& a, const ClosedInterval& b) const {
244  return a.start != b.start ? a.start < b.start : a.end < b.end;
245  }
246  };
247  typedef std::set<ClosedInterval, IntervalComparator> IntervalSet;
248  typedef IntervalSet::iterator Iterator;
249 
250  SortedDisjointIntervalList();
251  explicit SortedDisjointIntervalList(
252  const std::vector<ClosedInterval>& intervals);
253 
254  // Creates a SortedDisjointIntervalList and fills it with intervals
255  // [starts[i]..ends[i]]. All intervals must be consistent (starts[i] <=
256  // ends[i]). There's two version, one for int64, one for int.
257  //
258  // TODO(user): Explain why we favored this API to the more natural
259  // input std::vector<ClosedInterval> or std::vector<std::pair<int, int>>.
260  SortedDisjointIntervalList(const std::vector<int64>& starts,
261  const std::vector<int64>& ends);
262  SortedDisjointIntervalList(const std::vector<int>& starts,
263  const std::vector<int>& ends);
264 
265  // Builds the complement of the interval list on the interval [start, end].
266  SortedDisjointIntervalList BuildComplementOnInterval(int64 start, int64 end);
267 
268  // Adds the interval [start..end] to the list, and merges overlapping or
269  // immediately adjacent intervals ([2, 5] and [6, 7] are adjacent, but
270  // [2, 5] and [7, 8] are not).
271  //
272  // Returns an iterator to the inserted interval (possibly merged with others).
273  //
274  // If start > end, it does LOG(DFATAL) and returns end() (no interval added).
275  Iterator InsertInterval(int64 start, int64 end);
276 
277  // If value is in an interval, increase its end by one, otherwise insert the
278  // interval [value, value]. In both cases, this returns an iterator to the
279  // new/modified interval (possibly merged with others) and fills newly_covered
280  // with the new value that was just added in the union of all the intervals.
281  //
282  // If this causes an interval ending at kint64max to grow, it will die with a
283  // CHECK fail.
284  Iterator GrowRightByOne(int64 value, int64* newly_covered);
285 
286  // Adds all intervals [starts[i]..ends[i]]. Same behavior as InsertInterval()
287  // upon invalid intervals. There's a version with int64 and int32.
288  void InsertIntervals(const std::vector<int64>& starts,
289  const std::vector<int64>& ends);
290  void InsertIntervals(const std::vector<int>& starts,
291  const std::vector<int>& ends);
292 
293  // Returns the number of disjoint intervals in the list.
294  int NumIntervals() const { return intervals_.size(); }
295 
296  // Returns an iterator to either:
297  // - the first interval containing or above the given value, or
298  // - the last interval containing or below the given value.
299  // Returns end() if no interval fulfils that condition.
300  //
301  // If the value is within an interval, both functions will return it.
302  Iterator FirstIntervalGreaterOrEqual(int64 value) const;
303  Iterator LastIntervalLessOrEqual(int64 value) const;
304 
305  std::string DebugString() const;
306 
307  // This is to use range loops in C++:
308  // SortedDisjointIntervalList list;
309  // ...
310  // for (const ClosedInterval interval : list) {
311  // ...
312  // }
313  const Iterator begin() const { return intervals_.begin(); }
314  const Iterator end() const { return intervals_.end(); }
315 
316  // Returns a const& to the last interval. The list must not be empty.
317  const ClosedInterval& last() const { return *intervals_.rbegin(); }
318 
319  void clear() { intervals_.clear(); }
320  void swap(SortedDisjointIntervalList& other) {
321  intervals_.swap(other.intervals_);
322  }
323

328  IntervalSet intervals_;
