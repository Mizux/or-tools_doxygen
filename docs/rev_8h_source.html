26 // Interface for reversible objects used to maintain them in sync with a tree
27 // search organized by decision levels.
28 class ReversibleInterface {
29  public:
30  ReversibleInterface() {}
31  virtual ~ReversibleInterface() {}
32 
33  // Initially a reversible class starts at level zero. Increasing the level
34  // saves the state of the current old level. Decreasing the level restores the
35  // state to what it was at this level and all higher levels are forgotten.
36  // Everything done at level zero cannot be backtracked over.
37  //
38  // The level is assumed to be non-negative.
39  virtual void SetLevel(int level) = 0;
40 };

42 // A repository that maintains a set of reversible objects of type T.
43 // This is meant to be used for small types that are efficient to copy, like
44 // all the basic types, std::pair and things like this.
45 template <class T>
46 class RevRepository : public ReversibleInterface {
47  public:
48  RevRepository() : stamp_(0) {}
49 
50  // This works in O(level_diff) on level increase.
51  // For level decrease, it is in O(level_diff + num_restored_states).
52  void SetLevel(int level) final;
53  int Level() const { return end_of_level_.size(); }
54 
55  // Saves the given object value for the current level. If this is called
56  // multiple time by level, only the value of the first call matter. This is
57  // NOT optimized for many calls by level and should mainly be used just once
58  // for a given level. If a client cannot do that efficiently, it can use the
59  // SaveStateWithStamp() function below.
60  void SaveState(T* object) {
61  if (end_of_level_.empty()) return; // Not useful for level zero.
62  stack_.push_back({object, *object});
63  }
64 
65  // Calls SaveState() if the given stamp is not the same as the current one.
66  // This also sets the given stamp to the current one. The current stamp is
67  // maintained by this class and is updated on each level changes. The whole
68  // process make sure that only one SaveValue() par level will ever be called,
69  // so it is efficient to call this before each update to the object T.
70  void SaveStateWithStamp(T* object, int64* stamp) {
71  if (*stamp == stamp_) return;
72  *stamp = stamp_;
73  SaveState(object);
74  }
75 
76  private:
77  int64 stamp_;
78  std::vector<int> end_of_level_; // In stack_.
79 
80  // TODO(user): If we ever see this in any cpu profile, consider using two
81  // vectors for a better memory packing in case sizeof(T) is not sizeof(T*).
82  std::vector<std::pair<T*, T>> stack_;
83 };

85 template <class T>
86 void RevRepository<T>::SetLevel(int level) {
87  DCHECK_GE(level, 0);
88  if (level == Level()) return;
89  ++stamp_;
90  if (level < Level()) {
91  const int index = end_of_level_[level];
92  end_of_level_.resize(level); // Shrinks.
93  while (stack_.size() > index) {
94  const auto& p = stack_.back();
95  *p.first = p.second;
96  stack_.pop_back();
97  }
98  } else {
99  end_of_level_.resize(level, stack_.size()); // Grows.
100  }
101 }

103 // Like a normal map but support backtrackable operations.
104 //
105 // This works on any class "Map" that supports: begin(), end(), find(), erase(),
106 // insert(), key_type, value_type, mapped_type and const_iterator.
107 template <class Map>
108 class RevMap : ReversibleInterface {
109  public:
110  typedef typename Map::key_type key_type;
111  typedef typename Map::mapped_type mapped_type;
112  typedef typename Map::value_type value_type;
113  typedef typename Map::const_iterator const_iterator;
114 
115  // Backtracking support: changes the current "level" (always non-negative).
116  //
117  // Initially the class starts at level zero. Increasing the level works in
118  // O(level diff) and saves the state of the current old level. Decreasing the
119  // level restores the state to what it was at this level and all higher levels
120  // are forgotten. Everything done at level zero cannot be backtracked over.
121  void SetLevel(int level) final;
122  int Level() const { return first_op_index_of_next_level_.size(); }
123 
124  bool ContainsKey(key_type key) const { return gtl::ContainsKey(map_, key); }
125  const mapped_type& FindOrDie(key_type key) const {
126  return gtl::FindOrDie(map_, key);
127  }
128 
129  void EraseOrDie(key_type key);
130  void Set(key_type key, mapped_type value); // Adds or overwrites.
131 
132  // Wrapper to the underlying const map functions.
133  int size() const { return map_.size(); }
134  bool empty() const { return map_.empty(); }
135  const_iterator find(const key_type& k) const { return map_.find(k); }
136  const_iterator begin() const { return map_.begin(); }
137  const_iterator end() const { return map_.end(); }
138 
139  private:
140  Map map_;
141 
142  // The operation that needs to be performed to reverse one modification:
143  // - If is_deletion is true, then we need to delete the entry with given key.
144  // - Otherwise we need to add back (or overwrite) the saved entry.
145  struct UndoOperation {
146  bool is_deletion;
147  key_type key;
148  mapped_type value;
149  };
150 
151  // TODO(user): We could merge the operations with the same key from the same
152  // level. Investigate and implement if this is worth the effort for our use
153  // case.
154  std::vector<UndoOperation> operations_;
155  std::vector<int> first_op_index_of_next_level_;
156 };
157 
158 template <class Map>
159 void RevMap<Map>::SetLevel(int level) {
160  DCHECK_GE(level, 0);
161  if (level < Level()) {
162  const int backtrack_level = first_op_index_of_next_level_[level];
163  first_op_index_of_next_level_.resize(level); // Shrinks.
164  while (operations_.size() > backtrack_level) {
165  const UndoOperation& to_undo = operations_.back();
166  if (to_undo.is_deletion) {
167  map_.erase(to_undo.key);
168  } else {
169  map_.insert({to_undo.key, to_undo.value}).first->second = to_undo.value;
170  }
171  operations_.pop_back();
172  }
173  return;
174  }
175 
176  // This is ok even if level == Level().
177  first_op_index_of_next_level_.resize(level, operations_.size()); // Grows.
178 }

180 template <class Map>
181 void RevMap<Map>::EraseOrDie(key_type key) {
182  const auto iter = map_.find(key);
183  if (iter == map_.end()) LOG(FATAL) << "key not present: '" << key << "'.";
184  if (Level() > 0) {
185  operations_.push_back({false, key, iter->second});
186  }
187  map_.erase(iter);
188 }

190 template <class Map>
191 void RevMap<Map>::Set(key_type key, mapped_type value) {
192  auto insertion_result = map_.insert({key, value});
193  if (Level() > 0) {
194  if (insertion_result.second) {
195  // It is an insertion. Undo = delete.
196  operations_.push_back({true, key});
197  } else {
198  // It is a modification. Undo = change back to old value.
199  operations_.push_back({false, key, insertion_result.first->second});
200  }
201  }
202  insertion_result.first->second = value;
203 }

205 // A basic backtrackable multi map that can only grow (except on backtrack).
206 template <class Key, class Value>
207 class RevGrowingMultiMap : ReversibleInterface {
208  public:
209  void SetLevel(int level) final;
210 
211  // Adds a new value at the given key.
212  void Add(Key key, Value value);
213 
214  // Returns the list of values for a given key (can be empty).
215  const std::vector<Value>& Values(Key key) const;
216 
217  private:
218  std::vector<Value> empty_values_;
219 
220  // TODO(user): use inlined vectors. Another datastructure that may be more
221  // efficient is to use a linked list inside added_keys_ for the values sharing
222  // the same key.
223  absl::flat_hash_map<Key, std::vector<Value>> map_;
224 
225  // Backtracking data.
226  std::vector<Key> added_keys_;
227  std::vector<int> first_added_key_of_next_level_;
228 };
229 
230 template <class Key, class Value>
231 void RevGrowingMultiMap<Key, Value>::SetLevel(int level) {
232  DCHECK_GE(level, 0);
233  if (level < first_added_key_of_next_level_.size()) {
234  const int backtrack_level = first_added_key_of_next_level_[level];
235  first_added_key_of_next_level_.resize(level); // Shrinks.
236  while (added_keys_.size() > backtrack_level) {
237  auto it = map_.find(added_keys_.back());
238  if (it->second.size() > 1) {
239  it->second.pop_back();
240  } else {
241  map_.erase(it);
242  }
243  added_keys_.pop_back();
244  }
245  return;
246  }
247 
248  // This is ok even if level == Level().
249  first_added_key_of_next_level_.resize(level, added_keys_.size()); // Grows.
250 }

252 template <class Key, class Value>
253 const std::vector<Value>& RevGrowingMultiMap<Key, Value>::Values(
254  Key key) const {
255  const auto it = map_.find(key);
256  if (it != map_.end()) return it->second;
257  return empty_values_;
258 }

260 template <class Key, class Value>
261 void RevGrowingMultiMap<Key, Value>::Add(Key key, Value value) {
262  if (!first_added_key_of_next_level_.empty()) {
263  added_keys_.push_back(key);
264  }
265  map_[key].push_back(value);
