36 // Holds the mapping between CpModel proto indices and the sat::model ones.
37 //
38 // This also holds some information used when loading a CpModel proto.
39 class CpModelMapping {
40  public:
41  // Extracts all the used variables in the CpModelProto and creates a
42  // sat::Model representation for them. More precisely
43  // - All Boolean variables will be mapped.
44  // - All Interval variables will be mapped.
45  // - All non-Boolean variable will have a corresponding IntegerVariable, and
46  // depending on the view_all_booleans_as_integers, some or all of the
47  // BooleanVariable will also have an IntegerVariable corresponding to its
48  // "integer view".
49  //
50  // Note(user): We could create IntegerVariable on the fly as they are needed,
51  // but that loose the original variable order which might be useful in
52  // heuristics later.
53  void CreateVariables(const CpModelProto& model_proto,
54  bool view_all_booleans_as_integers, Model* m);
55 
56  // Automatically detect optional variables.
57  void DetectOptionalVariables(const CpModelProto& model_proto, Model* m);
58 
59  // Extract the encodings (IntegerVariable <-> Booleans) present in the model.
60  // This effectively load some linear constraints of size 1 that will be marked
61  // as already loaded.
62  void ExtractEncoding(const CpModelProto& model_proto, Model* m);
63 
64  // Returns true if the given CpModelProto variable reference refers to a
65  // Boolean varaible. Such variable will always have an associated Literal(),
66  // but not always an associated Integer().
67  bool IsBoolean(int ref) const {
68  DCHECK_LT(PositiveRef(ref), booleans_.size());
69  return booleans_[PositiveRef(ref)] != kNoBooleanVariable;
70  }
71 
72  bool IsInteger(int ref) const {
73  DCHECK_LT(PositiveRef(ref), integers_.size());
74  return integers_[PositiveRef(ref)] != kNoIntegerVariable;
75  }
76 
77  sat::Literal Literal(int ref) const {
78  DCHECK(IsBoolean(ref));
79  return sat::Literal(booleans_[PositiveRef(ref)], RefIsPositive(ref));
80  }
81 
82  IntegerVariable Integer(int ref) const {
83  DCHECK(IsInteger(ref));
84  const IntegerVariable var = integers_[PositiveRef(ref)];
85  return RefIsPositive(ref) ? var : NegationOf(var);
86  }
87 
88  IntervalVariable Interval(int i) const {
89  CHECK_GE(i, 0);
90  CHECK_LT(i, intervals_.size());
91  CHECK_NE(intervals_[i], kNoIntervalVariable);
92  return intervals_[i];
93  }
94 
95  template <typename List>
96  std::vector<IntegerVariable> Integers(const List& list) const {
97  std::vector<IntegerVariable> result;
98  for (const auto i : list) result.push_back(Integer(i));
99  return result;
100  }
101 
102  template <typename ProtoIndices>
103  std::vector<sat::Literal> Literals(const ProtoIndices& indices) const {
104  std::vector<sat::Literal> result;
105  for (const int i : indices) result.push_back(CpModelMapping::Literal(i));
106  return result;
107  }
108 
109  template <typename ProtoIndices>
110  std::vector<IntervalVariable> Intervals(const ProtoIndices& indices) const {
111  std::vector<IntervalVariable> result;
112  for (const int i : indices) result.push_back(Interval(i));
113  return result;
114  }
115 
116  // Depending on the option, we will load constraints in stages. This is used
117  // to detect constraints that are already loaded. For instance the interval
118  // constraints and the linear constraint of size 1 (encodings) are usually
119  // loaded first.
120  bool ConstraintIsAlreadyLoaded(const ConstraintProto* ct) const {
121  return already_loaded_ct_.contains(ct);
122  }
123 
124  // Returns true if the given constraint is a "half-encoding" constraint. That
125  // is, if it is of the form (b => size 1 linear) but there is no (<=) side in
126  // the model. Such constraint are detected while we extract integer encoding
127  // and are cached here so that we can deal properly with them during the
128  // linear relaxation.
129  bool IsHalfEncodingConstraint(const ConstraintProto* ct) const {
130  return is_half_encoding_ct_.contains(ct);
131  }
132 
133  // Note that both these functions returns positive reference or -1.
134  int GetProtoVariableFromBooleanVariable(BooleanVariable var) const {
135  if (var.value() >= reverse_boolean_map_.size()) return -1;
136  return reverse_boolean_map_[var];
137  }
138  int GetProtoVariableFromIntegerVariable(IntegerVariable var) const {
139  if (var.value() >= reverse_integer_map_.size()) return -1;
140  return reverse_integer_map_[var];
141  }
142 
143  const std::vector<IntegerVariable>& GetVariableMapping() const {
144  return integers_;
145  }
146 
147  // For logging only, these are not super efficient.
148  int NumIntegerVariables() const {
149  int result = 0;
150  for (const IntegerVariable var : integers_) {
151  if (var != kNoIntegerVariable) result++;
152  }
153  return result;
154  }
155  int NumBooleanVariables() const {
156  int result = 0;
157  for (const BooleanVariable var : booleans_) {
158  if (var != kNoBooleanVariable) result++;
159  }
160  return result;
161  }
162 
163  private:
164  // Note that only the variables used by at least one constraint will be
165  // created, the other will have a kNo[Integer,Interval,Boolean]VariableValue.
166  std::vector<IntegerVariable> integers_;
167  std::vector<IntervalVariable> intervals_;
168  std::vector<BooleanVariable> booleans_;
169 
170  // Recover from a IntervalVariable/BooleanVariable its associated CpModelProto
171  // index. The value of -1 is used to indicate that there is no correspondence
172  // (i.e. this variable is only used internally).
173  gtl::ITIVector<BooleanVariable, int> reverse_boolean_map_;
174  gtl::ITIVector<IntegerVariable, int> reverse_integer_map_;
175 
176  // Set of constraints to ignore because they were already dealt with by
177  // ExtractEncoding().
178  absl::flat_hash_set<const ConstraintProto*> already_loaded_ct_;
179  absl::flat_hash_set<const ConstraintProto*> is_half_encoding_ct_;
180 };
