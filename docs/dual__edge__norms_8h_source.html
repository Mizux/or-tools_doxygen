26 // This class maintains the dual edge squared norms to be used in the
27 // dual steepest edge pricing. The dual edge u_i associated with a basic
28 // variable of row index i is such that u_i.B = e_i where e_i is the unit row
29 // vector with a 1.0 at position i and B the current basis. We call such vector
30 // u_i an unit row left inverse, and it can be computed by
31 //
32 // basis_factorization.LeftSolveForUnitRow(i, &u_i);
33 //
34 // Instead of computing each ||u_i|| at every iteration, it is more efficient to
35 // update them incrementally for each basis pivot applied to B. See the code or
36 // the papers below for details:
37 //
38 // J.J. Forrest, D. Goldfarb, "Steepest-edge simplex algorithms for linear
39 // programming", Mathematical Programming 57 (1992) 341-374, North-Holland.
40 // http://www.springerlink.com/content/q645w3t2q229m248/
41 //
42 // Achim Koberstein, "The dual simplex method, techniques for a fast and stable
43 // implementation", PhD, Paderborn, Univ., 2005.
44 // http://digital.ub.uni-paderborn.de/hs/download/pdf/3885?originalFilename=true
45 class DualEdgeNorms {
46  public:
47  // Takes references to the linear program data we need.
48  explicit DualEdgeNorms(const BasisFactorization& basis_factorization);
49 
50  // Clears, i.e. reset the object to its initial value. This will trigger a
51  // full norm recomputation on the next GetEdgeSquaredNorms().
52  void Clear();
53 
54  // If this is true, then the caller must re-factorize the basis before the
55  // next call to GetEdgeSquaredNorms(). This is because the latter will
56  // recompute the norms from scratch and therefore needs a hightened precision
57  // and speed.
58  bool NeedsBasisRefactorization();
59 
60  // Returns the dual edge squared norms. This is only valid if the caller
61  // properly called UpdateBeforeBasisPivot() before each basis pivot, or just
62  // called Clear().
63  const DenseColumn& GetEdgeSquaredNorms();
64 
65  // Updates the norms if the columns of the basis where permuted.
66  void UpdateDataOnBasisPermutation(const ColumnPermutation& col_perm);
67 
68  // Updates the norms just before a basis pivot is applied:
69  // - The column at leaving_row will leave the basis and the column at
70  // entering_col will enter it.
71  // - direction is the right inverse of the entering column.
72  // - unit_row_left_inverse is the left inverse of the unit row with index
73  // given by the leaving_row. This is also the leaving dual edge.
74  void UpdateBeforeBasisPivot(ColIndex entering_col, RowIndex leaving_row,
75  const ScatteredColumn& direction,
76  const ScatteredRow& unit_row_left_inverse);
77 
78  // Sets the algorithm parameters.
79  void SetParameters(const GlopParameters& parameters) {
80  parameters_ = parameters;
81  }
82 
83  // Stats related functions.
84  std::string StatString() const { return stats_.StatString(); }
85 
86  private:
87  // Recomputes the dual edge squared norms from scratch with maximum precision.
88  // The matrix must have been refactorized before because we will do a lot of
89  // inversions. See NeedsBasisRefactorization(). This is checked in debug mode.
90  void ComputeEdgeSquaredNorms();
91 
92  // Computes the vector tau needed to update the norms using a right solve:
93  // B.tau = (u_i)^T, u_i.B = e_i for i = leaving_row.
94  const DenseColumn& ComputeTau(const ScatteredColumn& unit_row_left_inverse);

122 };
