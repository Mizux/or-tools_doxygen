96 // Class used to compute optimal values for dimension cumuls of routes,
97 // minimizing cumul soft lower and upper bound costs, and vehicle span costs of
98 // a route.
99 // In its methods, next_accessor is a callback returning the next node of a
100 // given node on a route.
101 class LocalDimensionCumulOptimizer {
102  public:
103  explicit LocalDimensionCumulOptimizer(const RoutingDimension* dimension);
104  // If feasible, computes the optimal cost of the route performed by a vehicle,
105  // minimizing cumul soft lower and upper bound costs and vehicle span costs,
106  // and stores it in "optimal_cost" (if not null).
107  // Returns true iff the route respects all constraints.
108  bool ComputeRouteCumulCost(int vehicle,
109  const std::function<int64(int64)>& next_accessor,
110  int64* optimal_cost);
111  // Same as ComputeRouteCumulCost, but the cost computed does not contain
112  // the part of the vehicle span cost due to fixed transits.
113  bool ComputeRouteCumulCostWithoutFixedTransits(
114  int vehicle, const std::function<int64(int64)>& next_accessor,
115  int64* optimal_cost_without_transits);
116  // If feasible, computes the optimal cumul values of the route performed by a
117  // vehicle, minimizing cumul soft lower and upper bound costs and vehicle span
118  // costs, stores them in "optimal_cumuls" (if not null), and returns true.
119  // Returns false if the route is not feasible.
120  bool ComputeRouteCumuls(int vehicle,
121  const std::function<int64(int64)>& next_accessor,
122  std::vector<int64>* optimal_cumuls);
123 
124  const RoutingDimension* dimension() const {
125  return optimizer_core_.dimension();
126  }
127 
128  private:
129  std::vector<std::unique_ptr<glop::LPSolver>> lp_solver_;
130  std::vector<std::unique_ptr<glop::LinearProgram>> linear_program_;
131  DimensionCumulOptimizerCore optimizer_core_;
132 };

134 class GlobalDimensionCumulOptimizer {
135  public:
136  explicit GlobalDimensionCumulOptimizer(const RoutingDimension* dimension);
137  // If feasible, computes the optimal cost of the entire model with regards to
138  // the optimizer_core_'s dimension costs, minimizing cumul soft lower/upper
139  // bound costs and vehicle/global span costs, and stores it in "optimal_cost"
140  // (if not null).
141  // Returns true iff all the constraints can be respected.
142  bool ComputeCumulCostWithoutFixedTransits(
143  const std::function<int64(int64)>& next_accessor,
144  int64* optimal_cost_without_transits);
145  // If feasible, computes the optimal cumul values, minimizing cumul soft
146  // lower/upper bound costs and vehicle/global span costs, stores them in
147  // "optimal_cumuls" (if not null), and returns true.
148  // Returns false if the routes are not feasible.
149  bool ComputeCumuls(const std::function<int64(int64)>& next_accessor,
150  std::vector<int64>* optimal_cumuls);
151 
152  // Returns true iff the routes resulting from the next_accessor are feasible
153  // wrt the constraints on the optimizer_core_.dimension()'s cumuls.
154  bool IsFeasible(const std::function<int64(int64)>& next_accessor);
155 
156  // Similar to ComputeCumuls, but also tries to pack the cumul values on all
157  // routes, such that the cost remains the same, the cumuls of route ends are
158  // minimized, and then the cumuls of the starts of the routes are maximized.
159  bool ComputePackedCumuls(const std::function<int64(int64)>& next_accessor,
160  std::vector<int64>* packed_cumuls);
161 
162  const RoutingDimension* dimension() const {
163  return optimizer_core_.dimension();
164  }
165 
166  private:
167  glop::LPSolver lp_solver_;
168  glop::LinearProgram linear_program_;
169  DimensionCumulOptimizerCore optimizer_core_;
170 };
