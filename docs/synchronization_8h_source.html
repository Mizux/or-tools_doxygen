32 // Manages the global best response kept by the solver.
33 // All functions are thread-safe.
34 class SharedResponseManager {
35  public:
36  // If log_updates is true, then all updates to the global "state" will be
37  // logged. This class is responsible for our solver log progress.
38  explicit SharedResponseManager(bool log_updates_, const CpModelProto* proto,
39  const WallTimer* wall_timer);
40 
41  // Returns the current solver response. That is the best known response at the
42  // time of the call with the best feasible solution and objective bounds.
43  //
44  // Note that the solver statistics correspond to the last time a better
45  // solution was found or SetStatsFromModel() was called.
46  CpSolverResponse GetResponse();
47 
48  // Adds a callback that will be called on each new solution (for
49  // statisfiablity problem) or each improving new solution (for an optimization
50  // problem). Returns its id so it can be unregistered if needed.
51  //
52  // Note that currently the class is waiting for the callback to finish before
53  // accepting any new updates. That could be changed if needed.
54  int AddSolutionCallback(
55  std::function<void(const CpSolverResponse&)> callback);
56  void UnregisterCallback(int callback_id);
57 
58  // The "inner" objective is the CpModelProto objective without scaling/offset.
59  // Note that these bound correspond to valid bound for the problem of finding
60  // a strictly better objective than the current one. Thus the lower bound is
61  // always a valid bound for the global problem, but the upper bound is NOT.
62  IntegerValue GetInnerObjectiveLowerBound();
63  IntegerValue GetInnerObjectiveUpperBound();
64 
65  // Updates the inner objective bounds.
66  void UpdateInnerObjectiveBounds(const std::string& worker_info,
67  IntegerValue lb, IntegerValue ub);
68 
69  // Reads the new solution from the response and update our state. For an
70  // optimization problem, we only do something if the solution is strictly
71  // improving.
72  //
73  // TODO(user): Only the follwing fields from response are accessed here, we
74  // might want a tighter API:
75  // - solution_info
76  // - solution
77  // - solution_lower_bounds and solution_upper_bounds.
78  void NewSolution(const CpSolverResponse& response, Model* model);
79 
80  // Changes the solution to reflect the fact that the "improving" problem is
81  // infeasible. This means that if we have a solution, we have proven
82  // optimality, otherwise the global problem is infeasible.
83  //
84  // Note that this shouldn't be called before the solution is actually
85  // reported. We check for this case in NewSolution().
86  void NotifyThatImprovingProblemIsInfeasible(const std::string& worker_info);
87 
88  // Sets the statistics in the response to the one of the solver inside the
89  // given in-memory model. This does nothing if the model is nullptr.
90  //
91  // TODO(user): Also support merging statistics together.
92  void SetStatsFromModel(Model* model);
93 
94  private:
95  void FillObjectiveValuesInBestResponse();
96  void SetStatsFromModelInternal(Model* model);
97 
98  const bool log_updates_;
99  const CpModelProto& model_proto_;
100  const WallTimer& wall_timer_;
101 
102  absl::Mutex mutex_;
103 
104  CpSolverResponse best_response_;
105 
106  int num_solutions_ = 0;
107  int64 inner_objective_lower_bound_ = kint64min;
108  int64 inner_objective_upper_bound_ = kint64max;
109  int64 best_solution_objective_value_ = kint64max;
110 
111  int next_callback_id_ = 0;
112  std::vector<std::pair<int, std::function<void(const CpSolverResponse&)>>>
113  callbacks_;
114 };

116 // This class manages a pool of lower and upper bounds on a set of variables in
117 // a parallel context.
118 class SharedBoundsManager {
119  public:
120  SharedBoundsManager(int num_workers, const CpModelProto& model_proto);
121 
122  // Reports a set of locally improved variable bounds to the shared bounds
123  // manager. The manager will compare these bounds changes against its
124  // global state, and incorporate the improving ones.
125  void ReportPotentialNewBounds(const CpModelProto& model_proto, int worker_id,
126  const std::string& worker_name,
127  const std::vector<int>& variables,
128  const std::vector<int64>& new_lower_bounds,
129  const std::vector<int64>& new_upper_bounds);
130 
131  // When called, returns the set of bounds improvements since
132  // the last time this method was called by the same worker.
133  void GetChangedBounds(int worker_id, std::vector<int>* variables,
134  std::vector<int64>* new_lower_bounds,
135  std::vector<int64>* new_upper_bounds);
136 
137  private:
138  const int num_workers_;
139  const int num_variables_;
140  std::vector<SparseBitset<int64>> changed_variables_per_workers_;
141  std::vector<int64> lower_bounds_;
142  std::vector<int64> upper_bounds_;
143  absl::Mutex mutex_;
144 };
